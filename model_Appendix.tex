\section{Proof of Proposition FB-alloc}

\begin{proof}
Let $\beta$, $\varepsilon(s)$, $\phi_i(\theta,s)$, and $\rho_i(\theta)$ be the Lagrange multipliers on the budget, the capacity, the transfer positivity, and the participation constraints, respectively. The Lagrangian is
\[\label{KKT}\tag{L$^*$}
\begin{aligned}
\mathcal{L} &= \sum_i \mu_i\, \mathbb{E}_{(s,\theta_i)}\!\left[\lambda_i(\theta)\,\theta\, U(q_i,s) - t_i(\theta,s)\right] + \beta\!\left(\sum_i \mu_i \mathbb{E}_{(s,\theta_i)}[t_i(\theta,s)] - I(k)\right) \\
&\quad + \mathbb{E}_s\!\left[\varepsilon(s)\!\left(k - \sum_i \mu_i \mathbb{E}_{\theta_i}[q_i(\theta,s)]\right)\right] + \sum_i  \mathbb{E}_{(s,\theta_i)}\left[ \rho_i(\theta)(\theta U(q_i,s) - t_i(\theta,s))\right] dG_i(\theta) \\
&\quad + \sum_i \mu_i\, \mathbb{E}_{(s,\theta_i)}\!\left[\phi_i(\theta,s)\,t_i(\theta,s)\right],
\end{aligned}
\]
with $\beta, \varepsilon(s), \rho_i(\theta), \phi_i(\theta,s) \geq 0$. The FOCs for each $(i,\theta,s)$ are
\begin{align*}
\frac{\partial \mathcal{L}}{\partial q_i} = 0 \quad &\Rightarrow \quad \left(\mu_i\lambda_i(\theta) + \rho_i(\theta)\right)\theta\, u(q_i,s) = \mu_i\varepsilon(s), \\
\frac{\partial \mathcal{L}}{\partial t_i} = 0 \quad &\Rightarrow \quad \mu_i\beta - \mu_i\lambda_i(\theta) - \rho_i(\theta) + \phi_i(\theta,s) = 0.
\end{align*}
Complementary slackness conditions are
\[
\varepsilon(s)\!\left(k - \sum_i \mu_i \mathbb{E}_{\theta_i}[q_i(\theta,s)]\right) = 0, \quad \beta\!\left(\sum_i \mu_i \mathbb{E}_{s,\theta}[t_i] - I(k)\right) = 0,
\]
\[
\rho_i(\theta)\,\mathbb{E}_s\!\left[\theta U(q_i,s) - t_i(\theta,s)\right] = 0, \quad \phi_i(\theta,s)\,t_i(\theta,s) = 0.
\]

\textit{Step 1: $s$-independence of transfers (WLOG).} The objective, the budget constraint, and the IR constraint all depend on $t_i(\theta,s)$ only through its expectation $\mathbb{E}_s[t_i(\theta,s)]$. Therefore any feasible state-contingent transfer schedule $t_i(\theta,s)$ can be replaced by $\bar{t}_i(\theta) := \mathbb{E}_s[t_i(\theta,s)]$ without altering the objective value or any constraint. Transfer positivity is preserved since $t_i(\theta,s) \geq 0$ for all $s$ implies $\bar{t}_i(\theta) \geq 0$. Hence without loss of generality $t_i^*(\theta,s) = t_i^*(\theta)$, and $\rho_i(\theta)$ and $\phi_i(\theta)$ are also $s$-independent.

\textit{Step 2: structure of optimal transfers.} Suppose two types $(\theta,i)$ and $(\theta',i')$ satisfy $\lambda_i(\theta) < \lambda_{i'}(\theta')$. Assume for contradiction that $t_{i'}(\theta') > 0$ and $t_i(\theta) = 0$. If both participation constraints are slack, then $\rho_i(\theta) = \rho_{i'}(\theta') = 0$ and $\phi_{i'}(\theta') = 0$, so the FOC on $t$ gives $\lambda_{i'}(\theta') = \beta$. But $t_i(\theta) = 0$ with slack participation requires $\phi_i(\theta) > 0$, giving $\lambda_i(\theta) = \beta + \phi_i(\theta)/\mu_i > \beta = \lambda_{i'}(\theta')$, contradicting $\lambda_i(\theta) < \lambda_{i'}(\theta')$. The case where the participation constraint of $(\theta', i')$ binds leads to the same contradiction. Therefore at the optimum, any type with $\lambda_i(\theta) < \lambda^*$ has $t_i^*(\theta) > 0$ and any type with $\lambda_i(\theta) > \lambda^*$ has $t_i^*(\theta) = 0$, where $\lambda^* = \beta$ is the marginal welfare weight pinned by the budget constraint.

\textit{Step 3: optimal quantities.} When capacity is slack, $\varepsilon(s) = 0$ and the FOC on $q$ gives $\theta\, u(q_i^*,s) = 0$, which is \eqref{eq:FB-S}. When capacity binds, the FOC on $q$ gives $\theta\, u(q_i^*,s) = \varepsilon(s)/(\lambda_i(\theta) + \rho_i(\theta)/\mu_i)$. For types with $\lambda_i(\theta) > \lambda^*$: $\rho_i(\theta) = 0$ and $\theta\, u(q_i^*,s) = \varepsilon(s)/\lambda_i(\theta)$. For types with $\lambda_i(\theta) \leq \lambda^*$: from the FOC on $t$, $\lambda_i(\theta) + \rho_i(\theta)/\mu_i = \beta$, so $\theta\, u(q_i^*,s) = \varepsilon(s)/\beta$. This establishes \eqref{eq:FB-T}.

\textit{Step 4: monotonicity of $q_i^*$ in $s$ and uniqueness of $s^*$.} When capacity is slack, totally differentiating $\theta\, u(q_i^*,s) = 0$ gives
\[
\frac{\partial q_i^*(\theta,s)}{\partial s} = -\frac{u_s(q_i^*,s)}{u_q(q_i^*,s)} > 0,
\]
since $u_q < 0$ and $u_s > 0$ by assumption. Hence aggregate demand $\sum_i \mu_i \mathbb{E}_{\theta_i}[q_i^*(\theta,s)]$ is strictly increasing in $s$, which guarantees existence of a unique threshold $s^*$ separating $S^*$ and $T^*$.
\end{proof}

\section{Proof of Proposition CI-welfare}

\subsection{Part i)}
\begin{proof}
We maintain throughout that $B(k)+M(k)\neq 0$, where $B(k)+M(k)$ is the derivative of the binding residual equation with respect to $\beta$ after substituting the optimal allocation; this is the local regularity condition needed to apply the implicit function theorem, while $B(k)+M(k)>0$ delivers the monotone comparative statics used below.

\medskip\noindent\textbf{Corner case: $\beta(k)=0$.} If $\beta(k)=0$ there are no contributing types. On $T^*$, the first-order condition is $\theta\,u(q_i^*(\theta,s),s) = \varepsilon(s)/\lambda_i(\theta)$. Differentiating and substituting into the capacity constraint gives $\frac{\partial\varepsilon(s)}{\partial k} = -1/\overline w^{nc}(s) < 0$ and $\frac{\partial q_i^*(\theta,s)}{\partial k} = w_i^{nc}(\theta,s)/\overline w^{nc}(s) > 0$ on $T^*$. Since non-contributing types pay no transfer:
\[
\frac{\partial CS_i(\theta;k)}{\partial k} = \mathbb{E}_s\!\left[\theta\,u(q_i^*(\theta,s),s)\,\frac{\partial q_i^*(\theta,s)}{\partial k}\,\mathbf{1}_{s\in T^*}\right] \geq 0.
\]

\medskip\noindent\textbf{Interior case: $\beta(k)>0$ with coexistence.} Assume contributing and non-contributing types coexist on a set of peak states of positive probability. On $T^*$, the first-order conditions are
\[
\theta\,u(q_i^*(\theta,s),s) = \frac{\varepsilon(s)}{\beta(k)}, \quad \lambda_i(\theta)\leq\beta(k), \qquad \theta\,u(q_i^*(\theta,s),s) = \frac{\varepsilon(s)}{\lambda_i(\theta)}, \quad \lambda_i(\theta)>\beta(k).
\]
The weights $w_i^c$, $w_i^{nc}$, $\overline w^c(s)$, $\overline w^{nc}(s)$, $\overline w(s)$ and the inframarginal component
\[
m^c(s) := \frac{1}{\overline w(s)}\sum_i\mu_i\,\mathbb{E}_{\theta_i}\!\left[w_i^c(\theta,s)\,\theta\,u(q_i^*(\theta,s),s)\,\mathbf{1}_{\lambda_i(\theta)\leq\beta(k)}\right]
\]
are as defined above. Differentiating the first-order conditions with respect to $k$:
\[
\frac{\partial q_i^*(\theta,s)}{\partial k} = w_i^c(\theta,s)\!\left(-\frac{\partial\varepsilon(s)}{\partial k} + \theta\,u(q_i^*,s)\,\frac{d\beta}{dk}\right), \quad \lambda_i(\theta)\leq\beta(k), \tag{$\partial_k q^c$}\label{dqc}
\]
\[
\frac{\partial q_i^*(\theta,s)}{\partial k} = -\,w_i^{nc}(\theta,s)\,\frac{\partial\varepsilon(s)}{\partial k}, \quad \lambda_i(\theta)>\beta(k). \tag{$\partial_k q^{nc}$}\label{dqnc}
\]
Differentiating the capacity constraint and using continuity of the allocation at each cutoff $\theta_i^*(k)$ --- so that boundary terms cancel --- yields
\[
\frac{\partial\varepsilon(s)}{\partial k} = m^c(s)\,\frac{d\beta}{dk} - \frac{1}{\overline w(s)}. \tag{$\partial_k\varepsilon$}\label{depsk}
\]

\medskip\noindent\textbf{Step 1: sign of $\frac{d\beta}{dk}$.} Since the IR constraint of marginal types binds, $R^M(k)\equiv 0$. Differentiating this identity with respect to $k$, substituting \eqref{dqc} and \eqref{depsk}, and collecting all terms in $\frac{d\beta}{dk}$ on one side yields
\[
\mathbb{E}_s\!\left[\sum_i\mu_i\,\mathbb{E}^w_i[u^c|s]\,\mathbf{1}_{s\in T^*}\right] - I'(k) + \frac{d\beta}{dk}\,B(k) = 0,
\]
where, for convenience of exposition, we write $\mathbb{E}^w_i[u^c|s]$ for the full per-category object defined above --- combining the inframarginal weighted marginal utility and the boundary term from the movement of $\theta_i^*(k)$ --- and
\[
B(k) := \mathbb{E}_s\!\left[\frac{\varepsilon(s)^2}{\beta(k)^2}\frac{\overline w^c(s)\,\overline w^{nc}(s)}{\overline w(s)}\,\mathbf{1}_{s\in T^*}\right] > 0.
\]
Solving for $\frac{d\beta}{dk}$:
\[
\frac{d\beta}{dk} = \frac{I'(k) - \mathbb{E}_s\!\left[\sum_i\mu_i\,\mathbb{E}^w_i[u^c|s]\,\mathbf{1}_{s\in T^*}\right]}{B(k)}.
\]
Since $B(k)>0$, condition \eqref{Ui*} implies $\frac{d\beta}{dk}\leq 0$.

\medskip\noindent\textbf{Step 2: surplus of non-contributing types.} Since $\frac{d\beta}{dk}\leq 0$ and $m^c(s)>0$, equation \eqref{depsk} gives $\frac{\partial\varepsilon(s)}{\partial k}<0$ for all $s\in T^*$. From \eqref{dqnc}: $\frac{\partial q_i^*(\theta,s)}{\partial k} = -w_i^{nc}(\theta,s)\frac{\partial\varepsilon(s)}{\partial k} > 0$ for every non-contributing type. Since $t_i^*(\theta,s)=0$:
\[
\frac{\partial CS_i(\theta;k)}{\partial k} = \mathbb{E}_s\!\left[\theta\,u(q_i^*(\theta,s),s)\,\frac{\partial q_i^*(\theta,s)}{\partial k}\,\mathbf{1}_{s\in T^*}\right] \geq 0.
\]

\medskip\noindent\textbf{Corner case: all types strictly contributing.} If all types satisfy $\lambda_i(\theta)<\beta(k)$, there are no non-contributing types and the statement is vacuous.
\end{proof}

\subsection{Part ii)}

\begin{proof}
From the Lagrangian \eqref{P-CI}, the FOC with respect to $k$ is
\[
-\beta\, I'(k) + \E_s[\varepsilon(s)\,\mathbf{1}_{T^*}(s)] = 0,
\]
where we used $\varepsilon(s)=0$ for all $s\in S^*$, so that
\begin{equation}\label{eq:FOC-k}
\beta\, I'(k) = \E_s[\varepsilon(s)\,\mathbf{1}_{T^*}(s)].
\end{equation}

\textit{Step 1: contributing types.} From \eqref{eq:FB-T}, for every type with $\lambda_i(\theta)\leq\lambda^*$ and every $s\in T^*$:
\[
\theta\, u(q_i^*(\theta,s),s) = \frac{\varepsilon(s)}{\beta}.
\]
Multiplying by $\mu_i$, integrating over contributing types, and taking the expectation over $s\in T^*$:
\begin{equation}\label{eq:step1}
\E_s\!\left[\sum_i \mu_i\,\E_{\theta_i}\!\left[\theta\, u(q_i^*,s)\,\mathbf{1}_{\lambda_i(\theta)\leq\lambda^*}\right]\mathbf{1}_{T^*}(s)\right] = \frac{\E_s[\varepsilon(s)\mathbf{1}_{T^*}(s)]}{\beta}\,\mu^c = I'(k)\,\mu^c,
\end{equation}
where the last equality uses \eqref{eq:FOC-k}. Dividing by $\mu^c$ gives \eqref{eq:FB-k1}.


\begin{lemma}\label{lem:CI_interior_k}
If $I(\cdot)$ is convex, then $V^{CI}$ is concave in $k$.
\end{lemma}

\begin{proof}
When $I(\cdot)$ is convex, the complete-information problem is a concave maximization problem with a convex feasible set: the objective is concave in the choice variables, the capacity constraint is affine in $k$, and the budget constraint is convex because $I(k)$ is convex. Hence the value function $V^{CI}$ is concave in $k$.
\end{proof}


\end{proof}

\section{Proof of Lemma Transfer}

\begin{proof}

For each category $i$, define
\[
B_i(\theta,s)
:=
\theta U\!\left(q_i^\times(\theta,s),s\right)
- \underline\theta_i U\!\left(q_i^\times(\underline\theta_i,s),s\right)
- \int_{\underline{\theta}_i}^{\theta}
U\!\left(q_i^\times(\tilde\theta,s),s\right)\,d\tilde\theta,
\]
and let
\[
\overline{B}_i(\theta)
:=
\mathbb{E}_s[B_i(\theta,s)].
\]

Start with the following individual transfer for each consumer of type $\theta$ in category $i$:
\begin{align*}
t_i^\times(\theta,s)
&=
\tau_i
+ \overline{B}_i(\theta)
- \sum_{j\neq i}  \frac{\mu_j}{1-\mu_j}\,
   \mathbb{E}_{\theta_j}[\overline{B}_j(\theta_j)] + \Big(I(k) -I_i+\sum_{j\neq i}  \frac{\mu_j}{1-\mu_j}I_j\Big),
\end{align*}
where
\[
\tau_i
=
T_{i0}
+\sum_{j\neq i}  \frac{\mu_j}{1-\mu_j}\,
   \mathbb{E}_{\theta_j}[\overline{B}_j(\theta_j)]
- \Big(I(k) -I_i+\sum_{j\neq i}  \frac{\mu_j}{1-\mu_j}I_j\Big).
\]

Taking expectations over $\theta$, and noting that only $\overline{B}_i(\theta)$ depends on $\theta$, yields
\begin{align*}
\mathbb{E}_{\theta_i}[t_i^\times(\theta,s)]
&=
\tau_i
+ \mathbb{E}_{\theta_i}[\overline{B}_i(\theta)]
- \sum_{j\neq i}  \frac{\mu_j}{1-\mu_j}\,
   \mathbb{E}_{\theta_j}[\overline{B}_j(\theta_j)] 
+ \Big(I(k) -I_i+\sum_{j\neq i}  \frac{\mu_j}{1-\mu_j}I_j\Big).
\end{align*}

Multiplying by the size $\mu_i$ of category $i$, summing over all categories, and using $\sum_{i=1}^n\mu_i = 1$, we obtain
\begin{align*}
\sum_{i=1}^n \mu_i\mathbb{E}_{\theta_i}[t_i^\times(\theta,s)]
&=
\sum_{i=1}^n \mu_i \tau_i
+ \sum_{i=1}^n\mu_i\,\mathbb{E}_{\theta_i}[\overline{B}_i(\theta)]-\sum_{i=1}^n\mu_i\sum_{j\neq i} \frac{\mu_j}{1-\mu_j}\,
   \mathbb{E}_{\theta_j}[\overline{B}_j(\theta_j)]
\\
&\quad
+ I(k)
-\sum_{i=1}^n\mu_iI_i
+\sum_{i=1}^n\mu_i\sum_{j\neq i}  \frac{\mu_j}{1-\mu_j}I_j.
\end{align*}

The second and third terms cancel, and the fifth and sixth terms also cancel, so that
\[
\sum_{i=1}^n \mu_i\mathbb{E}_{\theta_i}[t_i^\times(\theta,s)]
=
\sum_{i=1}^n \mu_i \tau_i + I(k).
\]
Therefore, an allocation satisfies the budget constraint if and only if $\sum_{i=1}^n \mu_i \tau_i \geq 0$.

Next, substitute the expression for $\tau_i$:
\begin{align*}
\sum_{i=1}^n \mu_i \tau_i
&=
\sum_{i=1}^n \mu_i \left(
T_{i0}
+  \sum_{j\neq i}  \frac{\mu_j}{1-\mu_j}\,
     \mathbb{E}_{\theta_j}[\overline{B}_j(\theta_j)]
\right)
- I(k) \\
&=
\sum_{i=1}^n \mu_i \left(
T_{i0}
+  \mathbb{E}_{\theta_i}[\overline{B}_i(\theta)]
\right)
- I(k).
\end{align*}

By Fubini's theorem and the usual hazard-rate argument, the expectation of $\overline{B}_i(\theta)$ can be written in terms of the virtual utility
\[
U(q^\times_i(\theta,s),s) J_i(\theta)
:= U\!\left(q_i^\times(\theta,s),s\right)(\theta -
\gamma_i(\theta)\,),
\]
as
\[
\mathbb{E}_{\theta_i}[\overline{B}_i(\theta)]
=
\mathbb{E}_{(s,\theta_i)}\left[
U(q^\times_i(\theta,s),s) J_i(\theta)
- \underline{\theta_i} U\!\left(q_i^\times(\underline{\theta_i},s),s\right)
\right].
\]
Hence
\[
\sum_{i=1}^n \mu_i \tau_i
=
\sum_{i=1}^n \mu_i \mathbb{E}_{(s,\theta_i)}\left[
T_{i0} + U(q^\times_i(\theta,s),s) J_i(\theta) - \underline{\theta_i} U\!\left(q_i^\times(\underline{\theta_i},s),s\right)
\right]
- I(k).
\]

Finally, replacing $\tau_i$ in $t_i^\times(\theta,s)$ yields
\[
t_i^\times(\theta,s)
=
T_{i0}
+ \overline{B}_i(\theta)
=
T_{i0}
+ \mathbb{E}_s\left[
\theta U\!\left(q_i^\times(\theta,s),s\right)
- \underline{\theta_i}U\!\left(q_i^\times(\underline{\theta_i},s),s\right)
- \int_{\underline{\theta}_i}^{\theta}
U\!\left(q_i^\times(\tilde\theta,s),s\right)\, d\tilde\theta
\right].
\]

Thus
\begin{align*}
\mathbb{E}_s\big[\theta U(q_i^\times(\theta,s),s) - t_i^\times(\theta,s)\big]
&=
\mathbb{E}_s\big[\underline{\theta_i}U(q_i^\times(\underline{\theta_i},s),s)\big]
- T_{i0}
\\
&\quad
+ \int_{\underline{\theta}_i}^{\theta}
\mathbb{E}_s\big[U(q_i^\times(\tilde\theta,s),s)\big]\,d\tilde\theta.
\end{align*}
Defining
\[
\E\underline{CS}_i^\times
:=
\mathbb{E}_s\big[\underline{\theta_i}U(q_i^\times(\underline{\theta_i},s),s)\big]
- T_{i0},
\]
we get
\[
\mathbb{E}_s[\theta U(q_i^\times(\theta,s),s) - t_i^\times(\theta,s)]
=
\E\underline{CS}_i^\times
+ \int_{\underline{\theta}_i}^{\theta}
\mathbb{E}_s\big[U(q_i^\times(\tilde\theta,s),s)\big]\,d\tilde\theta,
\]
which is the usual envelope representation and therefore satisfies the IC condition. Moreover, substituting $\underline{\theta_i}U(q_i^\times(\underline{\theta_i},s),s) = \E\underline{CS}_i^\times + T_{i0}$ into the expression for $\sum_i \mu_i \tau_i$ gives the claimed transformed budget constraint
\[
\sum_{i=1}^n \mu_i \tau_i
=
\sum_{i=1}^n \mu_i \mathbb{E}_{(s,\theta_i)}\big[ U(q^\times_i(\theta,s),s) J_i(\theta) - \E\underline{CS}_i^\times\big]
- I(k).
\]

Multiplying the utility of a consumer with type $\theta$ from category $i$ and taking the expectation over the whole category yields

\[
\mathbb{E}_{(s,\theta_i)}[\lambda_i(\theta) \;( \;\theta U(q_i^\times(\theta,s),s) - t_i^\times(\theta,s)\;)\;]
=
\tilde{\lambda}_i \; \E\underline{CS}_i^\times
+ \mathbb{E}_{\theta_i}[\lambda_i(\theta)\int_{\underline{\theta}_i}^{\theta}
\mathbb{E}_s\big[U(q_i^\times(\tilde\theta,s),s)\big]\,d\tilde\theta],
\]

We apply Fubini's theorem to the second term:

\begin{align*}
\mathbb{E}_{\theta_i}\!\left[\lambda_i(\theta)\int_{\underline{\theta}_i}^{\theta}
\mathbb{E}_s[U(q_i^\times(\tilde\theta,s),s)]\,d\tilde\theta\right]
&= \int_{\underline{\theta}_i}^{\overline{\theta}_i} \lambda_i(\theta) g_i(\theta) 
   \left(\int_{\underline{\theta}_i}^{\theta} \mathbb{E}_s[U(q_i^\times(\tilde\theta,s),s)]\,d\tilde\theta\right) d\theta \\
&= \int_{\underline{\theta}_i}^{\overline{\theta}_i} \mathbb{E}_s[U(q_i^\times(\tilde\theta,s),s)] 
   \left(\int_{\tilde\theta}^{\overline{\theta}_i} \lambda_i(\theta) g_i(\theta)\,d\theta\right) d\tilde\theta \\
&= \int_{\underline{\theta}_i}^{\overline{\theta}_i} \mathbb{E}_s[U(q_i^\times(\tilde\theta,s),s)] 
    \gamma_i(\theta) \E[ \ \lambda_i(\tilde \theta) \ | \ \tilde \theta \ \ge \ \theta \ ] d\tilde\theta \\
&= \mathbb{E}_{(s,\theta_i)}\!\left[\Lambda_i(\theta)\, U(q_i^\times(\theta,s),s)\right].
\end{align*}

This yields the desired expression:

\[ \tilde{\lambda}_i \;\E\underline{CS}_i^\times 
      + \E_{(s,\theta_i)} \big[\; \Lambda_i(\theta)\;  U(q_i^\times(\theta,s),s)\; \big] \]

\end{proof}

\section{Proof of Lemma FOC Without Ironing}
\begin{proof}
Consider the within-category problem
\[
\max_{q_i^\times(\theta,s)} 
\mathbb{E}_{(s,\theta_i)}\!\left[\Gamma_i(\theta)\,U\!\left(q_i^\times(\theta,s),s\right)\right]
\]
subject to
\[
\mathbb{E}_{(s,\theta_i)}\!\left[U\!\left(q_i^\times(\theta,s),s\right)J_i(\theta)\right]-I_i \ge 0,
\qquad
\mathbb{E}_{\theta_i}\!\left[q_i^\times(\theta,s)\right]\le k_i
\quad \forall s.
\]

Let \(\beta_i^\times \ge 0\) and \(\varepsilon_i^\times(s)\ge 0\) denote the multipliers on \eqref{Rmdi-def} and \eqref{Kmdi-def}, respectively. The associated Lagrangian is
\[
\mathcal L
=
\mathbb{E}_{(s,\theta_i)}\!\left[
\left(\Gamma_i(\theta)+\beta_i^\times J_i(\theta)\right)
U\!\left(q_i^\times(\theta,s),s\right)
\right]
-
\mathbb{E}_s\!\left[
\varepsilon_i^\times(s)
\left(\mathbb{E}_{\theta_i}[q_i^\times(\theta,s)]-k_i\right)
\right]
-
\beta_i^\times I_i.
\]
Hence the pointwise first-order condition is
\[
u\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta)-\varepsilon_i^\times(s)=0,
\qquad
\text{where }
\mathcal G_i(\theta):=\Gamma_i(\theta)+\beta_i^\times J_i(\theta).
\]

We first characterize the states in which the capacity constraint is slack. If \(\mathbb{E}_{\theta_i}[q_i^\times(\theta,s)]<k_i\), complementary slackness implies \(\varepsilon_i^\times(s)=0\). Since Assumption~\ref{ass:interior} gives \(\mathcal G_i(\theta)>0\) for all \(\theta\), the first-order condition reduces to
\[
u\!\left(q_i^\times(\theta,s),s\right)=0.
\]
By strict concavity of \(U(\cdot,s)\), this equation has a unique solution, denoted \(q_i^0(s)\). In particular, the off-peak allocation is independent of \(\theta\), so it automatically satisfies monotonicity.

Moreover, implicit differentiation of \(u(q_i^0(s),s)=0\) yields
\[
\frac{\partial q_i^0(s)}{\partial s}
=
-\frac{u_s\!\left(q_i^0(s),s\right)}{u_q\!\left(q_i^0(s),s\right)}
\ge 0,
\]
since \(u_s\ge 0\) and \(u_q<0\). Therefore the unconstrained aggregate demand $Q_i^0(s):=\mathbb{E}_{\theta_i}[q_i^0(s)]$ is nondecreasing in \(s\). It follows that the set of states in which the capacity constraint is slack is an interval of the form \(S_i^\times=[\underline s,s_i^\times)\), while the set of binding states is \(T_i^\times=[s_i^\times,\overline s]\), where \(s_i^\times\) is defined by $\mathbb{E}_{\theta_i}\!\left[q_i^\times(\theta,s_i^\times)\right]=k_i$, with the obvious boundary cases if one of the two sets is empty.

Now consider \(s\in T_i^\times\). For such states, the allocation solves the state-by-state problem with binding aggregate quantity constraint, and the first-order condition is exactly $u\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta) = \varepsilon_i^\times(s)$. Because \(u_q<0\) and \(\mathcal G_i(\theta)>0\), the objective is strictly concave in \(q_i^\times(\theta,s)\) pointwise, so this first-order condition characterizes a unique allocation for each \((\theta,s)\).

Finally, Assumption~\ref{ass:interior} also imposes that \(\mathcal G_i(\theta)\) is nondecreasing. Differentiating the first-order condition with respect to \(\theta\) gives
\[
\frac{\partial q_i^\times(\theta,s)}{\partial \theta}
=
-
\frac{u\!\left(q_i^\times(\theta,s),s\right)}{u_q\!\left(q_i^\times(\theta,s),s\right)}
\frac{\mathcal G_i'(\theta)}{\mathcal G_i(\theta)}
\ge 0,
\]
so the monotonicity constraint is satisfied. This proves part (i).

For part (ii), if \eqref{Rmdi-def} is slack at \((k_i,I_i)\), complementary slackness implies \(\beta_i^\times=0\). Hence $\mathcal G_i(\theta)=\Gamma_i(\theta)+\beta_i^\times J_i(\theta)\equiv \Gamma_i(\theta)$. This proves part (ii).
\end{proof}

\section{Proof of Single crossing}
\begin{proof}
For each binding state $s\in T_i^\times(k_i)$ and each $k_i$, define
\[
\mathsf H_i(k_i,s):=\mathbb E_i^w[uJ\mid s],
\qquad
\mathcal S_i(k_i):=\int_{s_i^\times(k_i)}^{\overline s}\mathsf H_i(k_i,s)\,f(s)\,ds.
\]

For every binding state $s\in T_i^\times(k_i)$, define the probability measure
\[
dP_{k_i,s}(\theta):=
\frac{R_q(\theta,k_i,s)\,g_i(\theta)}
{\mathbb E_{\theta_i}[R_q(\theta,k_i,s)]}\,d\theta.
\]
Since
\[
R_\theta(\theta,k_i)=\frac{J_i(\theta)}{\mathcal G_i(\theta,k_i)},
\]
we can write
\[
\mathsf H_i(k_i,s)
=
\frac{\mathbb E_{\theta_i}\!\left[R_q(\theta,k_i,s)\,R_\theta(\theta,k_i)\right]}
{\mathbb E_{\theta_i}\!\left[R_q(\theta,k_i,s)\right]}
=
\int_{\Theta_i} R_\theta(\theta,k_i)\,dP_{k_i,s}(\theta).
\]

Differentiating $\mathsf H_i(k_i,s)$ with respect to $k_i$ gives
\[
\frac{\partial \mathsf H_i(k_i,s)}{\partial k_i}
=
M_i(k_i,s)+D_i(k_i,s),
\]
where
\[
M_i(k_i,s)
:=
\left.
\frac{\partial}{\partial t}
\int_{\Theta_i}R_\theta(\theta,k_i)\,dP_{t,s}(\theta)
\right|_{t=k_i}
\]
is the pure measure-shift term, and
\[
D_i(k_i,s)
:=
\int_{\Theta_i}
\frac{\partial R_\theta(\theta,k_i)}{\partial k_i}\,dP_{k_i,s}(\theta)
\]
is the direct effect through the $k_i$-dependence of $R_\theta$.

By the log-monotone-differences assumption on $R_q$, the family $\{P_{k_i,s}\}_{k_i}$ is ordered by monotone likelihood ratio for each binding state $s$. Hence, since $\theta\mapsto R_\theta(\theta,k_i)$ is monotone for each $k_i$,
\[
M_i(k_i,s)\ge 0
\]
in the two cases
\[
R_\theta(\cdot,k_i)\ \text{increasing and }R_q\ \text{log-supermodular},
\qquad
R_\theta(\cdot,k_i)\ \text{decreasing and }R_q\ \text{log-submodular},
\]
while
\[
M_i(k_i,s)\le 0
\]
in the two opposite cases.

Next, since
\[
R_\theta(\theta,k_i)=\frac{J_i(\theta)}{\mathcal G_i(\theta,k_i)},
\qquad
\mathcal G_i(\theta,k_i)=\Gamma_i(\theta)+J_i(\theta)\beta_i^\times(k_i),
\]
we have
\[
\frac{\partial R_\theta(\theta,k_i)}{\partial k_i}
=
-\beta_i^{\times\prime}(k_i)\,
\frac{J_i(\theta)^2}{\mathcal G_i(\theta,k_i)^2}.
\]
Therefore
\[
D_i(k_i,s)
=
-\beta_i^{\times\prime}(k_i)\,\mathcal Q_i(k_i,s),
\]
where
\[
\mathcal Q_i(k_i,s)
:=
\int_{\Theta_i}
\frac{J_i(\theta)^2}{\mathcal G_i(\theta,k_i)^2}\,dP_{k_i,s}(\theta)
>0.
\]
Thus
\[
\frac{\partial \mathsf H_i(k_i,s)}{\partial k_i}
=
M_i(k_i,s)-\beta_i^{\times\prime}(k_i)\,\mathcal Q_i(k_i,s).
\]

We now relate $\beta_i^{\times\prime}(k_i)$ to $\mathcal S_i(k_i)$. On the IR-constrained region, define
\[
\Phi_i(k_i,\beta)
:=
\mathbb E_s\!\left[
\mathbb E_{\theta_i}\!\left[
U\!\left(q_i^\times(\theta,s;k_i,\beta),s\right)J_i(\theta)
\right]
\right]
-I_i.
\]
By construction,
\[
\Phi_i\!\left(k_i,\beta_i^\times(k_i)\right)=0.
\]

Holding $\beta$ fixed, the standard comparative-statics identity gives, for every binding state $s$,
\[
\frac{\partial q_i^\times(\theta,s;k_i,\beta)}{\partial k_i}
=
\frac{w_i^\times(\theta,s)}{\bar w_i^\times(s)}.
\]
Hence
\[
\partial_{k_i}\Phi_i(k_i,\beta)
=
\mathbb E_s\!\left[
\mathbf 1_{\{s\in T_i^\times(k_i)\}}
\,
\mathbb E_i^w[uJ\mid s]
\right]
=
\mathcal S_i(k_i),
\]
where the moving-threshold term vanishes because
\[
u\!\left(q_i^\times(\theta,s_i^\times(k_i)),s_i^\times(k_i)\right)=0
\qquad
\text{for every }\theta.
\]

Holding $k_i$ fixed, differentiating the pointwise FOC
\[
u\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta,k_i)
=
\varepsilon_i^\times(s)
\]
with respect to $\beta$, using $\partial_\beta\mathcal G_i(\theta,k_i)=J_i(\theta)$ and imposing
\[
\mathbb E_{\theta_i}\!\left[\partial_\beta q_i^\times(\theta,s)\right]=0,
\]
yields
\[
\frac{\partial \varepsilon_i^\times(s)}{\partial \beta}
=
\mathbb E_i^w[uJ\mid s]
=
\mathsf H_i(k_i,s)
\]
and
\[
\frac{\partial q_i^\times(\theta,s)}{\partial \beta}
=
w_i^\times(\theta,s)
\left(
u\!\left(q_i^\times(\theta,s),s\right)J_i(\theta)
-
\mathsf H_i(k_i,s)
\right).
\]
Therefore
\[
\partial_\beta \Phi_i(k_i,\beta)
=
\mathbb E_s\!\left[
\mathbf 1_{\{s\in T_i^\times(k_i)\}}
\,
\bar w_i^\times(s)\,
\mathrm{Var}_i^w(uJ\mid s)
\right]
=:
\mathcal V_i(k_i),
\]
with $\mathcal V_i(k_i)>0$ away from knife-edge cases. Differentiating
\[
\Phi_i\!\left(k_i,\beta_i^\times(k_i)\right)=0
\]
with respect to $k_i$ gives
\[
0
=
\mathcal S_i(k_i)+\mathcal V_i(k_i)\,\beta_i^{\times\prime}(k_i),
\]
hence
\[
\beta_i^{\times\prime}(k_i)
=
-\frac{\mathcal S_i(k_i)}{\mathcal V_i(k_i)}.
\]

Finally, differentiate
\[
\mathcal S_i(k_i)
=
\int_{s_i^\times(k_i)}^{\overline s}\mathsf H_i(k_i,s)\,f(s)\,ds.
\]
Using Leibniz' rule,
\[
\mathcal S_i'(k_i)
=
-
\mathsf H_i\!\left(k_i,s_i^\times(k_i)\right)
f\!\left(s_i^\times(k_i)\right)
s_i^{\times\prime}(k_i)
+
\int_{s_i^\times(k_i)}^{\overline s}
\frac{\partial \mathsf H_i(k_i,s)}{\partial k_i}\,f(s)\,ds.
\]
At the threshold state $s_i^\times(k_i)$ we have
\[
u\!\left(q_i^\times(\theta,s_i^\times(k_i)),s_i^\times(k_i)\right)=0
\qquad
\text{for every }\theta,
\]
so $\mathsf H_i\!\left(k_i,s_i^\times(k_i)\right)=0$. Hence the boundary term vanishes and
\[
\mathcal S_i'(k_i)
=
\overline M_i(k_i)-\beta_i^{\times\prime}(k_i)\,\overline{\mathcal Q}_i(k_i),
\]
where
\[
\overline M_i(k_i)
:=
\int_{s_i^\times(k_i)}^{\overline s}M_i(k_i,s)\,f(s)\,ds,
\qquad
\overline{\mathcal Q}_i(k_i)
:=
\int_{s_i^\times(k_i)}^{\overline s}\mathcal Q_i(k_i,s)\,f(s)\,ds
>0.
\]

Now let $k_i^0$ satisfy $\mathcal S_i(k_i^0)=0$. Then
\[
\beta_i^{\times\prime}(k_i^0)=0,
\]
so
\[
\mathcal S_i'(k_i^0)=\overline M_i(k_i^0).
\]
Since each $M_i(k_i^0,s)$ has the same sign across binding states, $\overline M_i(k_i^0)$ has that same sign. Therefore every sign-changing zero of $\mathcal S_i$ is crossed in the same direction:
\[
\mathcal S_i'(k_i^0)>0
\]
when
\[
R_\theta(\cdot,k_i)\ \text{increasing and }R_q\ \text{log-supermodular},
\qquad
R_\theta(\cdot,k_i)\ \text{decreasing and }R_q\ \text{log-submodular},
\]
so any sign change is from $-$ to $+$; and
\[
\mathcal S_i'(k_i^0)<0
\]
in the two opposite cases, so any sign change is from $+$ to $-$.

Thus the map
\[
k_i\longmapsto \mathcal S_i(k_i)
=
\mathbb E_s\!\left[
\mathbb E_i^w[uJ\mid s]\,
\mathbf 1_{\{s\in T_i^\times(k_i)\}}
\right]
\]
is single-crossing in $k_i$, with crossing direction pinned down by the combination of monotonicities stated in the lemma.
\end{proof}

\section{Proof of Theorem~\ref{thm:welfare}}
\begin{proof}
Define $\mathcal W_i(\theta) := \partial \mathbb E CS_i^\times(\theta)/\partial k_i$. From the envelope representation \eqref{ICmd-def},
\[\mathcal W_i(\theta) = \mathcal W_i(\underline\theta_i) + \int_{\underline\theta_i}^{\theta}\mathcal W_i'(\tilde\theta)\,d\tilde\theta, \qquad \mathcal W_i'(\theta) = \mathbb E_s\!\left[u\!\left(q_i^\times(\theta,s),s\right)\frac{\partial q_i^\times(\theta,s)}{\partial k_i}\mathbf 1_{\{s\in T_i^\times\}}\right].\]
For each binding state $s\in T_i^\times$, differentiating the pointwise FOC \eqref{KKT-qMD} with respect to $k_i$ and imposing the binding capacity constraint \eqref{Kmdi-def} gives
\[\frac{\partial q_i^\times(\theta,s)}{\partial k_i} = \frac{w_i^\times(\theta,s)}{\bar w_i^\times(s)} - w_i^\times(\theta,s)\,\mathsf H_i(k_i,s)\,\beta_i^{\times\prime}(k_i) + w_i^\times(\theta,s)\,u\!\left(q_i^\times(\theta,s),s\right)J_i(\theta)\,\beta_i^{\times\prime}(k_i),\]
where $w_i^\times$, $\bar w_i^\times$, and $\mathsf H_i$ are as defined in the preliminary step of Lemma~\ref{lem:SC-dir}.

\medskip
\noindent\textbf{IR-unconstrained allocation.} Since $\beta_i^\times = 0$, we have $\beta_i^{\times\prime}(k_i) = 0$ and $\mathcal G_i(\theta)\equiv\Gamma_i(\theta)$. The expression for $\partial q_i^\times/\partial k_i$ reduces to $w_i^\times(\theta,s)/\bar w_i^\times(s)$, so
\[\mathcal W_i'(\theta) = \mathbb E_s\!\left[\frac{u(q_i^\times(\theta,s),s)\,w_i^\times(\theta,s)}{\bar w_i^\times(s)}\mathbf 1_{\{s\in T_i^\times\}}\right] = \mathbb E_s\!\left[\frac{R_q(\theta,s)}{\Gamma_i(\theta)\,\bar w_i^\times(s)}\mathbf 1_{\{s\in T_i^\times\}}\right] > 0,\]
since $R_q > 0$, $\Gamma_i(\theta) > 0$, and $\bar w_i^\times(s) > 0$ by Assumption~\ref{ass:interior}. Hence $\mathcal W_i$ is strictly increasing on $[\underline\theta_i,\overline\theta_i]$. By the preliminary step of Lemma~\ref{lem:SC-dir}, $\mathcal W_i(\underline\theta_i) = \partial \mathbb E\underline{CS}_i^\times/\partial k_i$ has the sign of \eqref{Eujmd}. If \eqref{Eujmd} is positive, $\mathcal W_i(\underline\theta_i)>0$ and strict monotonicity implies every type gains. If \eqref{Eujmd} is negative, $\mathcal W_i(\underline\theta_i)<0$ and, by strict monotonicity, $\mathcal W_i$ crosses zero at most once; if and only if $\mathcal W_i(\overline\theta_i)>0$, a unique cutoff $\tilde\theta_i$ exists and types below lose while types above gain.

\medskip
\noindent\textbf{IR-constrained allocation.}

\noindent\textit{Step 1: initial condition.} Since $\mathbb E\underline{CS}_i^\times = 0$ holds identically along the IR-constrained region by definition, its total derivative with respect to $k_i$ is zero. By the envelope representation \eqref{ICmd-def}, this total derivative equals $\mathcal W_i(\underline\theta_i)$, so $\mathcal W_i(\underline\theta_i) = 0$.

\noindent\textit{Step 2: shape of $\mathcal W_i$.} Under Assumption~\ref{ass:mult}, $u(q,s)=\kappa(q)\zeta(s)$ so the ratio $R_q(\theta) := -\zeta(q_i^\times(\theta,s))/\zeta_q(q_i^\times(\theta,s))$ is $s$-independent on the binding region by Lemma~\ref{lem:statewise-zero-mult}. Using the first-order condition \eqref{KKT-qMD}, we have $u(q_i^\times(\theta,s),s)\,w_i^\times(\theta,s) = R_q(\theta)/\mathcal G_i(\theta)$ and $u(q_i^\times(\theta,s),s)\,J_i(\theta) = \varepsilon_i^\times(s)\,R_\theta(\theta)$. Substituting into $\mathcal W_i'(\theta)$ yields
\[\mathcal W_i'(\theta) = \frac{R_q(\theta)}{\mathcal G_i(\theta)}\left(\mathcal A_i - \beta_i^{\times\prime}(k_i)\mathcal B_i + \beta_i^{\times\prime}(k_i)\mathcal C_i R_\theta(\theta)\right),\]
where $\mathcal A_i := \mathbb E_s[\bar w_i^\times(s)^{-1}\mathbf 1_{\{s\in T_i^\times\}}]>0$, $\mathcal B_i := \mathbb E_s[\mathsf H_i(k_i,s)\mathbf 1_{\{s\in T_i^\times\}}] = \mathcal S_i(k_i)$, and $\mathcal C_i := \mathbb E_s[\varepsilon_i^\times(s)\mathbf 1_{\{s\in T_i^\times\}}]>0$. Since $\beta_i^{\times\prime}(k_i) = -\mathcal S_i(k_i)/\mathcal V_i(k_i)$ from the preliminary step of Lemma~\ref{lem:SC-dir}, the term $-\beta_i^{\times\prime}(k_i)\mathcal B_i = \mathcal B_i^2/\mathcal V_i(k_i)\ge 0$, so the constant part of the bracket is strictly positive. Since $R_q(\theta)/\mathcal G_i(\theta)>0$, the monotonicity of $\mathcal W_i'(\theta)$ is governed entirely by the term $\beta_i^{\times\prime}(k_i)\mathcal C_i R_\theta(\theta)$: if $R_\theta(\theta)$ is monotone, $\mathcal W_i'(\theta)$ is monotone in $\theta$, so $\mathcal W_i$ is either concave or convex. The sign of $\beta_i^{\times\prime}(k_i)$ is opposite to the sign of \eqref{Eujmd}, so if $R_\theta(\theta)$ is increasing, $\mathcal W_i$ is concave when \eqref{Eujmd} is positive and convex when \eqref{Eujmd} is negative; if $R_\theta(\theta)$ is decreasing, these conclusions are reversed.

\noindent\textit{Step 3: existence and direction of the second cutoff.} Since $\mathcal W_i(\underline\theta_i)=0$, a concave $\mathcal W_i$ crosses zero at most once in the interior. If $\mathcal W_i'(\underline\theta_i)>0$, the function initially rises from zero, eventually returns toward zero, and crosses it at a unique interior cutoff $\tilde\theta_i\in(\underline\theta_i,\overline\theta_i)$; types below $\tilde\theta_i$ gain and types above lose in case~(i), and the reverse in case~(ii). If $\mathcal W_i'(\underline\theta_i)\le 0$, no interior cutoff exists and either all types lose or all types gain. The convex case is symmetric, with concavity and convexity interchanged, and all welfare inequalities reversed. If $R_\theta(\theta)$ is decreasing, concavity and convexity are swapped and all inequalities reverse throughout.
\end{proof}

\section{Proof of Proposition Region}
\begin{proof}
Let
\[
Y_i(k_i):=\mathbb E\underline{CS}_i^\times(k_i).
\]
By the preliminary step of Lemma~\ref{lem:SC-dir},
\[
Y_i'(k_i)
=
\frac{\partial \mathbb E\underline{CS}_i^\times}{\partial k_i}
=
\mathcal S_i(k_i).
\]
Under the conditions of Lemma~\ref{lem:SC-dir}, the map $k_i\mapsto \mathcal S_i(k_i)$ is single-crossing. Hence $Y_i'$ is monotone and crosses zero at most once in a given direction.

When \eqref{Eujmd} crosses from $+$ to $-$, $Y_i'$ is decreasing and crosses zero at most once from above, so $Y_i$ is single-peaked on $[0,k_i^+]$. When \eqref{Eujmd} crosses from $-$ to $+$, $Y_i'$ is increasing and $Y_i$ is single-dipped. At $k_i=0$ the capacity constraint binds at zero, so $q_i^\times(\theta,s)=0$ for all $(\theta,s)$, giving
\[
Y_i(0)=-I_i<0
\qquad
\text{for any }I_i>0.
\]

\noindent\textit{Case: \eqref{Eujmd} crosses from $+$ to $-$.}
Suppose first that $Y_i(k_i^+)\ge 0$. Since $Y_i(0)<0$ and $Y_i(k_i^+)\ge 0$, continuity gives at least one zero in $(0,k_i^+]$. Single-peakedness implies that this zero is unique, defining $\tilde k_i$, and
\[
Y_i(k_i)\ge 0
\Longleftrightarrow
k_i\in[\tilde k_i,k_i^+].
\]

This proves part~(i).

Suppose instead that $Y_i(k_i^+)<0$. Then $Y_i$ is negative at both endpoints. If the peak value is negative, then $Y_i<0$ throughout. If the peak value is nonnegative, continuity and single-peakedness yield exactly two zeros
\[
\tilde k_i^-<\tilde k_i^+
\]
in $(0,k_i^+)$ such that
\[
Y_i(k_i)\ge 0
\Longleftrightarrow
k_i\in[\tilde k_i^-,\tilde k_i^+].
\]
This proves part~(ii).

\noindent\textit{Case: \eqref{Eujmd} crosses from $-$ to $+$.}
Then $Y_i$ is single-dipped with $Y_i(0)<0$. Either the trough value is negative and $Y_i<0$ throughout, or $Y_i$ rises above zero after the trough, crossing zero exactly once from below at a unique $\tilde k_i\in(0,k_i^+]$, and
\[
Y_i(k_i)\ge 0
\Longleftrightarrow
k_i\in[\tilde k_i,k_i^+].
\]
\end{proof}

\section{Proof of Preferences}
\begin{proof}
Write
\[
\lambda_i^1(\theta):=\lambda_i(\theta),
\qquad
\lambda_i^2(\theta):=\lambda_i^\Delta(\theta),
\]
and define
\[
\tilde\lambda_i^k
:=
\int_{\underline\theta_i}^{\overline\theta_i}\lambda_i^k(\theta)\,dG_i(\theta),
\qquad
\Lambda_i^k(\theta)
:=
\frac{1}{g_i(\theta)}
\int_\theta^{\overline\theta_i}\lambda_i^k(t)\,dG_i(t),
\qquad
k\in\{1,2\}.
\]
Let
\[
m(\theta):=\frac{\lambda_i^2(\theta)}{\lambda_i^1(\theta)},
\]
which is increasing in the upward-shift case and decreasing in the downward-shift case.

\medskip
\noindent\textit{Core step: monotonicity of $\theta\mapsto\Lambda_i^2(\theta)/\Lambda_i^1(\theta)$.}
For each $\theta\in[\underline\theta_i,\overline\theta_i)$, define the probability measure $Q_\theta$ on $[\theta,\overline\theta_i]$ by
\[
dQ_\theta(t)
:=
\frac{\lambda_i^1(t)\,dG_i(t)}
{\displaystyle\int_\theta^{\overline\theta_i}\lambda_i^1(x)\,dG_i(x)}
\,\mathbf 1\{t\ge\theta\}.
\]
Since $\lambda_i^2(t)=m(t)\lambda_i^1(t)$,
\[
\frac{\Lambda_i^2(\theta)}{\Lambda_i^1(\theta)}
=
\mathbb E_{Q_\theta}[m(\tilde\theta)].
\]
For $\theta'<\theta$, $Q_\theta$ is $Q_{\theta'}$ conditioned on $\{\tilde\theta\ge\theta\}$. When $m$ is increasing, this conditioning shifts mass toward larger values of $m$, so by first-order stochastic dominance,
\[
\mathbb E_{Q_\theta}[m(\tilde\theta)]
\ge
\mathbb E_{Q_{\theta'}}[m(\tilde\theta)].
\]
Hence $\theta\mapsto\Lambda_i^2(\theta)/\Lambda_i^1(\theta)$ is increasing on $[\underline\theta_i,\overline\theta_i)$. When $m$ is decreasing the inequality reverses.

\medskip
\noindent\textit{Proof of (i).}
Since $\theta\mapsto\Lambda_i^2(\theta)/\Lambda_i^1(\theta)$ is monotone, its logarithm is monotone too. Therefore, for almost every $\theta\in(\underline\theta_i,\overline\theta_i)$,
\[
0
\le
\frac{d}{d\theta}
\log\!\left(\frac{\Lambda_i^2(\theta)}{\Lambda_i^1(\theta)}\right)
=
\frac{\Lambda_i^{2\,\prime}(\theta)}{\Lambda_i^2(\theta)}
-
\frac{\Lambda_i^{1\,\prime}(\theta)}{\Lambda_i^1(\theta)}
\]
in the upward case, which yields
\[
\frac{\Lambda_i^{\Delta\,\prime}(\theta)}{\Lambda_i^\Delta(\theta)}
\ge
\frac{\Lambda_i'(\theta)}{\Lambda_i(\theta)}
\]
for almost every $\theta$. The reverse inequality holds in the downward case.

\medskip
\noindent\textit{Proof of (ii) under $\beta_i^\times=0$.}\footnote{When $\beta_i^\times\neq 0$, the effective weight is $\mathcal G_i(\theta)=J_i(\theta)(\tilde\lambda_i+\beta_i^\times)+\Lambda_i(\theta)$. An analogous argument applies with $c=(\tilde\lambda_i^\Delta+\beta_i^{\times,\Delta})/(\tilde\lambda_i+\beta_i^\times)$, but propagating the $\Lambda$ bound to the full $\mathcal G$ bound additionally requires $c\le\tilde\lambda_i^\Delta/\tilde\lambda_i$. Since this condition involves the endogenous multiplier $\beta_i^\times$ and cannot in general be verified from primitives alone, the main text restricts to the IR-unconstrained region.}
When $\beta_i^\times=0$, the effective weight reduces to
\[
\Gamma_i(\theta)=J_i(\theta)\tilde\lambda_i+\Lambda_i(\theta),
\qquad
\Gamma_i^\Delta(\theta)=J_i(\theta)\tilde\lambda_i^\Delta+\Lambda_i^\Delta(\theta).
\]

\medskip
\noindent\textbf{Upward case.}
Assume $m$ is increasing and $\tilde\lambda_i^\Delta\ge\tilde\lambda_i$. Set
\[
c:=\frac{\tilde\lambda_i^\Delta}{\tilde\lambda_i}\ge 1.
\]
The boundary evaluation gives
\[
\frac{\Lambda_i^2(\underline\theta_i)}{\Lambda_i^1(\underline\theta_i)}
=
\frac{\tilde\lambda_i^\Delta}{\tilde\lambda_i}
=
c.
\]
Since the ratio is increasing by the core step and starts at $c$,
\[
\Lambda_i^\Delta(\theta)\ge c\,\Lambda_i(\theta)
\qquad
\text{for all }\theta.
\]
Writing
\[
\Gamma_i^\Delta(\theta)
=
c\,J_i(\theta)\tilde\lambda_i+\Lambda_i^\Delta(\theta)
\ge
c\,\Gamma_i(\theta),
\]
and using $c\ge 1$ and $\Gamma_i(\theta)>0$, we obtain
\[
\Delta\mathcal G_i(\theta)\ge (c-1)\Gamma_i(\theta)\ge 0
\qquad
\text{for all }\theta\in[\underline\theta_i,\overline\theta_i].
\]

\medskip\noindent\textbf{Downward case.}
Assume $m$ is decreasing and $\tilde\lambda_i^\Delta\le\tilde\lambda_i$. Set
\[
c:=\frac{\tilde\lambda_i^\Delta}{\tilde\lambda_i}\le 1.
\]
The boundary evaluation gives
\[
\frac{\Lambda_i^\Delta(\underline\theta_i)}{\Lambda_i(\underline\theta_i)}
=
\frac{\tilde\lambda_i^\Delta}{\tilde\lambda_i}
=
c.
\]
Since the ratio is decreasing by the core step and starts at $c$,
\[
\Lambda_i^\Delta(\theta)\le c\,\Lambda_i(\theta)
\qquad
\text{for all }\theta.
\]
Therefore
\[
\Gamma_i^\Delta(\theta)
=
c\,J_i(\theta)\tilde\lambda_i+\Lambda_i^\Delta(\theta)
\le
c\,J_i(\theta)\tilde\lambda_i+c\,\Lambda_i(\theta)
=
c\,\Gamma_i(\theta).
\]
Since $c\le 1$ and $\Gamma_i(\theta)>0$, it follows that
\[
\Delta \mathcal G_i(\theta)
=
\Gamma_i^\Delta(\theta)-\Gamma_i(\theta)
\le
(c-1)\Gamma_i(\theta)
\le 0
\qquad
\text{for all }\theta\in[\underline\theta_i,\overline\theta_i].
\]
\qedhere
\end{proof}

\section{Proof of Proposition~\ref{prop:Elasticity}}
\begin{proof}
The proof proceeds in three steps.

\medskip
\noindent\textit{Step 1: the elasticity condition implies that $\Delta R_\theta(\theta)$ is monotone.}
Differentiating
\[
\Delta R_\theta(\theta)=-R_\Gamma(\theta)R_\theta(\theta)
\]
gives
\[
\Delta R_\theta'(\theta)
=
-R_\Gamma(\theta)R_\theta'(\theta)-R_\theta(\theta)R_\Gamma'(\theta).
\]
If the two terms on the right-hand side have the same sign, then $\Delta R_\theta'(\theta)$ inherits that sign directly. If they have opposite signs, then
\[
\operatorname{sign}\!\left(\Delta R_\theta'(\theta)\right)
=
\operatorname{sign}\!\left(-R_\Gamma(\theta)R_\theta'(\theta)\right)
\]
if and only if
\[
|R_\Gamma(\theta)R_\theta'(\theta)|
\ge
|R_\theta(\theta)R_\Gamma'(\theta)|,
\]
which is equivalent to
\[
\left|\frac{R_\Gamma'(\theta)}{R_\Gamma(\theta)}\right|
\le
\left|\frac{R_\theta'(\theta)}{R_\theta(\theta)}\right|.
\]
Thus the elasticity condition implies
\[
\operatorname{sign}\!\left(\Delta R_\theta'(\theta)\right)
=
\operatorname{sign}\!\left(-R_\Gamma(\theta)R_\theta'(\theta)\right)
\]
for all $\theta$. Since $R_\Gamma(\theta)$ has a constant sign and $R_\theta(\theta)$ is monotone, the right-hand side has constant sign, so $\Delta R_\theta(\theta)$ is globally monotone.

\medskip
\noindent\textit{Step 2: monotonicity of $\Delta R_\theta(\theta)$ determines the sign of the covariance.}
By Step~1, both $\Delta R_\theta(\theta)$ and $R_q(\theta)$ are monotone. The covariance-monotonicity then implies
\[
\operatorname{sign}\!\left(
\operatorname{Cov}_{\theta_i}\!\left(\Delta R_\theta(\theta),R_q(\theta)\right)
\right)
=
\operatorname{sign}\!\left(\Delta R_\theta'(\theta)\right)
\cdot
\operatorname{sign}\!\left(R_q'(\theta)\right),
\]
hence
\[
\operatorname{sign}\!\left(
\operatorname{Cov}_{\theta_i}\!\left(\Delta R_\theta(\theta),R_q(\theta)\right)
\right)
=
\operatorname{sign}\!\left(-R_\Gamma(\theta)R_\theta'(\theta)\right)
\cdot
\operatorname{sign}\!\left(R_q'(\theta)\right).
\]

\medskip
\noindent\textit{Step 3: if the elasticity condition fails, no primitive sign result remains.}
If the elasticity condition is violated at some $\theta_0$, namely if
\[
|R_\theta(\theta_0)R_\Gamma'(\theta_0)|
>
|R_\Gamma(\theta_0)R_\theta'(\theta_0)|,
\]
then $\Delta R_\theta'(\theta_0)$ has the opposite sign to $-R_\Gamma(\theta_0)R_\theta'(\theta_0)$. This creates a local reversal in the monotonicity of $\Delta R_\theta(\theta)$, so the covariance-monotonicity lemma no longer applies. The sign of
\[
\operatorname{Cov}_{\theta_i}\!\left(\Delta R_\theta(\theta),R_q(\theta)\right)
\]
then depends on the distribution of types and on the magnitude of the reversal, and no distribution-free sign result obtains.
\end{proof}

\section{Proof of Lemma~\ref{lem:realloc}}
\begin{proof}
When $\beta_i^\times=0$, one has $\mathcal G_i(\theta)=\Gamma_i(\theta)$ and therefore $R_{\mathcal G}(\theta)=R_\Gamma(\theta)$; the result in the main text is stated in terms of $R_\Gamma$ for that case. Fix $s\in T_i^\times$ and a local perturbation $\Delta\lambda_i(\theta)$ holding $k_i$ fixed.

Using the weights $w_i^\times(\theta,s)$ and $\mathbb E_i^w[\cdot\mid s]$ from the preliminary step of Lemma~\ref{lem:SC-dir}, note that
\[
R_q(\theta,s)=\varepsilon_i^\times(s)\,w_i^\times(\theta,s),
\]
which follows from
\[
R_q(\theta,s)=
-\frac{u(q_i^\times,s)}{u_q(q_i^\times,s)}
\qquad\text{and}\qquad
\varepsilon_i^\times(s)=u(q_i^\times,s)\mathcal G_i(\theta).
\]

Linearizing the pointwise first-order condition
\[
u(q_i^\times(\theta,s),s)\mathcal G_i(\theta)=\varepsilon_i^\times(s)
\]
gives
\[
u_q\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta)\Delta q_i^\times(\theta,s)
+
u\!\left(q_i^\times(\theta,s),s\right)\Delta\mathcal G_i(\theta)
-
\Delta\varepsilon_i^\times(s)
=
0.
\]
Solving for $\Delta q_i^\times(\theta,s)$ and substituting the definitions of $R_q$, $R_{\mathcal G}(\theta)=\Delta\mathcal G_i(\theta)/\mathcal G_i(\theta)$, and $\varepsilon_i^\times(s)$ yields
\[
\Delta q_i^\times(\theta,s)
=
R_q(\theta,s)
\left(
R_{\mathcal G}(\theta)
-
\frac{\Delta\varepsilon_i^\times(s)}{\varepsilon_i^\times(s)}
\right).
\]
Since the capacity constraint binds state-by-state and $k_i$ is held fixed,
\[
\mathbb E_{\theta_i}\!\left[\Delta q_i^\times(\theta,s)\right]=0
\qquad
\text{for each }s\in T_i^\times.
\]
Substituting the previous expression and using $R_q(\theta,s)=\varepsilon_i^\times(s)w_i^\times(\theta,s)$, dividing by $\varepsilon_i^\times(s)\bar w_i^\times(s)>0$, and rearranging gives
\[
\frac{\Delta\varepsilon_i^\times(s)}{\varepsilon_i^\times(s)}
=
\mathbb E_i^w\!\left[R_{\mathcal G}\mid s\right].
\]
Substituting back,
\[
\Delta q_i^\times(\theta,s)
=
R_q(\theta,s)
\left(
R_{\mathcal G}(\theta)-\mathbb E_i^w\!\left[R_{\mathcal G}\mid s\right]
\right),
\]
which is the desired identity.
\end{proof}

\section{Proof of Proposition~\ref{prop:rprefreallocation}}
\begin{proof}
Under the IR-unconstrained allocation, $\beta_i^\times=0$, so Lemma~\ref{lem:realloc} gives, for each $s\in T_i^\times$,
\[
\Delta q_i^\times(\theta,s)
=
R_q(\theta,s)
\left(
R_\Gamma(\theta)-\mathbb E_i^w\!\left[R_\Gamma\mid s\right]
\right).
\]

\medskip
\noindent\textit{First part: $\Delta q_i^\times(\cdot,s)$ is not monotone when $R_\Gamma$ is not constant.}
Since $\mathbb E_i^w[R_\Gamma\mid s]$ is a weighted average of $R_\Gamma$ with positive weights, if $R_\Gamma$ is not almost everywhere constant then
\[
R_\Gamma(\theta)-\mathbb E_i^w\!\left[R_\Gamma\mid s\right]
\]
must take both positive and negative values on $\Theta_i$. Because $R_q(\theta,s)>0$, the same is true of $\Delta q_i^\times(\theta,s)$.

Next, recall that
\[
\Gamma_i(\underline\theta_i)=\tilde\lambda_i\,\underline\theta_i,
\qquad
\Gamma_i(\overline\theta_i)=\tilde\lambda_i\,\overline\theta_i.
\]
Hence, for any perturbation,
\[
R_\Gamma(\underline\theta_i)=R_\Gamma(\overline\theta_i)
=
\frac{\Delta\tilde\lambda_i}{\tilde\lambda_i}.
\]
Therefore
\[
\left(
R_\Gamma-\mathbb E_i^w[R_\Gamma\mid s]
\right)(\underline\theta_i)
=
\left(
R_\Gamma-\mathbb E_i^w[R_\Gamma\mid s]
\right)(\overline\theta_i),
\]
so the endpoint values of $\Delta q_i^\times(\theta,s)$ have the same sign (or are both zero). Since $\Delta q_i^\times(\theta,s)$ takes both signs while its endpoint values coincide in sign, it cannot be monotone on $\Theta_i$.

\medskip
\noindent\textit{Second part: unimodality and mean-preserving perturbations.}
Assume now that $R_\Gamma(\theta)$ is unimodal and that $\Delta\lambda_i(\theta)$ is mean-preserving, so $\Delta\tilde\lambda_i=0$. Then
\[
R_\Gamma(\underline\theta_i)=R_\Gamma(\overline\theta_i)=0.
\]

Consider first the case $\Delta\mathcal G_i(\theta)\ge 0$ for all $\theta$. Since $\Gamma_i(\theta)>0$, this is equivalent to $R_\Gamma(\theta)\ge 0$ for all $\theta$. Set
\[
c_s:=\mathbb E_i^w[R_\Gamma\mid s]
=
\frac{\Delta\varepsilon_i^\times(s)}{\varepsilon_i^\times(s)}.
\]
Because $R_\Gamma$ is nonnegative, not constant, and vanishes at both endpoints, one has
\[
0<c_s<\max_{\theta\in\Theta_i}R_\Gamma(\theta).
\]
By unimodality, the equation
\[
R_\Gamma(\theta)=c_s
\]
has exactly two solutions, denoted $\theta_-<\theta_+$. Since
\[
\Delta q_i^\times(\theta,s)
=
R_q(\theta,s)\bigl(R_\Gamma(\theta)-c_s\bigr)
\qquad\text{and}\qquad
R_q(\theta,s)>0,
\]
it follows that
\[
\Delta q_i^\times(\theta,s)\le 0
\quad\text{for }\theta\in[\underline\theta_i,\theta_-)\cup(\theta_+,\overline\theta_i],
\]
and
\[
\Delta q_i^\times(\theta,s)\ge 0
\quad\text{for }\theta\in[\theta_-,\theta_+].
\]

Finally, using the binding first-order condition
\[
u(q_i^\times(\theta,s),s)\Gamma_i(\theta)=\varepsilon_i^\times(s),
\]
the equation $R_\Gamma(\theta)=c_s$ is equivalent to
\[
u(q_i^\times(\theta,s),s)\,\Delta\mathcal G_i(\theta)=\Delta\varepsilon_i^\times(s).
\]
This is exactly the characterization stated in the proposition.

If instead $\Delta\mathcal G_i(\theta)\le 0$ for all $\theta$, apply the same argument to $-\Delta\mathcal G_i(\theta)$. This reverses all inequalities and proves part~(ii).
\end{proof}
\section{Proof of Corollary~\ref{cor:CARA}}
\begin{proof}
Assume $\beta_i^\times=0$. In any on-peak state $s\in T_i^\times$ where the capacity constraint binds with multiplier $\varepsilon_i^\times(s)>0$, the interior FOC is
\[
u\!\left(q_i^\times(\theta,s),s\right)\Gamma_i(\theta)=\varepsilon_i^\times(s). \label{eq:FOC-CARA}\tag{FOC-CARA}
\]
\paragraph{Step 1: CARA implies $R_q$ is constant and the behavioral channel vanishes.}
Under Assumption~\ref{ass:CARA}, $u_q=-\alpha u$ for some $\alpha>0$, hence $R_q(\theta):=-u/u_q=1/\alpha$ is constant. For a preference perturbation $\Delta\lambda_i(\theta)$ with $R_\Gamma(\theta):=\Delta\Gamma_i(\theta)/\Gamma_i(\theta)$, we have
\[
\Delta R_\theta(\theta) = \Delta\!\left(\frac{J_i(\theta)}{\Gamma_i(\theta)}\right) = -R_\Gamma(\theta)R_\theta(\theta).
\]
In particular, $\mathbb{E}_{\theta_i}[\Delta R_\theta(\theta)] = -\operatorname{Cov}_{\theta_i}(R_\Gamma,R_\theta) - \mathbb{E}_{\theta_i}[R_\Gamma]\,\mathbb{E}_{\theta_i}[R_\theta]$.

\paragraph{Step 2: covariance formula under CARA.}
By Lemma~\ref{lem:realloc} with $\beta_i^\times=0$,
\[
\Delta q_i^\times(\theta,s) = R_q(\theta,s)\!\left(R_\Gamma(\theta)-\mathbb{E}^w_i[R_\Gamma\mid s]\right).
\]
Under Assumption~\ref{ass:CARA}, the weights $w_i^\times(\theta,s)$ from the preliminary step of Lemma~\ref{lem:SC-dir} satisfy $w_i^\times(\theta,s)=1/(\alpha\varepsilon_i^\times(s))$, which is constant in $\theta$. Therefore $\mathbb{E}^w_i[R_\Gamma\mid s]=\mathbb{E}_{\theta_i}[R_\Gamma(\theta)]$ and
\[
\Delta q_i^\times(\theta,s) = \frac{1}{\alpha}\!\left(R_\Gamma(\theta)-\mathbb{E}_{\theta_i}[R_\Gamma(\theta)]\right).
\]
Moreover, from \eqref{eq:FOC-CARA}, $\Delta\varepsilon_i^\times(s)/\varepsilon_i^\times(s) = \mathbb{E}_{\theta_i}[R_\Gamma(\theta)]$ for each $s\in T_i^\times$.

\paragraph{Step 3: comparative statics of $k^\ast$.}
Let $F(k):=\partial_k\mathbb{E}\underline{CS}_i^\times(k)$. Single-peakedness and an interior maximizer imply $F(k^\ast)=0$ and $F'(k^\ast)<0$, so $\operatorname{sign}(\Delta k^\ast)=\operatorname{sign}(\Delta F(k^\ast))$. Under Assumption~\ref{ass:CARA} with $\beta_i^\times=0$, the envelope expression gives
\[
F(k) = \mathbb{E}_s[\varepsilon_i^\times(s)]\,\mathbb{E}_{\theta_i}[R_\theta(\theta)].
\]
The full linearization is
\[
\Delta F = \Delta\!\left(\mathbb{E}_s[\varepsilon_i^\times]\right)\cdot\mathbb{E}_{\theta_i}[R_\theta] + \mathbb{E}_s[\varepsilon_i^\times]\cdot\mathbb{E}_{\theta_i}[\Delta R_\theta].
\]
From Step~2, $\Delta\varepsilon_i^\times(s)=\varepsilon_i^\times(s)\,\mathbb{E}_{\theta_i}[R_\Gamma]$ for each $s\in T_i^\times$, so $\Delta(\mathbb{E}_s[\varepsilon_i^\times])=\mathbb{E}_s[\varepsilon_i^\times]\,\mathbb{E}_{\theta_i}[R_\Gamma]$. Substituting and using $\mathbb{E}_{\theta_i}[\Delta R_\theta]=-\operatorname{Cov}_{\theta_i}(R_\Gamma,R_\theta)-\mathbb{E}_{\theta_i}[R_\Gamma]\,\mathbb{E}_{\theta_i}[R_\theta]$ from Step~1,
\[
\Delta F = \mathbb{E}_s[\varepsilon_i^\times]\!\left(\mathbb{E}_{\theta_i}[R_\Gamma]\,\mathbb{E}_{\theta_i}[R_\theta] - \operatorname{Cov}_{\theta_i}(R_\Gamma,R_\theta) - \mathbb{E}_{\theta_i}[R_\Gamma]\,\mathbb{E}_{\theta_i}[R_\theta]\right) = -\mathbb{E}_s[\varepsilon_i^\times]\,\operatorname{Cov}_{\theta_i}(R_\Gamma,R_\theta).
\]
The $\mathbb{E}[R_\Gamma]\mathbb{E}[R_\theta]$ terms cancel exactly. Since $\max_k\mathbb{E}\underline{CS}_i^\times(k_i)>0$, peak capacity binds with positive probability and $\mathbb{E}_s[\varepsilon_i^\times(s)]>0$, so
\[
\operatorname{sign}(\Delta k^\ast) = -\operatorname{sign}\!\left(\operatorname{Cov}_{\theta_i}(R_\Gamma(\theta),R_\theta(\theta))\right).
\]
When $\operatorname{Cov}(R_\Gamma,R_\theta)>0$, both $k^\ast$ shifts left and the implementation region expands, since the same argument applied to the boundary conditions of Proposition~\ref{prop:implementation} shows that $\Delta Y_i(k)>0$ for all $k$ in the implementation region, pushing $\tilde{k}_i^-$ down and $\tilde{k}_i^+$ up. The reverse holds when the covariance is negative.
\end{proof}

\section{Proof of Solution for the Across-category allocation}


\begin{proof}
The feasible set
\[
\mathcal F := \left\{ \{(k_i,I_i)\}_{i\in N} : \sum_{i\in N} k_i = k,\ \sum_{i\in N} I_i = I,\ k_i\ge 0,\ I_i\ge 0\ \forall i \right\}
\]
is compact. By Lemma~\ref{lem:Vi-concave} below, each $V_i$ is continuous and concave, so the outer objective $W:=\sum_i V_i$ is continuous and concave on $\mathcal F$. Existence and convexity of the solution set follow from the extreme value theorem and concavity of $W$. Uniqueness under the stated condition follows from Lemma~\ref{lem:Vi-concave}(i), which gives strict concavity of each $V_i$ on the IR-constrained, capacity-binding region, hence strict concavity of $W$ at the optimum.

\begin{lemma}[Concavity of the within-category value function]\label{lem:Vi-concave}
For each category $i$, the value function $V_i(k_i,I_i)$ is concave on its feasible domain. More precisely:
\begin{enumerate}[label=(\roman*)]
    \item on the IR-constrained, capacity-binding region, where $\beta_i^\times(k_i,I_i)>0$ and $\Pr(s\in T_i^\times(k_i))>0$, the Hessian of $V_i$ is negative definite and $V_i$ is strictly concave;
    \item when $\beta_i^\times(k_i,I_i)=0$, the Hessian is negative semidefinite;
    \item when $T_i^\times(k_i)=\varnothing$, the Hessian is negative semidefinite.
\end{enumerate}
\end{lemma}

\begin{proof}[Proof of Lemma~\ref{lem:Vi-concave}]
Fix a category $i$. On binding states $s\in T_i^\times(k_i):=[s_i^\times(k_i),\overline s]$, the pointwise first-order condition is
\[
u\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta)=\varepsilon_i^\times(s), \qquad \mathcal G_i(\theta)=\Gamma_i(\theta)+\beta_i^\times J_i(\theta).
\]

\medskip
\noindent\textbf{Step 1: \(k_i\)-derivative and the boundary term.}

By the envelope theorem,
\[
\partial_{k_i}V_i(k_i,I_i) = \int_{s_i^\times(k_i,I_i)}^{\overline s}\varepsilon_i^\times(s;k_i,I_i)\,f(s)\,ds.
\]
Differentiating with respect to $k_i$ and applying Leibniz' rule gives a boundary term proportional to $\varepsilon_i^\times(s_i^\times)$. At the threshold state, the off-peak condition $u(q_i^\times(\theta,s_i^\times),s_i^\times)=0$ holds for every $\theta$, and combined with the first-order condition this gives $\varepsilon_i^\times(s_i^\times)=0$. Hence the boundary term vanishes and
\[
V_{i,k_ik_i}(k_i,I_i) = \int_{s_i^\times(k_i,I_i)}^{\overline s} \partial_{k_i}\varepsilon_i^\times(s;k_i,I_i)\,f(s)\,ds.
\]

\medskip
\noindent\textbf{Step 2: \(\partial_{I_i}\varepsilon_i^\times(s)\) and the cross derivative.}

Fix $k_i$ and $s\in T_i^\times$. Differentiating the first-order condition with respect to $I_i$ and imposing the binding capacity constraint $\mathbb E_{\theta_i}[\partial_{I_i}q_i^\times(\theta,s)]=0$ yields
\[
\partial_{I_i}\varepsilon_i^\times(s) = \partial_{I_i}\beta_i^\times\,\mathsf H_i(k_i,s), \qquad \mathsf H_i(k_i,s):=\mathbb E_i^w[uJ\mid s],
\]
where $w_i^\times(\theta,s):=-1/(u_q(q_i^\times,s)\mathcal G_i(\theta))>0$ and the weighted expectation is as defined in Lemma~\ref{lem:SC-dir}. Differentiating $\Phi_i(k_i,\beta_i^\times(k_i,I_i))=0$ with respect to $I_i$, using $\partial_\beta\Phi_i=\mathcal V_i(k_i)>0$, gives $\partial_{I_i}\beta_i^\times=1/\mathcal V_i(k_i)$ and hence $\partial_{I_i}\varepsilon_i^\times(s)=\mathsf H_i(k_i,s)/\mathcal V_i(k_i)$. The boundary term vanishes again since $\varepsilon_i^\times(s_i^\times)=0$, so defining $\mathcal S_i(k_i):=\int_{s_i^\times}^{\overline s}\mathsf H_i(k_i,s)\,f(s)\,ds$,
\[
V_{i,I_i}(k_i,I_i)= -\tilde\lambda_i-\beta_i^\times(k_i,I_i), \qquad V_{i,I_iI_i}(k_i,I_i)= -\frac{1}{\mathcal V_i(k_i)}<0, \qquad V_{i,k_iI_i}(k_i,I_i) = \frac{\mathcal S_i(k_i)}{\mathcal V_i(k_i)}.
\]

\medskip
\noindent\textbf{Step 3: \(V_{i,k_ik_i}\) on the IR-constrained region.}

Holding $\beta$ fixed gives $\partial_{k_i}\varepsilon_i^\times(s)|_\beta = -1/\overline w_i^\times(s)$. Along the optimal path, the chain rule and the identity $\partial_\beta\varepsilon_i^\times(s)=\mathsf H_i(k_i,s)$ (established in the proof of Lemma~\ref{lem:SC-dir}) give
\[
\partial_{k_i}\varepsilon_i^\times(s) = -\frac{1}{\overline w_i^\times(s)} - \mathsf H_i(k_i,s)\frac{\mathcal S_i(k_i)}{\mathcal V_i(k_i)}.
\]
Integrating over binding states and defining $\mathcal A_i(k_i):=\int_{s_i^\times}^{\overline s}f(s)/\overline w_i^\times(s)\,ds>0$,
\[
V_{i,k_ik_i}(k_i,I_i) = -\mathcal A_i(k_i)-\frac{\mathcal S_i(k_i)^2}{\mathcal V_i(k_i)}<0.
\]
The Hessian on the IR-constrained, capacity-binding region is
\[
\nabla^2 V_i(k_i,I_i) = \begin{pmatrix} -\mathcal A_i-\mathcal S_i^2/\mathcal V_i & \mathcal S_i/\mathcal V_i \\[0.4em] \mathcal S_i/\mathcal V_i & -1/\mathcal V_i \end{pmatrix},
\]
with determinant $\det\nabla^2 V_i = \mathcal A_i/\mathcal V_i > 0$. Both diagonal entries are negative and the determinant is positive, so $\nabla^2 V_i$ is negative definite, proving part~(i).

\medskip
\noindent\textbf{Step 4: the remaining regimes.}

If $\beta_i^\times=0$, then $V_i$ is locally affine in $I_i$, so $V_{i,I_iI_i}=0$ and $V_{i,k_iI_i}=0$. If capacity binds on a positive-mass set of states, the fixed-$\beta$ formula gives $V_{i,k_ik_i}=-\mathcal A_i\le 0$, so the Hessian is negative semidefinite, proving part~(ii). If $T_i^\times(k_i)=\varnothing$, all second derivatives vanish and the Hessian is the zero matrix, proving part~(iii).

\medskip
\noindent\textbf{Step 5: global concavity across regime boundaries.}

It remains to verify that $\nabla V_i$ is continuous across the two regime boundaries, which is required for the piecewise concavity argument to yield global concavity.

\textit{IR-constrained to IR-unconstrained.} From Step~2, $V_{i,I_i}=-\tilde\lambda_i-\beta_i^\times$. As $(k_i,I_i)$ approaches the boundary from the constrained side, $\beta_i^\times\to 0^+$ by definition, so $V_{i,I_i}\to -\tilde\lambda_i$, matching the unconstrained value. As $\beta_i^\times\to 0^+$, the allocation and hence $\varepsilon_i^\times(s)$ converge to their unconstrained counterparts, so $\partial_{k_i}V_i$ is continuous across this boundary.

\textit{Capacity-binding to capacity-slack.} As $s_i^\times(k_i)\to\overline s$, the integral $\partial_{k_i}V_i=\int_{s_i^\times}^{\overline s}\varepsilon_i^\times(s)\,f(s)\,ds\to 0$ continuously, matching the slack-side value of zero.

With $\nabla V_i$ continuous at both boundaries, fix any two feasible points $x^0,x^1$ and let $g_i(t):=V_i((1-t)x^0+tx^1)$. Then $g_i$ is piecewise $C^2$ with $g_i'$ continuous at every switching point. On each regime piece $g_i''(t)\le 0$, so $g_i'$ is nonincreasing on each piece. Continuity at switching points prevents upward jumps, so $g_i'$ is globally nonincreasing and $g_i$ is concave. Hence $V_i$ is globally concave.
\end{proof}
\end{proof}
