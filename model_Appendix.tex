\section{Ok Proof of Proposition FB-alloc}

\begin{proof}
Let $\beta$, $\varepsilon(s)$, $\phi_i(\theta,s)$, and $\rho_i(\theta)$ be the Lagrange multipliers on the budget, the capacity, the transfer positivity, and the participation constraints, respectively. The Lagrangian is
\[\label{KKT}\tag{L$^*$}
\begin{aligned}
\mathcal{L} &= \sum_i \mu_i\, \mathbb{E}_{(s,\theta_i)}\!\left[\lambda_i(\theta)\,\theta\, U(q_i,s) - t_i(\theta,s)\right] + \beta\!\left(\sum_i \mu_i \mathbb{E}_{(s,\theta_i)}[t_i(\theta,s)] - I(k)\right) \\
&\quad + \mathbb{E}_s\!\left[\varepsilon(s)\!\left(k - \sum_i \mu_i \mathbb{E}_{\theta_i}[q_i(\theta,s)]\right)\right] + \sum_i  \mathbb{E}_{(s,\theta_i)}\left[ \rho_i(\theta)(\theta U(q_i,s) - t_i(\theta,s))\right] dG_i(\theta) \\
&\quad + \sum_i \mu_i\, \mathbb{E}_{(s,\theta_i)}\!\left[\phi_i(\theta,s)\,t_i(\theta,s)\right],
\end{aligned}
\]
with $\beta, \varepsilon(s), \rho_i(\theta), \phi_i(\theta,s) \geq 0$. The FOCs for each $(i,\theta,s)$ are
\begin{align*}
\frac{\partial \mathcal{L}}{\partial q_i} = 0 \quad &\Rightarrow \quad \left(\mu_i\lambda_i(\theta) + \rho_i(\theta)\right)\theta\, u(q_i,s) = \mu_i\varepsilon(s), \\
\frac{\partial \mathcal{L}}{\partial t_i} = 0 \quad &\Rightarrow \quad \mu_i\beta - \mu_i\lambda_i(\theta) - \rho_i(\theta) + \phi_i(\theta,s) = 0.
\end{align*}
Complementary slackness conditions are
\[
\varepsilon(s)\!\left(k - \sum_i \mu_i \mathbb{E}_{\theta_i}[q_i(\theta,s)]\right) = 0, \quad \beta\!\left(\sum_i \mu_i \mathbb{E}_{s,\theta}[t_i] - I(k)\right) = 0,
\]
\[
\rho_i(\theta)\,\mathbb{E}_s\!\left[\theta U(q_i,s) - t_i(\theta,s)\right] = 0, \quad \phi_i(\theta,s)\,t_i(\theta,s) = 0.
\]

\textit{Step 1: $s$-independence of transfers (WLOG).} The objective, the budget constraint, and the IR constraint all depend on $t_i(\theta,s)$ only through its expectation $\mathbb{E}_s[t_i(\theta,s)]$. Therefore any feasible state-contingent transfer schedule $t_i(\theta,s)$ can be replaced by $\bar{t}_i(\theta) := \mathbb{E}_s[t_i(\theta,s)]$ without altering the objective value or any constraint. Transfer positivity is preserved since $t_i(\theta,s) \geq 0$ for all $s$ implies $\bar{t}_i(\theta) \geq 0$. Hence without loss of generality $t_i^*(\theta,s) = t_i^*(\theta)$, and $\rho_i(\theta)$ and $\phi_i(\theta)$ are also $s$-independent.

\textit{Step 2: structure of optimal transfers.} Suppose two types $(\theta,i)$ and $(\theta',i')$ satisfy $\lambda_i(\theta) < \lambda_{i'}(\theta')$. Assume for contradiction that $t_{i'}(\theta') > 0$ and $t_i(\theta) = 0$. If both participation constraints are slack, then $\rho_i(\theta) = \rho_{i'}(\theta') = 0$ and $\phi_{i'}(\theta') = 0$, so the FOC on $t$ gives $\lambda_{i'}(\theta') = \beta$. But $t_i(\theta) = 0$ with slack participation requires $\phi_i(\theta) > 0$, giving $\lambda_i(\theta) = \beta + \phi_i(\theta)/\mu_i > \beta = \lambda_{i'}(\theta')$, contradicting $\lambda_i(\theta) < \lambda_{i'}(\theta')$. The case where the participation constraint of $(\theta', i')$ binds leads to the same contradiction. Therefore at the optimum, any type with $\lambda_i(\theta) < \lambda^*$ has $t_i^*(\theta) > 0$ and any type with $\lambda_i(\theta) > \lambda^*$ has $t_i^*(\theta) = 0$, where $\lambda^* = \beta$ is the marginal welfare weight pinned by the budget constraint.

\textit{Step 3: optimal quantities.} When capacity is slack, $\varepsilon(s) = 0$ and the FOC on $q$ gives $\theta\, u(q_i^*,s) = 0$, which is \eqref{eq:FB-S}. When capacity binds, the FOC on $q$ gives $\theta\, u(q_i^*,s) = \varepsilon(s)/(\lambda_i(\theta) + \rho_i(\theta)/\mu_i)$. For types with $\lambda_i(\theta) > \lambda^*$: $\rho_i(\theta) = 0$ and $\theta\, u(q_i^*,s) = \varepsilon(s)/\lambda_i(\theta)$. For types with $\lambda_i(\theta) \leq \lambda^*$: from the FOC on $t$, $\lambda_i(\theta) + \rho_i(\theta)/\mu_i = \beta$, so $\theta\, u(q_i^*,s) = \varepsilon(s)/\beta$. This establishes \eqref{eq:FB-T}.

\textit{Step 4: monotonicity of $q_i^*$ in $s$ and uniqueness of $s^*$.} When capacity is slack, totally differentiating $\theta\, u(q_i^*,s) = 0$ gives
\[
\frac{\partial q_i^*(\theta,s)}{\partial s} = -\frac{u_s(q_i^*,s)}{u_q(q_i^*,s)} > 0,
\]
since $u_q < 0$ and $u_s > 0$ by assumption. Hence aggregate demand $\sum_i \mu_i \mathbb{E}_{\theta_i}[q_i^*(\theta,s)]$ is strictly increasing in $s$, which guarantees existence of a unique threshold $s^*$ separating $S^*$ and $T^*$.
\end{proof}

\section{Ok Proof of Proposition CI-welfare}

\subsection{Part i)}
\begin{proof}
We maintain throughout that $B(k)+M(k)\neq 0$, where $B(k)+M(k)$ is the derivative of the binding residual equation with respect to $\beta$ after substituting the optimal allocation; this is the local regularity condition needed to apply the implicit function theorem, while $B(k)+M(k)>0$ delivers the monotone comparative statics used below.

\medskip\noindent\textbf{Corner case: $\beta(k)=0$.} If $\beta(k)=0$ there are no contributing types. On $T^*$, the first-order condition is $\theta\,u(q_i^*(\theta,s),s) = \varepsilon(s)/\lambda_i(\theta)$. Differentiating and substituting into the capacity constraint gives $\frac{\partial\varepsilon(s)}{\partial k} = -1/\overline w^{nc}(s) < 0$ and $\frac{\partial q_i^*(\theta,s)}{\partial k} = w_i^{nc}(\theta,s)/\overline w^{nc}(s) > 0$ on $T^*$. Since non-contributing types pay no transfer:
\[
\frac{\partial CS_i(\theta;k)}{\partial k} = \mathbb{E}_s\!\left[\theta\,u(q_i^*(\theta,s),s)\,\frac{\partial q_i^*(\theta,s)}{\partial k}\,\mathbf{1}_{s\in T^*}\right] \geq 0.
\]

\medskip\noindent\textbf{Interior case: $\beta(k)>0$ with coexistence.} Assume contributing and non-contributing types coexist on a set of peak states of positive probability. On $T^*$, the first-order conditions are
\[
\theta\,u(q_i^*(\theta,s),s) = \frac{\varepsilon(s)}{\beta(k)}, \quad \lambda_i(\theta)\leq\beta(k), \qquad \theta\,u(q_i^*(\theta,s),s) = \frac{\varepsilon(s)}{\lambda_i(\theta)}, \quad \lambda_i(\theta)>\beta(k).
\]
The weights $w_i^c$, $w_i^{nc}$, $\overline w^c(s)$, $\overline w^{nc}(s)$, $\overline w(s)$ and the inframarginal component
\[
m^c(s) := \frac{1}{\overline w(s)}\sum_i\mu_i\,\mathbb{E}_{\theta_i}\!\left[w_i^c(\theta,s)\,\theta\,u(q_i^*(\theta,s),s)\,\mathbf{1}_{\lambda_i(\theta)\leq\beta(k)}\right]
\]
are as defined above. Differentiating the first-order conditions with respect to $k$:
\[
\frac{\partial q_i^*(\theta,s)}{\partial k} = w_i^c(\theta,s)\!\left(-\frac{\partial\varepsilon(s)}{\partial k} + \theta\,u(q_i^*,s)\,\frac{d\beta}{dk}\right), \quad \lambda_i(\theta)\leq\beta(k), \tag{$\partial_k q^c$}\label{dqc}
\]
\[
\frac{\partial q_i^*(\theta,s)}{\partial k} = -\,w_i^{nc}(\theta,s)\,\frac{\partial\varepsilon(s)}{\partial k}, \quad \lambda_i(\theta)>\beta(k). \tag{$\partial_k q^{nc}$}\label{dqnc}
\]
Differentiating the capacity constraint and using continuity of the allocation at each cutoff $\theta_i^*(k)$ --- so that boundary terms cancel --- yields
\[
\frac{\partial\varepsilon(s)}{\partial k} = m^c(s)\,\frac{d\beta}{dk} - \frac{1}{\overline w(s)}. \tag{$\partial_k\varepsilon$}\label{depsk}
\]

\medskip\noindent\textbf{Step 1: sign of $\frac{d\beta}{dk}$.} Since the IR constraint of marginal types binds, $R^M(k)\equiv 0$. Differentiating this identity with respect to $k$, substituting \eqref{dqc} and \eqref{depsk}, and collecting all terms in $\frac{d\beta}{dk}$ on one side yields
\[
\mathbb{E}_s\!\left[\sum_i\mu_i\,\mathbb{E}^w_i[u^c|s]\,\mathbf{1}_{s\in T^*}\right] - I'(k) + \frac{d\beta}{dk}\,B(k) = 0,
\]
where, for convenience of exposition, we write $\mathbb{E}^w_i[u^c|s]$ for the full per-category object defined above --- combining the inframarginal weighted marginal utility and the boundary term from the movement of $\theta_i^*(k)$ --- and
\[
B(k) := \mathbb{E}_s\!\left[\frac{\varepsilon(s)^2}{\beta(k)^2}\frac{\overline w^c(s)\,\overline w^{nc}(s)}{\overline w(s)}\,\mathbf{1}_{s\in T^*}\right] > 0.
\]
Solving for $\frac{d\beta}{dk}$:
\[
\frac{d\beta}{dk} = \frac{I'(k) - \mathbb{E}_s\!\left[\sum_i\mu_i\,\mathbb{E}^w_i[u^c|s]\,\mathbf{1}_{s\in T^*}\right]}{B(k)}.
\]
Since $B(k)>0$, condition \eqref{Ui*} implies $\frac{d\beta}{dk}\leq 0$.

\medskip\noindent\textbf{Step 2: surplus of non-contributing types.} Since $\frac{d\beta}{dk}\leq 0$ and $m^c(s)>0$, equation \eqref{depsk} gives $\frac{\partial\varepsilon(s)}{\partial k}<0$ for all $s\in T^*$. From \eqref{dqnc}: $\frac{\partial q_i^*(\theta,s)}{\partial k} = -w_i^{nc}(\theta,s)\frac{\partial\varepsilon(s)}{\partial k} > 0$ for every non-contributing type. Since $t_i^*(\theta,s)=0$:
\[
\frac{\partial CS_i(\theta;k)}{\partial k} = \mathbb{E}_s\!\left[\theta\,u(q_i^*(\theta,s),s)\,\frac{\partial q_i^*(\theta,s)}{\partial k}\,\mathbf{1}_{s\in T^*}\right] \geq 0.
\]

\medskip\noindent\textbf{Corner case: all types strictly contributing.} If all types satisfy $\lambda_i(\theta)<\beta(k)$, there are no non-contributing types and the statement is vacuous.
\end{proof}

When $I(\cdot)$ is convex, the complete-information problem is a concave maximization problem with a convex feasible set: the objective is concave in the choice variables, the capacity constraint is affine in $k$, and the budget constraint is convex because $I(k)$ is convex. Hence the value function $V^{CI}$ is concave in $k$.
\end{proof}

\section{Ok Proof of Lemma Transfer}

\begin{proof}

For each category $i$, define
\[
B_i(\theta,s)
:=
\theta U\!\left(q_i^\times(\theta,s),s\right)
- \underline\theta_i U\!\left(q_i^\times(\underline\theta_i,s),s\right)
- \int_{\underline{\theta}_i}^{\theta}
U\!\left(q_i^\times(\tilde\theta,s),s\right)\,d\tilde\theta,
\]
and let
\[
\overline{B}_i(\theta)
:=
\mathbb{E}_s[B_i(\theta,s)].
\]

Start with the following individual transfer for each consumer of type $\theta$ in category $i$:
\begin{align*}
t_i^\times(\theta,s)
&=
\tau_i
+ \overline{B}_i(\theta)
- \sum_{j\neq i}  \frac{\mu_j}{1-\mu_j}\,
   \mathbb{E}_{\theta_j}[\overline{B}_j(\theta_j)] + \Big(I(k) -I_i+\sum_{j\neq i}  \frac{\mu_j}{1-\mu_j}I_j\Big),
\end{align*}
where
\[
\tau_i
=
T_{i0}
+\sum_{j\neq i}  \frac{\mu_j}{1-\mu_j}\,
   \mathbb{E}_{\theta_j}[\overline{B}_j(\theta_j)]
- \Big(I(k) -I_i+\sum_{j\neq i}  \frac{\mu_j}{1-\mu_j}I_j\Big).
\]

Taking expectations over $\theta$, and noting that only $\overline{B}_i(\theta)$ depends on $\theta$, yields
\begin{align*}
\mathbb{E}_{\theta_i}[t_i^\times(\theta,s)]
&=
\tau_i
+ \mathbb{E}_{\theta_i}[\overline{B}_i(\theta)]
- \sum_{j\neq i}  \frac{\mu_j}{1-\mu_j}\,
   \mathbb{E}_{\theta_j}[\overline{B}_j(\theta_j)] 
+ \Big(I(k) -I_i+\sum_{j\neq i}  \frac{\mu_j}{1-\mu_j}I_j\Big).
\end{align*}

Multiplying by the size $\mu_i$ of category $i$, summing over all categories, and using $\sum_{i=1}^n\mu_i = 1$, we obtain
\begin{align*}
\sum_{i=1}^n \mu_i\mathbb{E}_{\theta_i}[t_i^\times(\theta,s)]
&=
\sum_{i=1}^n \mu_i \tau_i
+ \sum_{i=1}^n\mu_i\,\mathbb{E}_{\theta_i}[\overline{B}_i(\theta)]-\sum_{i=1}^n\mu_i\sum_{j\neq i} \frac{\mu_j}{1-\mu_j}\,
   \mathbb{E}_{\theta_j}[\overline{B}_j(\theta_j)]
\\
&\quad
+ I(k)
-\sum_{i=1}^n\mu_iI_i
+\sum_{i=1}^n\mu_i\sum_{j\neq i}  \frac{\mu_j}{1-\mu_j}I_j.
\end{align*}

The second and third terms cancel, and the fifth and sixth terms also cancel, so that
\[
\sum_{i=1}^n \mu_i\mathbb{E}_{\theta_i}[t_i^\times(\theta,s)]
=
\sum_{i=1}^n \mu_i \tau_i + I(k).
\]
Therefore, an allocation satisfies the budget constraint if and only if $\sum_{i=1}^n \mu_i \tau_i \geq 0$.

Next, substitute the expression for $\tau_i$:
\begin{align*}
\sum_{i=1}^n \mu_i \tau_i
&=
\sum_{i=1}^n \mu_i \left(
T_{i0}
+  \sum_{j\neq i}  \frac{\mu_j}{1-\mu_j}\,
     \mathbb{E}_{\theta_j}[\overline{B}_j(\theta_j)]
\right)
- I(k) \\
&=
\sum_{i=1}^n \mu_i \left(
T_{i0}
+  \mathbb{E}_{\theta_i}[\overline{B}_i(\theta)]
\right)
- I(k).
\end{align*}

By Fubini's theorem and the usual hazard-rate argument, the expectation of $\overline{B}_i(\theta)$ can be written in terms of the virtual utility
\[
U(q^\times_i(\theta,s),s) J_i(\theta)
:= U\!\left(q_i^\times(\theta,s),s\right)(\theta -
\gamma_i(\theta)\,),
\]
as
\[
\mathbb{E}_{\theta_i}[\overline{B}_i(\theta)]
=
\mathbb{E}_{(s,\theta_i)}\left[
U(q^\times_i(\theta,s),s) J_i(\theta)
- \underline{\theta_i} U\!\left(q_i^\times(\underline{\theta_i},s),s\right)
\right].
\]
Hence
\[
\sum_{i=1}^n \mu_i \tau_i
=
\sum_{i=1}^n \mu_i \mathbb{E}_{(s,\theta_i)}\left[
T_{i0} + U(q^\times_i(\theta,s),s) J_i(\theta) - \underline{\theta_i} U\!\left(q_i^\times(\underline{\theta_i},s),s\right)
\right]
- I(k).
\]

Finally, replacing $\tau_i$ in $t_i^\times(\theta,s)$ yields
\[
t_i^\times(\theta,s)
=
T_{i0}
+ \overline{B}_i(\theta)
=
T_{i0}
+ \mathbb{E}_s\left[
\theta U\!\left(q_i^\times(\theta,s),s\right)
- \underline{\theta_i}U\!\left(q_i^\times(\underline{\theta_i},s),s\right)
- \int_{\underline{\theta}_i}^{\theta}
U\!\left(q_i^\times(\tilde\theta,s),s\right)\, d\tilde\theta
\right].
\]

Thus
\begin{align*}
\mathbb{E}_s\big[\theta U(q_i^\times(\theta,s),s) - t_i^\times(\theta,s)\big]
&=
\mathbb{E}_s\big[\underline{\theta_i}U(q_i^\times(\underline{\theta_i},s),s)\big]
- T_{i0}
\\
&\quad
+ \int_{\underline{\theta}_i}^{\theta}
\mathbb{E}_s\big[U(q_i^\times(\tilde\theta,s),s)\big]\,d\tilde\theta.
\end{align*}
Defining
\[
\E\underline{CS}_i^\times
:=
\mathbb{E}_s\big[\underline{\theta_i}U(q_i^\times(\underline{\theta_i},s),s)\big]
- T_{i0},
\]
we get
\[
\mathbb{E}_s[\theta U(q_i^\times(\theta,s),s) - t_i^\times(\theta,s)]
=
\E\underline{CS}_i^\times
+ \int_{\underline{\theta}_i}^{\theta}
\mathbb{E}_s\big[U(q_i^\times(\tilde\theta,s),s)\big]\,d\tilde\theta,
\]
which is the usual envelope representation and therefore satisfies the IC condition. Moreover, substituting $\underline{\theta_i}U(q_i^\times(\underline{\theta_i},s),s) = \E\underline{CS}_i^\times + T_{i0}$ into the expression for $\sum_i \mu_i \tau_i$ gives the claimed transformed budget constraint
\[
\sum_{i=1}^n \mu_i \tau_i
=
\sum_{i=1}^n \mu_i \mathbb{E}_{(s,\theta_i)}\big[ U(q^\times_i(\theta,s),s) J_i(\theta) - \E\underline{CS}_i^\times\big]
- I(k).
\]

Multiplying the utility of a consumer with type $\theta$ from category $i$ and taking the expectation over the whole category yields

\[
\mathbb{E}_{(s,\theta_i)}[\lambda_i(\theta) \;( \;\theta U(q_i^\times(\theta,s),s) - t_i^\times(\theta,s)\;)\;]
=
\tilde{\lambda}_i \; \E\underline{CS}_i^\times
+ \mathbb{E}_{\theta_i}[\lambda_i(\theta)\int_{\underline{\theta}_i}^{\theta}
\mathbb{E}_s\big[U(q_i^\times(\tilde\theta,s),s)\big]\,d\tilde\theta],
\]

We apply Fubini's theorem to the second term:

\begin{align*}
\mathbb{E}_{\theta_i}\!\left[\lambda_i(\theta)\int_{\underline{\theta}_i}^{\theta}
\mathbb{E}_s[U(q_i^\times(\tilde\theta,s),s)]\,d\tilde\theta\right]
&= \int_{\underline{\theta}_i}^{\overline{\theta}_i} \lambda_i(\theta) g_i(\theta) 
   \left(\int_{\underline{\theta}_i}^{\theta} \mathbb{E}_s[U(q_i^\times(\tilde\theta,s),s)]\,d\tilde\theta\right) d\theta \\
&= \int_{\underline{\theta}_i}^{\overline{\theta}_i} \mathbb{E}_s[U(q_i^\times(\tilde\theta,s),s)] 
   \left(\int_{\tilde\theta}^{\overline{\theta}_i} \lambda_i(\theta) g_i(\theta)\,d\theta\right) d\tilde\theta \\
&= \int_{\underline{\theta}_i}^{\overline{\theta}_i} \mathbb{E}_s[U(q_i^\times(\tilde\theta,s),s)] 
    \gamma_i(\theta) \E[ \ \lambda_i(\tilde \theta) \ | \ \tilde \theta \ \ge \ \theta \ ] d\tilde\theta \\
&= \mathbb{E}_{(s,\theta_i)}\!\left[\Lambda_i(\theta)\, U(q_i^\times(\theta,s),s)\right].
\end{align*}

This yields the desired expression:

\[ \tilde{\lambda}_i \;\E\underline{CS}_i^\times 
      + \E_{(s,\theta_i)} \big[\; \Lambda_i(\theta)\;  U(q_i^\times(\theta,s),s)\; \big] \]

\end{proof}

\section{Ok Proof of Lemma FOC Without Ironing}
\begin{proof}
Consider the within-category problem
\[
\max_{q_i^\times(\theta,s)} 
\mathbb{E}_{(s,\theta_i)}\!\left[\Gamma_i(\theta)\,U\!\left(q_i^\times(\theta,s),s\right)\right]
\]
subject to
\[
\mathbb{E}_{(s,\theta_i)}\!\left[U\!\left(q_i^\times(\theta,s),s\right)J_i(\theta)\right]-I_i \ge 0,
\qquad
\mathbb{E}_{\theta_i}\!\left[q_i^\times(\theta,s)\right]\le k_i
\quad \forall s.
\]

Let \(\beta_i^\times \ge 0\) and \(\varepsilon_i^\times(s)\ge 0\) denote the multipliers on \eqref{Rmdi-def} and \eqref{Kmdi-def}, respectively. The associated Lagrangian is
\[
\mathcal L
=
\mathbb{E}_{(s,\theta_i)}\!\left[
\left(\Gamma_i(\theta)+\beta_i^\times J_i(\theta)\right)
U\!\left(q_i^\times(\theta,s),s\right)
\right]
-
\mathbb{E}_s\!\left[
\varepsilon_i^\times(s)
\left(\mathbb{E}_{\theta_i}[q_i^\times(\theta,s)]-k_i\right)
\right]
-
\beta_i^\times I_i.
\]
Hence the pointwise first-order condition is
\[
u\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta)-\varepsilon_i^\times(s)=0,
\qquad
\text{where }
\mathcal G_i(\theta):=\Gamma_i(\theta)+\beta_i^\times J_i(\theta).
\]

We first characterize the states in which the capacity constraint is slack. If \(\mathbb{E}_{\theta_i}[q_i^\times(\theta,s)]<k_i\), complementary slackness implies \(\varepsilon_i^\times(s)=0\). Since Assumption~\ref{ass:interior} gives \(\mathcal G_i(\theta)>0\) for all \(\theta\), the first-order condition reduces to
\[
u\!\left(q_i^\times(\theta,s),s\right)=0.
\]
By strict concavity of \(U(\cdot,s)\), this equation has a unique solution, denoted \(q_i^0(s)\). In particular, the off-peak allocation is independent of \(\theta\), so it automatically satisfies monotonicity.

Moreover, implicit differentiation of \(u(q_i^0(s),s)=0\) yields
\[
\frac{\partial q_i^0(s)}{\partial s}
=
-\frac{u_s\!\left(q_i^0(s),s\right)}{u_q\!\left(q_i^0(s),s\right)}
\ge 0,
\]
since \(u_s\ge 0\) and \(u_q<0\). Therefore the unconstrained aggregate demand $Q_i^0(s):=\mathbb{E}_{\theta_i}[q_i^0(s)]$ is nondecreasing in \(s\). It follows that the set of states in which the capacity constraint is slack is an interval of the form \(S_i^\times=[\underline s,s_i^\times)\), while the set of binding states is \(T_i^\times=[s_i^\times,\overline s]\), where \(s_i^\times\) is defined by $\mathbb{E}_{\theta_i}\!\left[q_i^\times(\theta,s_i^\times)\right]=k_i$, with the obvious boundary cases if one of the two sets is empty.

Now consider \(s\in T_i^\times\). For such states, the allocation solves the state-by-state problem with binding aggregate quantity constraint, and the first-order condition is exactly $u\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta) = \varepsilon_i^\times(s)$. Because \(u_q<0\) and \(\mathcal G_i(\theta)>0\), the objective is strictly concave in \(q_i^\times(\theta,s)\) pointwise, so this first-order condition characterizes a unique allocation for each \((\theta,s)\).

Finally, Assumption~\ref{ass:interior} also imposes that \(\mathcal G_i(\theta)\) is nondecreasing. Differentiating the first-order condition with respect to \(\theta\) gives
\[
\frac{\partial q_i^\times(\theta,s)}{\partial \theta}
=
-
\frac{u\!\left(q_i^\times(\theta,s),s\right)}{u_q\!\left(q_i^\times(\theta,s),s\right)}
\frac{\mathcal G_i'(\theta)}{\mathcal G_i(\theta)}
\ge 0,
\]
so the monotonicity constraint is satisfied. This proves part (i).

For part (ii), if \eqref{Rmdi-def} is slack at \((k_i,I_i)\), complementary slackness implies \(\beta_i^\times=0\). Hence $\mathcal G_i(\theta)=\Gamma_i(\theta)+\beta_i^\times J_i(\theta)\equiv \Gamma_i(\theta)$. This proves part (ii).
\end{proof}

\section{Single crossing}


\begin{proof}
For each binding state $s\in T_i^\times(k_i)$ and $k_i$, defined
\[
H_i(k_i,s)
:=
\mathbb E_i^w[uJ\mid s].
\]
and
\[
\mathcal S_i(k_i)
:=
\int_{s_i^\times(k_i)}^{\overline s}
H_i(k_i,s)\,f(s)\,ds.
\]

\medskip
\noindent
For every binding state $s\in T_i^\times(k_i)$, define
\[
dP_{k_i,s}(\theta)
:=
\frac{R_q(\theta,k_i,s)\,g_i(\theta)}
{\mathbb E_{\theta_i}[R_q(\theta,k_i,s)]}\,d\theta.
\]
Since
\[
R_\theta(\theta,k_i)
=
\frac{J_i(\theta)}{\mathcal G_i(\theta,k_i)},
\]
we can write
\[
H_i(k_i,s)
=
\frac{\mathbb E_{\theta_i}\!\left[R_q(\theta,k_i,s)\,R_\theta(\theta,k_i)\right]}
{\mathbb E_{\theta_i}\!\left[R_q(\theta,k_i,s)\right]}
=
\int_{\Theta_i} R_\theta(\theta,k_i)\,dP_{k_i,s}(\theta).
\]

Differentiate $H_i(k_i,s)$ with respect to $k_i$. This gives
\[
\frac{\partial H_i(k_i,s)}{\partial k_i}
=
M_i(k_i,s)+D_i(k_i,s),
\]
where
\[
M_i(k_i,s)
:=
\left.
\frac{\partial}{\partial t}
\int_{\Theta_i} R_\theta(\theta,k_i)\,dP_{t,s}(\theta)
\right|_{t=k_i}
\]
is the pure measure-shift term, and
\[
D_i(k_i,s)
:=
\int_{\Theta_i}
\frac{\partial R_\theta(\theta,k_i)}{\partial k_i}
\,dP_{k_i,s}(\theta)
\]
is the direct effect through the $k_i$-dependence of $R_\theta$.

By the log-monotone-differences assumption on $R_q$, the family $\{P_{k_i,s}\}_{k_i}$ is ordered by monotone likelihood ratio for each binding state $s$. Hence, since $\theta\mapsto R_\theta(\theta,k_i)$ is monotone for each $k_i$,
\[
M_i(k_i,s)\ge 0
\]
in the two cases
\[
R_\theta(\cdot,k_i)\ \text{increasing and }R_q\ \text{log-supermodular},
\qquad
R_\theta(\cdot,k_i)\ \text{decreasing and }R_q\ \text{log-submodular},
\]
while
\[
M_i(k_i,s)\le 0
\]
in the two opposite cases.

Next, since
\[
R_\theta(\theta,k_i)
=
\frac{J_i(\theta)}{\mathcal G_i(\theta,k_i)},
\qquad
\mathcal G_i(\theta,k_i)
=
\Gamma_i(\theta)+J_i(\theta)\beta_i^\times(k_i),
\]
we have
\[
\frac{\partial R_\theta(\theta,k_i)}{\partial k_i}
=
-\beta_i^{\times\prime}(k_i)\,
\frac{J_i(\theta)^2}{\mathcal G_i(\theta,k_i)^2}.
\]
Therefore
\[
D_i(k_i,s)
=
-\beta_i^{\times\prime}(k_i)\,A_i(k_i,s),
\]
where
\[
A_i(k_i,s)
:=
\int_{\Theta_i}
\frac{J_i(\theta)^2}{\mathcal G_i(\theta,k_i)^2}
\,dP_{k_i,s}(\theta)
>0.
\]
Thus
\[
\frac{\partial H_i(k_i,s)}{\partial k_i}
=
M_i(k_i,s)-\beta_i^{\times\prime}(k_i)\,A_i(k_i,s).
\]

\medskip
\noindent
We now relate $\beta_i^{\times\prime}(k_i)$ to $\mathcal S_i(k_i)$. On the IR-constrained region, define
\[
\Phi_i(k_i,\beta)
:=
\mathbb E_s\!\left[
\mathbf 1_{\{s\in T_i^\times(k_i)\}}
\,
\mathbb E_{\theta_i}\!\left[
U\!\left(q_i^\times(\theta,s;k_i,\beta),s\right)J_i(\theta)
\right]
\right]
-I_i.
\]
By construction,
\[
\Phi_i\!\left(k_i,\beta_i^\times(k_i)\right)=0.
\]

Holding $\beta$ fixed, the comparative-statics identity gives, for every binding state $s$,
\[
\frac{\partial q_i^\times(\theta,s;k_i,\beta)}{\partial k_i}
=
\frac{w_i^\times(\theta,s)}{\bar w_i^\times(s)}.
\]
Hence
\[
\partial_{k_i}\Phi_i(k_i,\beta)
=
\mathbb E_s\!\left[
\mathbf 1_{\{s\in T_i^\times(k_i)\}}
\,\mathbb E_i^w[uJ\mid s]
\right]
=
\mathcal S_i(k_i),
\]
where the moving-threshold term vanishes because
\[
u\!\left(q_i^\times(\theta,s_i^\times(k_i)),s_i^\times(k_i)\right)=0
\qquad
\text{for every }\theta.
\]

Holding $k_i$ fixed, differentiating the pointwise FOC
\[
u\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta,k_i)
=
\varepsilon_i^\times(s)
\]
with respect to $\beta$, using $\partial_\beta \mathcal G_i(\theta,k_i)=J_i(\theta)$ and imposing
\[
\mathbb E_{\theta_i}\!\left[\partial_\beta q_i^\times(\theta,s)\right]=0,
\]
yields
\[
\frac{\partial \varepsilon_i^\times(s)}{\partial \beta}
=
\mathbb E_i^w[uJ\mid s]
\]
and
\[
\frac{\partial q_i^\times(\theta,s)}{\partial \beta}
=
w_i^\times(\theta,s)
\left(
u\!\left(q_i^\times(\theta,s),s\right)J_i(\theta)
-
\mathbb E_i^w[uJ\mid s]
\right).
\]
Therefore
\[
\partial_\beta \Phi_i(k_i,\beta)
=
\mathbb E_s\!\left[
\mathbf 1_{\{s\in T_i^\times(k_i)\}}
\,\bar w_i^\times(s)\,\mathrm{Var}_i^w(uJ\mid s)
\right]
=:
\mathcal V_i(k_i),
\]
with $\mathcal V_i(k_i)>0$ away from knife-edge cases.

Differentiating
\[
\Phi_i\!\left(k_i,\beta_i^\times(k_i)\right)=0
\]
with respect to $k_i$ gives
\[
0
=
\mathcal S_i(k_i)+\mathcal V_i(k_i)\,\beta_i^{\times\prime}(k_i),
\]
hence
\[
\beta_i^{\times\prime}(k_i)
=
-\frac{\mathcal S_i(k_i)}{\mathcal V_i(k_i)}.
\]
In particular, at every $k_i^0$ such that $\mathcal S_i(k_i^0)=0$, one has
\[
\beta_i^{\times\prime}(k_i^0)=0.
\]
The same conclusion is immediate on the IR-unconstrained region, where $\beta_i^\times(k_i)\equiv 0$ locally.

\medskip
\noindent
Now differentiate
\[
\mathcal S_i(k_i)
=
\int_{s_i^\times(k_i)}^{\overline s}
H_i(k_i,s)\,f(s)\,ds.
\]
By Leibniz' rule,
\[
\mathcal S_i'(k_i)
=
-\,s_i^{\times\prime}(k_i)\,
H_i\!\left(k_i,s_i^\times(k_i)\right)\,
f\!\left(s_i^\times(k_i)\right)
+
\int_{s_i^\times(k_i)}^{\overline s}
\frac{\partial H_i(k_i,s)}{\partial k_i}\,f(s)\,ds.
\]
At the threshold state $s_i^\times(k_i)$,
\[
u\!\left(q_i^\times(\theta,s_i^\times(k_i)),s_i^\times(k_i)\right)=0
\qquad
\text{for every }\theta,
\]
so
\[
H_i\!\left(k_i,s_i^\times(k_i)\right)=0.
\]
Hence the boundary term vanishes and
\[
\mathcal S_i'(k_i)
=
\overline M_i(k_i)-\beta_i^{\times\prime}(k_i)\,\overline A_i(k_i),
\]
where
\[
\overline M_i(k_i)
:=
\int_{s_i^\times(k_i)}^{\overline s} M_i(k_i,s)\,f(s)\,ds,
\qquad
\overline A_i(k_i)
:=
\int_{s_i^\times(k_i)}^{\overline s} A_i(k_i,s)\,f(s)\,ds
>0.
\]

Now let $k_i^0$ satisfy $\mathcal S_i(k_i^0)=0$. Then $\beta_i^{\times\prime}(k_i^0)=0$, so
\[
\mathcal S_i'(k_i^0)=\overline M_i(k_i^0).
\]
Since each $M_i(k_i^0,s)$ has the same sign across binding states, $\overline M_i(k_i^0)$ has that same sign. Therefore every sign-changing zero of $\mathcal S_i$ is crossed in the same direction:

\[
\mathcal S_i'(k_i^0)>0
\]
when
\[
R_\theta(\cdot,k_i)\ \text{increasing and }R_q\ \text{log-supermodular},
\qquad
R_\theta(\cdot,k_i)\ \text{decreasing and }R_q\ \text{log-submodular},
\]
so any sign change is from $-$ to $+$; and
\[
\mathcal S_i'(k_i^0)<0
\]
in the two opposite cases, so any sign change is from $+$ to $-$.

Thus the map
\[
k_i\longmapsto \mathcal S_i(k_i)
=
\mathbb E_s\!\left[
\mathbb E_i^w[uJ\mid s]\mathbf 1_{\{s\in T_i^\times(k_i)\}}
\right]
\]
is single-crossing in $k_i$, with crossing direction pinned down by the combination of monotonicities stated in the lemma.
\end{proof}
\begin{lemma}[From the aggregated zero to the statewise zero under multiplicative separability]\label{lem:statewise-zero-mult}
Assume \eqref{eq:mult_u}, namely $u(q,s)=\kappa(s)\zeta(q)$, $\kappa(s)>0$. Fix \(k_i\), and let $\mathcal{G}_i(\theta,k_i):=\Gamma_i(\theta)+J_i(\theta)\beta_i^\times(k_i)$, $R_\theta(\theta,k_i):=J_i(\theta)/\mathcal{G}_i(\theta,k_i)$, and $R_q^\times(\theta,s):=-u(q_i^\times(\theta,s),s)/u_q(q_i^\times(\theta,s),s)$, so that $H_i(k_i,s)=\mathbb{E}_{\theta_i}[R_\theta(\theta,k_i)\,R_q^\times(\theta,s)]$ as in the preliminary step. Then \(H_i(k_i,s)\) is constant across all binding states \(s\in T_i^\times(k_i)\). In particular, on the IR-constrained region, $\overline H_i(k_i)=0 \Longleftrightarrow H_i(k_i,s)=0$ for every $s\in T_i^\times(k_i)$, and consequently $\beta_i^{\times\prime}(k_i)=0 \Longleftrightarrow H_i(k_i,s)=0$ for every $s\in T_i^\times(k_i)$.
\end{lemma}

\begin{proof}
Fix \(k_i\). For every binding state \(s\in T_i^\times(k_i)\), the first-order condition under \eqref{eq:mult_u} gives $\kappa(s)\zeta\!\left(q_i^\times(\theta,s)\right)\mathcal{G}_i(\theta,k_i)=\varepsilon_i^\times(s)$. Define $z_i(s):=\varepsilon_i^\times(s)/\kappa(s)$, so $q_i^\times(\theta,s)=\zeta^{-1}(z_i(s)/\mathcal{G}_i(\theta,k_i))$. Differentiating the binding capacity constraint $\mathbb{E}_{\theta_i}[q_i^\times(\theta,s)]=k_i$ with respect to $s$ yields $z_i'(s)=0$ for all $s\in T_i^\times(k_i)$, since $\mathcal{G}_i(\theta,k_i)>0$ and $\zeta_q<0$ make the expectation nonzero. Hence $z_i(s)$ is constant on the binding region, $q_i^\times(\theta,s)$ is $s$-independent there, and therefore $R_q^\times(\theta,s)=-\zeta(q_i^\times(\theta,s))/\zeta_q(q_i^\times(\theta,s))$ is also $s$-independent. Since $R_\theta(\theta,k_i)$ does not depend on $s$, $H_i(k_i,s)$ is constant across $s\in T_i^\times(k_i)$. Denoting this common value $H_i(k_i)$, we get $\overline H_i(k_i)=\Pr(s\in T_i^\times(k_i))\,H_i(k_i)$, so the two equivalences follow immediately from $\Pr(s\in T_i^\times(k_i))>0$ on the IR-constrained region and $\beta_i^{\times\prime}(k_i)=-\overline H_i(k_i)/\overline Q_i(k_i)$ with $\overline Q_i(k_i)>0$.
\end{proof}

\section{Proof of Theorem Individual}

\begin{proof}
Define $H_i(\theta):=\partial\mathbb{E}CS_i^\times(\theta)/\partial k_i$. From the envelope representation \eqref{ICmd-def},
\[H_i(\theta)=H_i(\underline\theta_i)+\int_{\underline\theta_i}^\theta H_i'(\tilde\theta)\,d\tilde\theta, \qquad H_i'(\theta)=\mathbb{E}_s\!\left[u\!\left(q_i^\times(\theta,s),s\right)\frac{\partial q_i^\times(\theta,s)}{\partial k_i}\,\mathbf{1}_{s\in T_i^\times}\right].\]
Under Definition~\ref{ass:multiplicative}, $u(q,s)=\kappa(q)\zeta(s)$ so $u_q=\kappa'(q)\zeta(s)$, and the ratio $R_q(\theta):=-\kappa(q_i^\times(\theta,s))/\kappa'(q_i^\times(\theta,s))$ is $s$-independent at the optimum by Lemma~\ref{lem:statewise-zero-mult}. Using this and the first-order condition $u(q_i^\times,s)\mathcal{G}_i(\theta)=\varepsilon_i^\times(s)$, one verifies $u(q_i^\times,s)w_i^\times(\theta,s)=R_q(\theta)/\mathcal{G}_i(\theta)$ and $u(q_i^\times,s)J_i(\theta)=\varepsilon_i^\times(s)R_\theta(\theta)$. Substituting \eqref{partialkq1} into $H_i'(\theta)$ and applying these identities yields
\[H_i'(\theta)=\frac{R_q(\theta)}{\mathcal{G}_i(\theta)}\!\left(A_i-\beta_i' B_i+\beta_i' C_i R_\theta(\theta)\right),\]
where
\[A_i:=\mathbb{E}_s\!\left[\frac{\mathbf{1}_{s\in T_i^\times}}{\bar w_i^\times(s)}\right]>0, \qquad B_i:=\mathbb{E}_s\!\left[\mathbb{E}^w_i[uJ\mid s]\,\mathbf{1}_{s\in T_i^\times}\right], \qquad C_i:=\mathbb{E}_s\!\left[\varepsilon_i^\times(s)\,\mathbf{1}_{s\in T_i^\times}\right]>0,\]
and $\beta_i':=\partial\beta_i^\times/\partial k_i$. By Definition~\ref{ass:multiplicative} of multiplicative separability $\mathbb{E}^w_i[uJ\mid s]$ is constant across all $s\in T_i^\times$, so $B_i = \Pr(s\in T_i^\times)\cdot\mathbb{E}^w_i[uJ\mid s]$ and $\mathrm{sign}(B_i)=\mathrm{sign}(\eqref{Eujmd})$.

\medskip\noindent\textbf{IR-unconstrained allocation.} Since $\beta_i^\times=0$, we have $\beta_i'=0$ and $\mathcal{G}_i(\theta)\equiv\Gamma_i(\theta)$, so $H_i'(\theta)=R_q(\theta)A_i/\Gamma_i(\theta)>0$ for all $\theta$. Hence $H_i$ is strictly increasing on $[\underline\theta_i,\overline\theta_i]$. By the preliminary step of Lemma~\ref{lem:SC-dir}, $H_i(\underline\theta_i)=\partial\mathbb{E}\underline{CS}_i^\times/\partial k_i$ carries the sign of \eqref{Eujmd}. If \eqref{Eujmd} is positive then $H_i(\underline\theta_i)>0$, and since $H_i$ is strictly increasing, $H_i(\theta)>0$ for all $\theta$: every type gains. If \eqref{Eujmd} is negative then $H_i(\underline\theta_i)<0$, and strict monotonicity implies $H_i$ crosses zero at most once; whenever the crossing $\tilde\theta_i$ exists, $H_i<0$ for $\theta<\tilde\theta_i$ and $H_i>0$ for $\theta>\tilde\theta_i$.

\medskip\noindent\textbf{IR-constrained allocation.}

\noindent\textit{Step 1: initial condition.} Since \eqref{Rmdi-def} binds, differentiating the identity $\mathbb{E}_{(s,\theta_i)}[U(q_i^\times,s)J_i(\theta)]-I_i=0$ with respect to $k_i$ yields $H_i(\underline\theta_i)=\partial\mathbb{E}\underline{CS}_i^\times/\partial k_i=0$.

\noindent\textit{Step 2: monotonicity of $H_i'$ and shape of $H_i$.} Since $R_q(\theta)/\mathcal{G}_i(\theta)>0$, the sign of $H_i'(\theta)$ equals the sign of the affine function $A_i-\beta_i'B_i+\beta_i'C_iR_\theta(\theta)$. If $R_\theta(\theta)$ is monotone this affine function is monotone in $\theta$, hence $H_i'(\theta)$ is monotone in $\theta$ and $H_i$ is either concave or convex. By the preliminary step of Lemma~\ref{lem:SC-dir}, $\mathrm{sign}(\beta_i')=-\mathrm{sign}(\eqref{Eujmd})$, and by the observation above $\mathrm{sign}(B_i)=\mathrm{sign}(\eqref{Eujmd})$, so $\mathrm{sign}(-\beta_i'B_i)=\mathrm{sign}(\eqref{Eujmd})^2\ge 0$: the term $-\beta_i'B_i$ is always non-negative and does not affect the monotonicity direction. The direction is determined by $\beta_i'C_iR_\theta(\theta)$: if $R_\theta(\theta)$ is increasing and \eqref{Eujmd} is positive then $\beta_i'<0$, so $H_i'$ is decreasing and $H_i$ is concave; if \eqref{Eujmd} is negative then $\beta_i'>0$ and $H_i$ is convex. If $R_\theta(\theta)$ is decreasing these conclusions are reversed.

\noindent\textit{Step 3: uniqueness and welfare direction.} Since $H_i(\underline\theta_i)=0$, a concave $H_i$ can depart from zero in only one direction and cross zero at most once, and symmetrically for the convex case. In case~(i) with $R_\theta$ increasing: $H_i$ is concave, rises above zero and then crosses back down, so $H_i>0$ for $\theta\in(\underline\theta_i,\tilde\theta_i)$ and $H_i<0$ for $\theta>\tilde\theta_i$: lower types gain and higher types lose. In case~(ii): $H_i$ is convex, dips below zero and crosses back up, so $H_i<0$ for $\theta<\tilde\theta_i$ and $H_i>0$ for $\theta>\tilde\theta_i$, reversing the welfare ordering. If $R_\theta(\theta)$ is decreasing, concavity and convexity are swapped and all inequalities reverse.
\end{proof}

\section{Proof of Proposition Region}

\begin{proof}
Let $F_i(k_i):=\mathbb{E}\underline{CS}_i^\times(k_i)$. By the preliminary step of Lemma~\ref{lem:SC-dir}, $F_i'(k_i)=\partial\mathbb{E}\underline{CS}_i^\times/\partial k_i=\mathbb{E}_s[\mathbb{E}^w_i[uJ\mid s]\,\mathbf{1}_{s\in T_i^\times}]$. Under the conditions of Lemma~\ref{lem:SC-dir}, the map $k_i\mapsto\mathbb{E}^w_i[uJ\mid s]$ is monotone in $k_i$ for each fixed $s\in T_i^\times$: decreasing when \eqref{Eujmd} crosses from $+$ to $-$, increasing in the opposite case. Since monotonicity is preserved under expectation over $s$, the map $k_i\mapsto F_i'(k_i)$ is itself monotone. When \eqref{Eujmd} crosses from $+$ to $-$, $F_i'$ is decreasing and crosses zero at most once from above, so $F_i$ is single-peaked on $[0,k_i^+]$. The case crossing from $-$ to $+$ gives a single-dipped $F_i$ by the same argument.

At $k_i=0$ the capacity constraint binds at zero, so $q_i^\times(\theta,s)=0$ for all $(\theta,s)$, giving $F_i(0)=-I_i<0$ for any $I_i>0$.

\noindent\textit{Case: \eqref{Eujmd} crosses from $+$ to $-$.} $F_i$ is single-peaked with $F_i(0)<0$. Suppose first that $F_i(k_i^+)\ge 0$. Since $F_i(0)<0$ and $F_i(k_i^+)\ge 0$, continuity gives at least one zero in $(0,k_i^+]$. Single-peakedness implies $F_i$ can cross zero at most once from below on the ascending part and at most once from above on the descending part; but since $F_i(0)<0$ the ascending part must cross zero before the peak, after which $F_i$ is decreasing and cannot cross zero from below again. Hence the zero crossing from below is unique, defining $\tilde k_i$, and $F_i(k_i)\ge 0$ if and only if $k_i\in[\tilde k_i,k_i^+]$. When $J_i(\theta)\ge 0$ for all $\theta$, the revenue integral is non-negative at $k_i^+$, ensuring $F_i(k_i^+)\ge 0$. This proves part~(i). Suppose instead that $F_i(k_i^+)<0$. Then $F_i$ is negative at both endpoints. If the peak value is negative, $F_i<0$ throughout. If the peak value is non-negative, continuity and single-peakedness yield exactly two zeros $\tilde k_i^-<\tilde k_i^+$ in $(0,k_i^+)$ with $F_i(k_i)\ge 0$ if and only if $k_i\in[\tilde k_i^-,\tilde k_i^+]$. This proves part~(ii).

\noindent\textit{Case: \eqref{Eujmd} crosses from $-$ to $+$.} $F_i$ is single-dipped with $F_i(0)<0$. Either the trough value is negative and $F_i<0$ throughout, or $F_i$ climbs above zero after the trough, crossing zero exactly once from below at a unique $\tilde k_i\in(0,k_i^+]$, and $F_i(k_i)\ge 0$ if and only if $k_i\in[\tilde k_i,k_i^+]$.
\end{proof}

\section{Proof of Preferences}

\begin{proof}
Write $\lambda_i^1(\theta):=\lambda_i(\theta)$, $\lambda_i^2(\theta):=\lambda_i^\Delta(\theta)$, and define \[\tilde\lambda_i^k := \int_{\underline\theta_i}^{\overline\theta_i}\lambda_i^k(\theta)\,dG_i(\theta), \qquad \Lambda_i^k(\theta) := \frac{1}{g_i(\theta)}\int_\theta^{\overline\theta_i}\lambda_i^k(t)\,dG_i(t), \qquad k\in\{1,2\}.\] Let $m(\theta):=\lambda_i^2(\theta)/\lambda_i^1(\theta)$, which is increasing in the upward-shift case and decreasing in the downward-shift case.

\medskip\noindent\textit{Core step: monotonicity of $\theta\mapsto\Lambda_i^2(\theta)/\Lambda_i^1(\theta)$.} For each $\theta\in[\underline\theta_i,\overline\theta_i)$, define the probability measure $Q_\theta$ on $[\theta,\overline\theta_i]$ by \[dQ_\theta(t) := \frac{\lambda_i^1(t)\,dG_i(t)}{\displaystyle\int_\theta^{\overline\theta_i}\lambda_i^1(x)\,dG_i(x)}\,\mathbf{1}\{t\ge\theta\}.\] Since $\lambda_i^2(t)=m(t)\lambda_i^1(t)$, \[\frac{\Lambda_i^2(\theta)}{\Lambda_i^1(\theta)} = \mathbb{E}_{Q_\theta}[m(\tilde\theta)].\] For $\theta'<\theta$, $Q_\theta$ is $Q_{\theta'}$ conditioned on $\{\tilde\theta\ge\theta\}$. When $m$ is increasing, this conditioning shifts mass toward larger values of $m$, so by first-order stochastic dominance, $\mathbb{E}_{Q_\theta}[m(\tilde\theta)]\ge\mathbb{E}_{Q_{\theta'}}[m(\tilde\theta)]$. Hence $\theta\mapsto\Lambda_i^2(\theta)/\Lambda_i^1(\theta)$ is increasing on $[\underline\theta_i,\overline\theta_i)$. When $m$ is decreasing the stochastic-dominance inequality reverses and the ratio is decreasing.

\medskip\noindent\textit{Proof of (i).} Since $\theta\mapsto\Lambda_i^2(\theta)/\Lambda_i^1(\theta)$ is increasing, its logarithm is increasing. Therefore, for almost every $\theta\in(\underline\theta_i,\overline\theta_i)$,\footnote{The derivative exists a.e.\ on $(\underline\theta_i,\overline\theta_i)$ under the maintained assumptions that $g_i$ is continuous and strictly positive and $\lambda_i^k$ is continuous; pointwise existence is not guaranteed without additional regularity on the weight functions.}
\[0 \;\le\; \frac{d}{d\theta}\log\!\left(\frac{\Lambda_i^2(\theta)}{\Lambda_i^1(\theta)}\right) = \frac{\Lambda_i^{2\,\prime}(\theta)}{\Lambda_i^2(\theta)} - \frac{\Lambda_i^{1\,\prime}(\theta)}{\Lambda_i^1(\theta)},\]
which yields $\frac{\Lambda_i^{\Delta\,\prime}(\theta)}{\Lambda_i^\Delta(\theta)}\ge\frac{\Lambda_i^{\prime}(\theta)}{\Lambda_i(\theta)}$ for almost every $\theta$. The reverse inequality holds when $m$ is decreasing.

\medskip\noindent\textit{Proof of (ii) under $\beta_i^\times=0$.\footnote{When $\beta_i^\times\neq 0$, the effective weight is $\mathcal{G}_i(\theta)=J_i(\theta)(\tilde\lambda_i+\beta_i^\times)+\Lambda_i(\theta)$. An analogous argument applies with $c=(\tilde\lambda_i^\Delta+\beta_i^{\times,\Delta})/(\tilde\lambda_i+\beta_i^\times)$, but propagating the $\Lambda$ bound to the full $\mathcal{G}$ bound additionally requires $c\le\tilde\lambda_i^\Delta/\tilde\lambda_i$. Since this condition involves the endogenous multiplier $\beta_i^\times$ and cannot in general be verified from primitives alone, the main text restricts to the IR-unconstrained region.}} When $\beta_i^\times=0$, the effective weight reduces to $\Gamma_i(\theta)=J_i(\theta)\tilde\lambda_i+\Lambda_i(\theta)$ and $\Gamma_i^\Delta(\theta)=J_i(\theta)\tilde\lambda_i^\Delta+\Lambda_i^\Delta(\theta)$.

\medskip\noindent\textbf{Upward case.} Assume $m$ is increasing and $\tilde\lambda_i^\Delta\ge\tilde\lambda_i$. Set $c:=\tilde\lambda_i^\Delta/\tilde\lambda_i\ge 1$. The boundary evaluation gives $\Lambda_i^2(\underline\theta_i)/\Lambda_i^1(\underline\theta_i)=\tilde\lambda_i^\Delta/\tilde\lambda_i=c$. Since the ratio is increasing by the core step and starts at $c$, $\Lambda_i^\Delta(\theta)\ge c\,\Lambda_i(\theta)$ for all $\theta$. Writing $\Gamma_i^\Delta(\theta)=c\,J_i(\theta)\tilde\lambda_i+\Lambda_i^\Delta(\theta)\ge c\,\Gamma_i(\theta)$, and since $c\ge 1$ and $\Gamma_i(\theta)>0$: $\Delta\mathcal{G}_i(\theta)\ge(c-1)\Gamma_i(\theta)\ge 0$ for all $\theta\in[\underline\theta_i,\overline\theta_i]$.

\medskip\noindent\textbf{Downward case.} Assume $m$ is decreasing and $\tilde\lambda_i^\Delta\le\tilde\lambda_i$. Set $c:=\tilde\lambda_i^\Delta/\tilde\lambda_i\le 1$. By the core step, $\Lambda_i^\Delta(\theta)\le c\,\Lambda_i(\theta)$ for all $\theta$. Then $\Gamma_i^\Delta(\theta)\le c\,\Gamma_i(\theta)$, and since $c\le 1$ and $\Gamma_i(\theta)>0$: $\Delta\mathcal{G}_i(\theta)\le(c-1)\Gamma_i(\theta)\le 0$ for all $\theta\in[\underline\theta_i,\overline\theta_i]$.\qedhere
\end{proof}

\section{Proof of Proposition PrefTradeoff}
\begin{proof} The proof proceeds in three steps.

\medskip\noindent\textit{Step 1: The elasticity condition implies that $\Delta R_\theta(\theta)$ is monotone.} Differentiating $\Delta R_\theta(\theta) = -R_\Gamma(\theta)\,R_\theta(\theta)$ gives \[\Delta R_\theta'(\theta) = -R_\Gamma(\theta)\,R_\theta'(\theta) - R_\theta(\theta)\,R_\Gamma'(\theta).\] The two terms on the right-hand side may have opposite signs. If they share the same sign, $\Delta R_\theta'(\theta)$ inherits that sign directly. If they have opposite signs, $\text{sign}(\Delta R_\theta'(\theta)) = \text{sign}(-R_\Gamma(\theta)\,R_\theta'(\theta))$ if and only if the first term dominates in absolute value, i.e., $|R_\Gamma(\theta)\,R_\theta'(\theta)| \geq |R_\theta(\theta)\,R_\Gamma'(\theta)|$, which is equivalent to $\left|\frac{R_\Gamma'(\theta)}{R_\Gamma(\theta)}\right| \leq \left|\frac{R_\theta'(\theta)}{R_\theta(\theta)}\right|$. Thus, in both cases, the elasticity condition implies $\text{sign}(\Delta R_\theta'(\theta)) = \text{sign}(-R_\Gamma(\theta)\,R_\theta'(\theta))$ for all $\theta$. Since $\Delta\mathcal{G}_i(\theta)$ has a constant sign, $R_\Gamma(\theta)$ has a constant sign; since $R_\theta(\theta)$ is monotone, $R_\theta'(\theta)$ has a constant sign. Therefore $\text{sign}(-R_\Gamma(\theta)\,R_\theta'(\theta))$ is constant, and $\Delta R_\theta(\theta)$ is globally monotone.

\medskip\noindent\textit{Step 2: Monotonicity of $\Delta R_\theta(\theta)$ determines the sign of the covariance.} By Step~1, both $\Delta R_\theta(\theta)$ and $R_q(\theta)$ are monotone. The standard covariance-monotonicity lemma states that two monotone functions on a probability space have non-negative covariance if they are co-monotone and non-positive covariance if they are counter-monotone. Therefore \[\mathrm{sign}\!\left(\mathrm{Cov}_{\theta_i}\!\left(\Delta R_\theta(\theta),\,R_q(\theta)\right)\right) = \mathrm{sign}\!\left(\Delta R_\theta'(\theta)\right) \cdot \mathrm{sign}\!\left(R_q'(\theta)\right) = \mathrm{sign}\!\left(-R_\Gamma(\theta)\,R_\theta'(\theta)\right) \cdot \mathrm{sign}\!\left(R_q'(\theta)\right).\]

\medskip\noindent\textit{Step 3: Necessity of the elasticity condition.} If the elasticity condition is violated at some $\theta_0$, i.e., $|R_\theta(\theta_0)\,R_\Gamma'(\theta_0)| > |R_\Gamma(\theta_0)\,R_\theta'(\theta_0)|$, then $\Delta R_\theta'(\theta_0)$ has the opposite sign to $-R_\Gamma(\theta_0)\,R_\theta'(\theta_0)$. This creates a local reversal in the monotonicity of $\Delta R_\theta(\theta)$, so the covariance-monotonicity lemma no longer applies. The sign of $\mathrm{Cov}_{\theta_i}(\Delta R_\theta(\theta),\,R_q(\theta))$ then depends on the specific distribution $F_i$ and the magnitude of the reversal, and no distribution-free sign result obtains.
\end{proof}

\section{Proof of Mean deviation of $R_{\mathcal{G}}$}

\begin{proof}
When $\beta_i^\times=0$, one has $\mathcal{G}_i(\theta)=\Gamma_i(\theta)$ and therefore $R_{\mathcal{G}}(\theta)=R_\Gamma(\theta)$; the result in the main text is stated in terms of $R_\Gamma$ for that case. Fix \(s\in T_i^\times\) and a local perturbation \(\Delta\lambda_i(\theta)\) holding \(k_i\) fixed. Using the weights $w_i^\times(\theta,s)$ and $\mathbb{E}^w_i[\cdot\mid s]$ from the preliminary step of Lemma~\ref{lem:SC-dir}, note that $R_q(\theta,s)=\varepsilon_i^\times(s)\,w_i^\times(\theta,s)$, which follows from $R_q(\theta,s)=-u(q_i^\times,s)/u_q(q_i^\times,s)$ and $\varepsilon_i^\times(s)=u(q_i^\times,s)\mathcal{G}_i(\theta)$.

Linearizing the pointwise first-order condition $u(q_i^\times(\theta,s),s)\,\mathcal{G}_i(\theta)=\varepsilon_i^\times(s)$ gives
\[
u_q\!\left(q_i^\times(\theta,s),s\right)\,\mathcal{G}_i(\theta)\,\Delta q_i^\times(\theta,s)
+
u\!\left(q_i^\times(\theta,s),s\right)\,\Delta\mathcal{G}_i(\theta)
-
\Delta\varepsilon_i^\times(s)
=
0.
\]
Solving for \(\Delta q_i^\times(\theta,s)\) and substituting the definitions of \(R_q\), $R_{\mathcal{G}}(\theta)=\Delta\mathcal{G}_i(\theta)/\mathcal{G}_i(\theta)$, and $\varepsilon_i^\times(s)$ yields
\[
\Delta q_i^\times(\theta,s)
=
R_q(\theta,s)\left(
R_{\mathcal{G}}(\theta)-\frac{\Delta\varepsilon_i^\times(s)}{\varepsilon_i^\times(s)}
\right).
\]

Since the capacity constraint binds state-by-state and \(k_i\) is held fixed, $\mathbb{E}_{\theta_i}\!\left[\Delta q_i^\times(\theta,s)\right]=0$ for each $s\in T_i^\times$. Substituting the previous expression and using $R_q(\theta,s)=\varepsilon_i^\times(s)\,w_i^\times(\theta,s)$, dividing by $\varepsilon_i^\times(s)\bar{w}_i^\times(s)>0$ and rearranging gives
\[
\frac{\Delta\varepsilon_i^\times(s)}{\varepsilon_i^\times(s)}
=
\mathbb{E}^w_i\!\left[R_{\mathcal{G}}\mid s\right].
\]
Substituting back yields
\[
\Delta q_i^\times(\theta,s)
=
R_q(\theta,s)\left(
R_{\mathcal{G}}(\theta)-\mathbb{E}^w_i\!\left[R_{\mathcal{G}}\mid s\right]
\right),
\]
as claimed.
\end{proof}

\section{Proof of Proposition rprefreallocation}
\subsection{Mean Preserving}

\begin{proof}
From the first-order condition \eqref{KKT-qMD}, we derive the change in the optimal allocation as

\[\Delta q (\theta,s) = \frac{\Delta \varepsilon(s)-u(q^\times_i,s) \Delta\Gamma_i(\theta)}{u_q(q^\times_i,s)\Gamma_i(\theta)}\]

From the capacity constraint we have:

\[ \E_{\theta_i}[\Delta q (\theta,s)] = 0 \leftrightarrow  \E_{\theta_i}\left[\frac{\Delta \varepsilon(s)-u(q^\times_i,s) \Delta\Gamma_i(\theta)}{u_q(q^\times_i,s)\Gamma_i(\theta)}\right] = 0\]

This yields

\[\Delta \varepsilon(s)=\E_{\theta_i}\left[\frac{u(q^\times_i,s) \Delta\Gamma_i(\theta)}{u_q(q^\times_i,s)\Gamma_i(\theta)}\right]/ \E_{\theta_i}\left[\frac{1}{u_q(q^\times_i,s)\Gamma_i(\theta)} \right]\]

Hence if $\Delta \Gamma_i > 0$ for all $\theta$, then $\Delta \varepsilon(s)>0$, and $\Delta \varepsilon(s)<0$ if $\Delta \Gamma_i < 0$ for all $\theta$. We start by noting that if $\Delta\Gamma_i$ is mean preserving then $\Delta\Gamma_i(\underline{\theta}_i) = \Delta\Gamma_i(\overline{\theta})=0$. Indeed note that $\Gamma_i(\underline{\theta}_i) = \tilde{\lambda}_i(\underline{\theta}_i-1/g(\underline{\theta}_i)) + \tilde{\lambda}_i/g(\underline{\theta}_i)= \tilde{\lambda}_i\underline{\theta}_i$ and $\Gamma_i(\overline{\theta}) = \tilde{\lambda}_i\overline{\theta}$ as $\Lambda_i(\overline{\theta}) = 0$ and $\gamma_i(\overline{\theta})=0$.

Therefore if $\Delta\Gamma_i>0$ for all $\theta$ and $\Delta\lambda_i(\theta)$ is mean preserving, then $\Delta q (\theta,s)<0$ at both $\underline{\theta}_i$ and $\overline{\theta}$ (recall that $u_q(q^\times_i,s)\Gamma_i(\theta) \le 0$). Next we show that if $R'_\Gamma$ is single peaked then the term $\Delta \varepsilon(s)-u(q^\times_i,s) \Delta\Gamma_i(\theta)$ is quasi-convex. The derivative of this term with respect to $\theta$ yields

\[-\frac{\partial q^\times_i}{\partial \theta} \; u_q(q^\times_i,s) \;  \Delta\Gamma_i(\theta)-u(q^\times_i,s) \; \Delta\Gamma_i'(\theta)\]

Using $\partial_\theta q_i^\times = -u(q_i^\times,s)\mathcal{G}_i'(\theta)/(u_q(q_i^\times,s)\mathcal{G}_i(\theta))$, obtained by differentiating \eqref{KKT-qMD} with respect to $\theta$, yields

\[\frac{u(q^\times_i,s)}{u_q(q^\times_i,s)} \frac{\Gamma_i'(\theta)}{\Gamma_i(\theta)} \; u_q(q^\times_i,s) \;  \Delta\Gamma_i(\theta)-u(q^\times_i,s) \; \Delta\Gamma_i'(\theta)\]

Simplifying and noting that $u(\cdot)\ge 0$ gives that the sign of the term is provided by the sign of

\[ \Gamma_i'(\theta) \;  \Delta\Gamma_i(\theta)- \; \Gamma_i(\theta) \Delta\Gamma_i'(\theta)\]

Note that $R'_\Gamma(\theta) = \frac{\partial}{\partial \theta}\left(\frac{\Delta\Gamma_i(\theta)}{\Gamma_i(\theta)} \right)= \frac{ \Delta\Gamma_i'(\theta)\Gamma_i(\theta)-\Gamma_i'(\theta)\Delta\Gamma_i(\theta)}{ (\Gamma_i(\theta))^2}$, so the sign of the term is given by the sign of $-R'_\Gamma(\theta)$. If $R'_\Gamma(\theta)$ is single peaked, then $R'_\Gamma$ changes sign at most once from $+$ to $-$. To see this we have $R'_\Gamma(\underline{\theta}_i)=-\Delta \lambda_i(\theta)$ and $R'_\Gamma(\overline{\theta})=-\Delta \lambda_i(\theta)$ with a mean preserving perturbation. Under the same assumption such that $\Delta\Gamma_i>0$, then $\lambda_i(\theta)$ decreases for lower types ($R'_\Gamma(\underline{\theta}_i)>0$) and increases for higher types ($R'_\Gamma(\overline{\theta})<0$). It yields that the term $\Delta \varepsilon(s)-u(q^\times_i,s) \Delta\Gamma_i(\theta)$ changes sign at most once from $-$ to $+$. Hence, $\Delta\Gamma_i(\theta)$ is quasi-convex. Since $u_q<0$, $\Delta q^\times_i$ is quasi-concave. We conclude by showing that this yields the two cutoffs $\theta_-$ and $\theta_+$ such that $\Delta\Gamma_i(\theta)>0$ for all $\theta$, $\Delta q^\times_i(\theta)<0$ for $\theta \in [\underline{\theta}_i,\theta_-)\cup(\theta_+,\overline{\theta}]$ and $\Delta q^\times_i(\theta)>0$ for $\theta \in [\theta_-,\theta_+]$.

Indeed, for a given $k$, we must have $\mathbb{E}_{\theta_i}[q^\times_i(\theta,s)] = k$, therefore $\mathbb{E}_{\theta_i}[\Delta q (\theta,s)] = 0$. Therefore, if the sign of $\Delta q^\times_i(\theta,s)$ is negative at the boundaries and we must have some positive values in an interior interval. As the sign of $\Delta q^\times_i(\theta,s)$ is quasi-concave, then the interval is unique which gives the cutoffs $\theta_-$ and $\theta_+$. The same analysis yields the desired result when $\Delta \Gamma_i <0$ for all $\theta$.
\end{proof}

\subsection{Non-monotonic first-order response to a preference perturbation}

\begin{proof}
By Lemma~\ref{lem:realloc}, under $\beta_i^\times=0$ (so that $R_{\mathcal{G}}=R_\Gamma$), for each $s\in T_i^\times$:
\[
\Delta q_i^\times(\theta,s)=R_q(\theta,s)\!\left(R_\Gamma(\theta)-\mathbb{E}^w_i\!\left[R_\Gamma\mid s\right]\right).
\]
Since $\mathbb{E}^w_i[R_\Gamma\mid s]$ is a weighted average of $R_\Gamma$ with positive weights, if $R_\Gamma$ is not (a.e.) constant then the centered term $R_\Gamma(\theta)-\mathbb{E}^w_i[R_\Gamma\mid s]$ must take both positive and negative values on $\Theta_i$; otherwise it would be nonnegative (or nonpositive) a.e., with equality only if $R_\Gamma$ is constant a.e. Hence $\Delta q_i^\times(\theta,s)$ also takes both signs since $R_q(\theta,s)>0$.

Next, recall that $\Gamma_i(\underline\theta_i)=\tilde\lambda_i\,\underline\theta_i$ and $\Gamma_i(\overline\theta_i)=\tilde\lambda_i\,\overline\theta_i$ (because $G_i(\underline\theta_i)=0$ implies $\Lambda_i(\underline\theta_i)=0$, while $\gamma_i(\overline\theta_i)=0$ and $\Lambda_i(\overline\theta_i)=\tilde\lambda_i$ imply $h_i(\overline\theta_i)=0$). Therefore $R_\Gamma(\underline\theta_i)=R_\Gamma(\overline\theta_i)$ for any perturbation, and hence
\[
\bigl(R_\Gamma-\mathbb{E}^w_i[R_\Gamma\mid s]\bigr)(\underline\theta_i)
=
\bigl(R_\Gamma-\mathbb{E}^w_i[R_\Gamma\mid s]\bigr)(\overline\theta_i).
\]
Since $\Delta q_i^\times(\theta,s)=R_q(\theta,s)(R_\Gamma(\theta)-\mathbb{E}^w_i[R_\Gamma\mid s])$ takes both signs while its endpoint values have the same sign (or are both zero), it cannot be monotone on $\Theta_i$. Finally, because $\Delta q_i^\times(\cdot,s)$ is differentiable, $\partial_\theta\Delta q_i^\times(\cdot,s)$ cannot have a fixed sign; using $\Delta(\partial_\theta q_i^\times)=\partial_\theta\Delta q_i^\times$, it follows that $\Delta(\partial_\theta q_i^\times)$ takes both positive and negative values on $\Theta_i$.

For the unimodal mean-preserving case, Lemma~\ref{lem:realloc} gives $\Delta q_i^\times(\theta,s)=R_q(\theta,s)(R_\Gamma(\theta)-\mathbb{E}^w_i[R_\Gamma\mid s])$. The boundary values satisfy $\Delta q_i^\times(\underline\theta_i,s)=\Delta q_i^\times(\overline\theta_i,s)\le 0$ whenever $\Delta\mathcal{G}_i\ge 0$ (since the mean-preserving condition forces $R_\Gamma(\underline\theta_i)=R_\Gamma(\overline\theta_i)=\Delta\tilde\lambda_i/\tilde\lambda_i$ which under $\Delta\tilde\lambda_i=0$ equals zero, so the boundary values equal zero). The feasibility condition $\mathbb{E}_{\theta_i}[\Delta q_i^\times(\theta,s)]=0$ and the quasi-concavity of $\Delta q_i^\times$ established in the preceding subsection then yield the unique cutoffs $\theta_-<\theta_+$ with the stated allocation properties.
\end{proof}

\section{Proof of Corollary CARA}

\begin{proof}
Assume $\beta_i^\times=0$. In any on-peak state $s\in T_i^\times$ where the capacity constraint binds with multiplier $\varepsilon_i^\times(s)>0$, the interior FOC is
\begin{equation}\label{eq:FOC-CARA}
u\!\left(q_i^\times(\theta,s),s\right)\,\Gamma_i(\theta)=\varepsilon_i^\times(s).
\end{equation}

\paragraph{Step 1 (CARA implies $R_q$ is constant and the behavioral channel vanishes).}
Under CARA, $u_q=-\alpha u$ for some $\alpha>0$, hence $R_q(\theta):=-u/u_q=1/\alpha$ is constant. Therefore
\[
\mathbb{E}_{\theta_i}\!\left[R_\theta(\theta)R_q(\theta)\right]
=\frac{1}{\alpha}\,\mathbb{E}_{\theta_i}\!\left[R_\theta(\theta)\right],
\qquad
\Delta\mathbb{E}_{\theta_i}\!\left[R_\theta(\theta)R_q(\theta)\right]
=\frac{1}{\alpha}\,\mathbb{E}_{\theta_i}\!\left[\Delta R_\theta(\theta)\right].
\]
Since $R_q$ is constant, any covariance term involving $R_q$ equals zero. For a preference perturbation $\Delta\lambda_i(\theta)$ with $R_\Gamma(\theta):=\Delta\Gamma_i(\theta)/\Gamma_i(\theta)$, we have $\Delta R_\theta(\theta)=\Delta(J_i(\theta)/\Gamma_i(\theta))=-R_\Gamma(\theta)\,R_\theta(\theta)$.

\paragraph{Step 2 (CARA simplification and covariance computation).}
By Lemma~\ref{lem:realloc} with $\beta_i^\times=0$: $\Delta q_i^\times(\theta,s)=R_q(\theta,s)(R_\Gamma(\theta)-\mathbb{E}^w_i[R_\Gamma\mid s])$. Under CARA, the weights $w_i^\times(\theta,s)$ from the preliminary step of Lemma~\ref{lem:SC-dir} simplify: since $u_q=-\alpha u$, from \eqref{eq:FOC-CARA} one has $w_i^\times(\theta,s)=1/(\alpha u\Gamma_i(\theta))=1/(\alpha\varepsilon_i^\times(s))$, which is constant in $\theta$. Therefore $\mathbb{E}^w_i[R_\Gamma\mid s]=\mathbb{E}_{\theta_i}[R_\Gamma(\theta)]$ (the weighted average reduces to an unweighted average) and $\Delta q_i^\times(\theta,s)=(1/\alpha)(R_\Gamma(\theta)-\mathbb{E}_{\theta_i}[R_\Gamma(\theta)])$.

Now compute $\Delta\mathbb{E}_{\theta_i}[U(q_i^\times(\theta,s),s)J_i(\theta)]=\mathbb{E}_{\theta_i}[u(q_i^\times,s)J_i(\theta)\Delta q_i^\times(\theta,s)]$. Using $u(q_i^\times,s)=\varepsilon_i^\times(s)/\Gamma_i(\theta)$ from \eqref{eq:FOC-CARA}:
\[
\Delta\mathbb{E}_{\theta_i}\!\left[U\!\left(q_i^\times,s\right)J_i(\theta)\right]
=\frac{\varepsilon_i^\times(s)}{\alpha}\mathbb{E}_{\theta_i}\!\left[\frac{J_i(\theta)}{\Gamma_i(\theta)}\left(R_\Gamma(\theta)-\mathbb{E}_{\theta_i}[R_\Gamma(\theta)]\right)\right]
=\frac{\varepsilon_i^\times(s)}{\alpha}\,\mathrm{Cov}_{\theta_i}\!\left(R_\Gamma(\theta),R_\theta(\theta)\right).
\]
Averaging over $s$ yields
\[
\Delta\mathbb{E}_{s,\theta_i}\!\left[U\!\left(q_i^\times,s\right)J_i(\theta)\right]
=\frac{\mathbb{E}_s[\varepsilon_i^\times(s)]}{\alpha}\,\mathrm{Cov}_{\theta_i}\!\left(R_\Gamma(\theta),R_\theta(\theta)\right),
\]
since off-peak states satisfy $\varepsilon_i^\times(s)=0$.

\paragraph{Step 3 (comparative statics of $k^\ast$).}
Let $F(k):=\partial_k\mathbb{E}\underline{CS}_i^\times(k)$. Single-peakedness and an interior maximizer imply $F(k^\ast)=0$ and $F'(k^\ast)<0$, so $\mathrm{sign}(\Delta k^\ast)=\mathrm{sign}(\Delta F(k^\ast))$. Under CARA and $\beta_i^\times=0$, the envelope expression implies $F(k)=\mathbb{E}_s[\varepsilon_i^\times(s)]\,\mathbb{E}_{\theta_i}[R_\theta(\theta)]$. Using Step~1 and Step~2:
\[
\Delta F(k)
=-\,\mathbb{E}_s\!\left[\varepsilon_i^\times(s)\right]\,
\mathrm{Cov}_{\theta_i}\!\left(R_\Gamma(\theta),R_\theta(\theta)\right).
\]
Whenever $\max_k\mathbb{E}\underline{CS}_i^\times(k)>0$, peak capacity binds with positive probability, so $\mathbb{E}_s[\varepsilon_i^\times(s)]>0$ on the relevant region. Therefore $\mathrm{sign}(\Delta k^\ast)=-\mathrm{sign}(\mathrm{Cov}_{\theta_i}(R_\Gamma(\theta),R_\theta(\theta)))$, which is the desired conclusion.
\end{proof}

\subsubsection{Proof details for Lemma~\ref{lem:suffstat}}
\label{app:suffstat-proof}

\begin{proof}[Proof details for Lemma~\ref{lem:suffstat}]
For \(s\in S_i^\times\), the capacity constraint is slack, so
\(\varepsilon_i^\times(s)=0\) and therefore
\[
u\!\left(q_i^\times(\theta,s),s\right)=0.
\]
Differentiating with respect to \(k_i\) gives
\[
\frac{\partial q_i^\times(\theta,s)}{\partial k_i}=0.
\]

Now fix a binding state \(s\in T_i^\times\). Differentiating the pointwise
first-order condition with respect to \(k_i\) yields
\[
u_q\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta)\,
\frac{\partial q_i^\times(\theta,s)}{\partial k_i}
+
u\!\left(q_i^\times(\theta,s),s\right)J_i(\theta)\,
\frac{\partial \beta_i^\times}{\partial k_i}
-
\frac{\partial \varepsilon_i^\times(s)}{\partial k_i}=0.
\]
Define
\[
w_i^\times(\theta,s):=
-\frac{1}{u_q\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta)},
\qquad
\overline w_i^\times(s):=\E_{\theta_i}\!\left[w_i^\times(\theta,s)\right].
\]
Then
\[
\frac{\partial q_i^\times(\theta,s)}{\partial k_i}
=
-w_i^\times(\theta,s)\frac{\partial \varepsilon_i^\times(s)}{\partial k_i}
+
w_i^\times(\theta,s)\,u\!\left(q_i^\times(\theta,s),s\right)J_i(\theta)\,
\frac{\partial \beta_i^\times}{\partial k_i}.
\]

Differentiating the binding capacity constraint
\[
\E_{\theta_i}[q_i^\times(\theta,s)]=k_i
\]
gives
\[
1=\E_{\theta_i}\!\left[\frac{\partial q_i^\times(\theta,s)}{\partial k_i}\right].
\]
Hence
\[
\frac{\partial \varepsilon_i^\times(s)}{\partial k_i}
=
-\frac{1}{\overline w_i^\times(s)}
+
\frac{\partial \beta_i^\times}{\partial k_i}\,
\E_{w_i}\!\left[uJ\mid s\right],
\]
where
\[
\E_{w_i}\!\left[X\mid s\right]
:=
\frac{1}{\overline w_i^\times(s)}
\E_{\theta_i}\!\left[w_i^\times(\theta,s)X(\theta,s)\right].
\]
Substituting back yields the decomposition
\[
\frac{\partial q_i^\times(\theta,s)}{\partial k_i}
=
\frac{w_i^\times(\theta,s)}{\overline w_i^\times(s)}
+
w_i^\times(\theta,s)\frac{\partial \beta_i^\times}{\partial k_i}
\left(
u\!\left(q_i^\times(\theta,s),s\right)J_i(\theta)
-
\E_{w_i}\!\left[uJ\mid s\right]
\right).
\]
This is the expression used in the text to separate the pure expansion term from
the reallocation term.

Finally, if the IR constraint binds, then
\[
\E_{s,\theta_i}\!\left[U\!\left(q_i^\times(\theta,s),s\right)J_i(\theta)\right]=I_i.
\]
Differentiating with respect to \(k_i\) gives
\[
0=
\E_s\!\left[
\E_{\theta_i}\!\left[
u\!\left(q_i^\times(\theta,s),s\right)J_i(\theta)\,
\frac{\partial q_i^\times(\theta,s)}{\partial k_i}
\right]
\mathbf 1_{\{s\in T_i^\times\}}
\right].
\]
Substituting the previous decomposition and using the definition of
\(\E_{w_i}[\cdot\mid s]\), we obtain
\[
0=
\E_s\!\left[\E_{w_i}[uJ\mid s]\mathbf 1_{\{s\in T_i^\times\}}\right]
+
\frac{\partial \beta_i^\times}{\partial k_i}
\E_s\!\left[
\overline w_i^\times(s)\Var_{w_i}(uJ\mid s)\mathbf 1_{\{s\in T_i^\times\}}
\right].
\]
Hence
\[
\frac{\partial \beta_i^\times}{\partial k_i}
=
-
\frac{
\E_s\!\left[\E_{w_i}[uJ\mid s]\mathbf 1_{\{s\in T_i^\times\}}\right]
}{
\E_s\!\left[
\overline w_i^\times(s)\Var_{w_i}(uJ\mid s)\mathbf 1_{\{s\in T_i^\times\}}
\right]
}.
\]
The denominator is nonnegative by construction, and strictly positive except in
knife-edge cases.
\end{proof}

\subsubsection{Auxiliary comparative-statics identities for the single-crossing and welfare analysis}
\label{app:aux-identities}

The identities in this subsection are not needed for
Lemma~\ref{lem:suffstat}. They are used instead to characterize how the marginal
capacity expansion varies across types, and to relate these shape properties to
the objects \(R_q\) and \(R_\theta\).

\begin{proof}[Auxiliary comparative-statics identities]
Fix a binding state \(s\in T_i^\times\). In this proof, treat
\(\mathcal G_i(\theta)\) as fixed with respect to \(k_i\). In particular, this
covers the IR-unconstrained region, where \(\beta_i^\times=0\) and
\(\mathcal G_i(\theta)\equiv \Gamma_i(\theta)\).

The pointwise first-order condition is
\[
u\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta)=\varepsilon_i^\times(s).
\]

Differentiating with respect to \(k_i\) gives the marginal expansion
\[
u_q\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta)\,
\frac{\partial q_i^\times(\theta,s)}{\partial k_i}
=
\frac{\partial \varepsilon_i^\times(s)}{\partial k_i},
\]
hence
\[
\frac{\partial q_i^\times(\theta,s)}{\partial k_i}
=
\frac{\partial \varepsilon_i^\times(s)/\partial k_i}
{u_q\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta)}.
\]

Differentiating instead with respect to \(\theta\) yields the slope of the
allocation in type space:
\[
u_q\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta)\,
\frac{\partial q_i^\times(\theta,s)}{\partial \theta}
+
u\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i'(\theta)=0,
\]
so
\[
\frac{\partial q_i^\times(\theta,s)}{\partial \theta}
=
-\frac{u\!\left(q_i^\times(\theta,s),s\right)}
{u_q\!\left(q_i^\times(\theta,s),s\right)}
\frac{\mathcal G_i'(\theta)}{\mathcal G_i(\theta)}
=
R_q(\theta)\frac{\mathcal G_i'(\theta)}{\mathcal G_i(\theta)}.
\]

We now study how the marginal capacity expansion varies across types.
Differentiating \(\partial q_i^\times(\theta,s)/\partial k_i\) with respect to
\(\theta\), and using that \(\partial \varepsilon_i^\times(s)/\partial k_i\) does
not depend on \(\theta\), gives
\[
\frac{\partial^2 q_i^\times(\theta,s)}{\partial k_i\,\partial \theta}
=
-\frac{\partial \varepsilon_i^\times(s)/\partial k_i}
{\left(u_q\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta)\right)^2}
\left[
\mathcal G_i'(\theta)\,u_q\!\left(q_i^\times(\theta,s),s\right)
+
u_{qq}\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta)
\frac{\partial q_i^\times(\theta,s)}{\partial \theta}
\right].
\]
Substituting the expression for \(\partial q_i^\times(\theta,s)/\partial \theta\)
yields
\[
\frac{\partial^2 q_i^\times(\theta,s)}{\partial k_i\,\partial \theta}
=
\frac{\partial \varepsilon_i^\times(s)/\partial k_i}
{u_q\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta)}
\frac{\mathcal G_i'(\theta)}{\mathcal G_i(\theta)}
\frac{\partial}{\partial q}
\left(
-\frac{u\!\left(q_i^\times(\theta,s),s\right)}
{u_q\!\left(q_i^\times(\theta,s),s\right)}
\right).
\]
Equivalently,
\[
\frac{\partial^2 q_i^\times(\theta,s)}{\partial k_i\,\partial \theta}
=
-
\frac{\partial q_i^\times(\theta,s)}{\partial k_i}
\frac{\partial q_i^\times(\theta,s)}{\partial \theta}
\frac{u_q\!\left(q_i^\times(\theta,s),s\right)}
{u\!\left(q_i^\times(\theta,s),s\right)}
\frac{\partial R_q(\theta)}{\partial q}.
\]
Thus, the sign of the cross-derivative is governed by the slope of \(R_q\):
\[
\sign\!\left(
\frac{\partial^2 q_i^\times(\theta,s)}{\partial k_i\,\partial \theta}
\right)
=
\sign\!\left(\frac{\partial R_q(\theta)}{\partial q}\right).
\]
This identity is the one used in the text to study whether a marginal increase
in capacity is tilted toward low or high types.

Next, consider the type profile of the marginal surplus effect
\[
u\!\left(q_i^\times(\theta,s),s\right)\,
\frac{\partial q_i^\times(\theta,s)}{\partial k_i}.
\]
Differentiating with respect to \(\theta\) yields
\[
\frac{\partial}{\partial \theta}
\left(
u\!\left(q_i^\times(\theta,s),s\right)
\frac{\partial q_i^\times(\theta,s)}{\partial k_i}
\right)
=
u_q\!\left(q_i^\times(\theta,s),s\right)
\frac{\partial q_i^\times(\theta,s)}{\partial \theta}
\frac{\partial q_i^\times(\theta,s)}{\partial k_i}
+
u\!\left(q_i^\times(\theta,s),s\right)
\frac{\partial^2 q_i^\times(\theta,s)}{\partial k_i\,\partial \theta}.
\]
Substituting the previous identity gives
\[
\frac{\partial}{\partial \theta}
\left(
u\!\left(q_i^\times(\theta,s),s\right)
\frac{\partial q_i^\times(\theta,s)}{\partial k_i}
\right)
=
-
u_q\!\left(q_i^\times(\theta,s),s\right)
\frac{\partial q_i^\times(\theta,s)}{\partial \theta}
\frac{\partial q_i^\times(\theta,s)}{\partial k_i}
\left(
\frac{\partial R_q(\theta)}{\partial q}-1
\right).
\]
This is the identity used when studying how the marginal welfare gain from
capacity varies across types.

Finally, we turn to the type variation in marginal virtual surplus. Define
\[
R_\theta(\theta):=\frac{J_i(\theta)}{\mathcal G_i(\theta)}.
\]
Then
\[
\frac{\partial}{\partial \theta}
\left(
u\!\left(q_i^\times(\theta,s),s\right)J_i(\theta)
\right)
=
u_q\!\left(q_i^\times(\theta,s),s\right)
\frac{\partial q_i^\times(\theta,s)}{\partial \theta}J_i(\theta)
+
u\!\left(q_i^\times(\theta,s),s\right)J_i'(\theta).
\]
Substituting the expression for \(\partial q_i^\times(\theta,s)/\partial \theta\)
gives
\[
\frac{\partial}{\partial \theta}
\left(
u\!\left(q_i^\times(\theta,s),s\right)J_i(\theta)
\right)
=
u\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta)
\frac{\partial}{\partial \theta}
\left(
\frac{J_i(\theta)}{\mathcal G_i(\theta)}
\right)
=
u\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta)\,R_\theta'(\theta).
\]
Equivalently,
\[
R_\theta'(\theta)
=
\frac{J_i'(\theta)\mathcal G_i(\theta)-J_i(\theta)\mathcal G_i'(\theta)}
{\mathcal G_i(\theta)^2}.
\]

In the IR-unconstrained region, \(\mathcal G_i(\theta)=\Gamma_i(\theta)\), so
\[
R_\theta'(\theta)
=
\frac{J_i'(\theta)\Gamma_i(\theta)-J_i(\theta)\Gamma_i'(\theta)}
{\Gamma_i(\theta)^2}.
\]
Using \(\Gamma_i(\theta)=\tilde\lambda_i J_i(\theta)+\Lambda_i(\theta)\), hence
\(\Gamma_i'(\theta)=\tilde\lambda_i J_i'(\theta)+\Lambda_i'(\theta)\), we obtain
\[
R_\theta'(\theta)
=
\frac{J_i'(\theta)\Lambda_i(\theta)-J_i(\theta)\Lambda_i'(\theta)}
{\Gamma_i(\theta)^2}.
\]
This is the primitive decomposition used in the text when signing the slope of
\(R_\theta\).
\end{proof}

\subsection{Non-monotonicity of q}

\begin{lemma}\label{lem:boundary-failure-beta}
Suppose $\beta_i^\times>0$. Then, in general,
\[
R_{\mathcal G}(\underline\theta_i)\neq R_{\mathcal G}(\overline\theta_i).
\]
More precisely,
\[
R_{\mathcal G}(\overline\theta_i)
=
\frac{\Delta\tilde\lambda_i+\Delta\beta_i^\times}
{\tilde\lambda_i+\beta_i^\times},
\qquad
R_{\mathcal G}(\underline\theta_i)
=
\frac{\Delta\tilde\lambda_i\,\underline\theta_i+\Delta\beta_i^\times J_i(\underline\theta_i)}
{\tilde\lambda_i\,\underline\theta_i+\beta_i^\times J_i(\underline\theta_i)}.
\]
These two boundary values coincide if and only if
\[
\frac{\Delta\beta_i^\times}{\beta_i^\times}
=
\frac{\Delta\tilde\lambda_i}{\tilde\lambda_i}.
\]
Consequently, when $\beta_i^\times>0$, the endpoint derivative argument does not by itself deliver a strict monotonicity or non-monotonicity result for $\Delta q_i^\times(\theta,s)$.
\end{lemma}

\begin{proof}
By definition,
\[
R_{\mathcal G}(\theta)=\frac{\Delta\mathcal G_i(\theta)}{\mathcal G_i(\theta)},
\qquad
\mathcal G_i(\theta)=\Gamma_i(\theta)+\beta_i^\times J_i(\theta),
\qquad
\Delta\mathcal G_i(\theta)=\Delta\Gamma_i(\theta)+\Delta\beta_i^\times J_i(\theta).
\]

At the upper boundary, $\Gamma_i(\overline\theta_i)=\tilde\lambda_i\overline\theta_i$ and
$J_i(\overline\theta_i)=\overline\theta_i$, hence
\[
\mathcal G_i(\overline\theta_i)
=
(\tilde\lambda_i+\beta_i^\times)\overline\theta_i,
\qquad
\Delta\mathcal G_i(\overline\theta_i)
=
(\Delta\tilde\lambda_i+\Delta\beta_i^\times)\overline\theta_i,
\]
which gives
\[
R_{\mathcal G}(\overline\theta_i)
=
\frac{\Delta\tilde\lambda_i+\Delta\beta_i^\times}
{\tilde\lambda_i+\beta_i^\times}.
\]

At the lower boundary, $\Gamma_i(\underline\theta_i)=\tilde\lambda_i\underline\theta_i$ and
$\Delta\Gamma_i(\underline\theta_i)=\Delta\tilde\lambda_i\,\underline\theta_i$, so
\[
\mathcal G_i(\underline\theta_i)
=
\tilde\lambda_i\underline\theta_i+\beta_i^\times J_i(\underline\theta_i),
\qquad
\Delta\mathcal G_i(\underline\theta_i)
=
\Delta\tilde\lambda_i\,\underline\theta_i+\Delta\beta_i^\times J_i(\underline\theta_i),
\]
hence
\[
R_{\mathcal G}(\underline\theta_i)
=
\frac{\Delta\tilde\lambda_i\,\underline\theta_i+\Delta\beta_i^\times J_i(\underline\theta_i)}
{\tilde\lambda_i\,\underline\theta_i+\beta_i^\times J_i(\underline\theta_i)}.
\]

Now impose $R_{\mathcal G}(\underline\theta_i)=R_{\mathcal G}(\overline\theta_i)$. Cross-multiplying yields
\[
\left(\Delta\tilde\lambda_i\,\underline\theta_i+\Delta\beta_i^\times J_i(\underline\theta_i)\right)
(\tilde\lambda_i+\beta_i^\times)
=
(\Delta\tilde\lambda_i+\Delta\beta_i^\times)
\left(\tilde\lambda_i\,\underline\theta_i+\beta_i^\times J_i(\underline\theta_i)\right).
\]
After simplification,
\[
\left(\underline\theta_i-J_i(\underline\theta_i)\right)
\left(\beta_i^\times\Delta\tilde\lambda_i-\tilde\lambda_i\Delta\beta_i^\times\right)=0.
\]
Since $\underline\theta_i-J_i(\underline\theta_i)=\gamma_i(\underline\theta_i)>0$, this is equivalent to
\[
\beta_i^\times\Delta\tilde\lambda_i=\tilde\lambda_i\Delta\beta_i^\times,
\]
or, equivalently,
\[
\frac{\Delta\beta_i^\times}{\beta_i^\times}
=
\frac{\Delta\tilde\lambda_i}{\tilde\lambda_i}.
\]
Thus, for $\beta_i^\times>0$, boundary equality is not generic.

Finally, for every $s\in T_i^\times$,
\[
\partial_\theta\Delta q_i^\times(\theta,s)
=
\partial_\theta R_q(\theta,s)
\left(R_{\mathcal G}(\theta)-\E_i^w\!\left[R_{\mathcal G}\mid s\right]\right)
+
R_q(\theta,s)\,R_{\mathcal G}'(\theta).
\]
When $\beta_i^\times=0$, the equality of the boundary values of $R_{\mathcal G}$ makes the first term comparable across endpoints. When $\beta_i^\times>0$, this common boundary normalization is lost: the first term already differs across endpoints even before considering $R_{\mathcal G}'$. Therefore endpoint derivatives alone cannot deliver a strict monotonicity or non-monotonicity conclusion without restrictions on the equilibrium object.
\end{proof}

\subsection{Proof of Lemma FOC With Ironing}

\begin{proof}
    
Assume first that the capacity is not binding, then $\varepsilon_i^\times(\theta) = 0$ and the optimal allocation is given by $q^\times_i$ solve pointwise:

\[u(q^\times_i,s)=0 \tag{FOC-Smd}\label{FOC-Smd}\]

The assumption on the form of $u$ ensures that a solution exists for any $s$. Note also that the FOC ensures that when capacity is slack, the allocation is independent of the type and therefore satisfies the IC constraint. Implicit differentiation of \eqref{FOC-Smd} with respect to $s$ gives

\[ \frac{\partial q^\times_i}{\partial s} = -\frac{u_s}{u_q}.\]

Since $u_q\le 0$ and $u_s\ge 0$, then so $q^\times_i(\theta,s)$ is non-decreasing in $s$ for every $\theta$ when the capacity is slack. Let

\[Q_i(s):=\mathbb{E}_{\theta_i}[q^\times_i(\theta,s)]\]

Denote aggregate allocation in state $s$. Under the above monotonicity and mild continuity assumptions in $s$, $Q(s)$ is non-decreasing and continuous. For a given capacity $k$, there exists a unique threshold $s^\times_i\equiv s^\times_i(k)$ such that the capacity constraint is slack for $s<s^\times_i$ and binding for $s\ge s^\times_i$. Moreover, note that implicit differentiation also yields 

\[\frac{\partial s^\times}{\partial k} =\frac{1}{\E_{\theta}[\partial q^\times_i/\partial s]}>0\]

Therefore, there exists a $k_{\max}$ such that $s^\times(k_{\max})=\overline{s}$.

The market designer solves for every $s\ge s^\times_i$
\begin{align*}
\max_{\substack{\\q^\times_i(\theta,s)}}  \quad \quad &  \mathbb{E}_{\theta} \left[\; \;U(q^\times_i(\theta,s),s)\Gamma_i(\theta)\;\right]  \\
 \text{s.t.} \quad \quad 
& \mathbb{E}_{\theta}\left[ q^\times_i(\theta,s) \right] \;=\; k. \\
& q^\times_i(\theta,s)\ge q^\times_i(\theta',s) \quad \quad \forall \theta, \theta' \in \Theta: \theta\ge\theta'
\end{align*}

The stationary condition of the associated Lagrangian is:

\[
 u(q^\times_i,s) \; \G_i- \varepsilon_i^{\times}(s) =0
\]

Where $\G_i (\theta) := \Gamma_i(\theta)+J_i(\theta)\;\beta^\times_i$. And the second order derivatives yields 

\[
u\!\left(q_i^\times(\theta,s),s\right)  \G_i(\theta) = 0
\]

Which ensures the strict concavity of the objective function as  $u_q(\cdot)<0$. Implicit differentiation of the stationary condition with respect to $\theta$ yields

\[
\frac{\partial q_i^\times}{\partial \theta} =-  \frac{u\!\left(q_i^\times(\theta,s),s\right) }{u_q\!\left(q_i^\times(\theta,s),s\right)}\frac{\G_i' (\theta)}{\G_i (\theta)} 
\]

Recall that $u(\cdot)>0$, $u_q(\cdot)<0$ and from assumption XXXX, we have $\G_i (\theta)\ge0$, then the IC constraint might not be satisfied whenever $\G_i' (\theta)<0$. Note the expression for a unique type, the objective can be rewritten as follows: 

\[U(q^\times_i(\theta,s),s)\overline{\G}_i (\theta) +U(q^\times_i(\theta,s),s)(  \G_i (\theta) - \overline{\G}_i (\theta)) \]

Let $x:=G_i(\theta)\in[0,1]$ and define $\tilde U(x):=U(q^\times_i(G_i^{-1}(x),s),s)$.
Recall $\Phi_i(x):=\int_0^x \G_i (\theta)(G_i^{-1}(t))\,dt$ and $\overline{\G}_i (\theta):=(\co\Phi_i)'\!\big(G_i(\theta)\big)$, so that
$\G_i (\theta)(G_i^{-1}(x))=\Phi_i'(x)$ and $\overline{\G}_i (\theta)(G_i^{-1}(x))=(\co\Phi_i)'(x)$. Expressing the second term in quantiles  and aggregating over the types yields 
\[
\int_\Theta U(q^\times_i(\theta,s),s)\big(\G_i (\theta)-\overline{\G}_i (\theta)\big)\,dG_i(\theta)
=
\int_0^1 \tilde U(x)\Big(\Phi_i'(x)-(\co\Phi_i)'(x)\Big)\,dx.
\]

Integration by parts yields
\[
\int_0^1 \tilde U(x)\Big(\Phi_i'(x)-(\co\Phi_i)'(x)\Big)\,dx
=
\Big[\tilde U(x)\big(\Phi_i(x)-\co\Phi_i(x)\big)\Big]_0^1
-\int_0^1 \big(\Phi_i(x)-\co\Phi_i(x)\big)\,d\tilde U(x).
\]
Since $\Phi_i(0)=\co\Phi_i(0)$ and $\Phi_i(1)=\co\Phi_i(1)$, the boundary term is $0$, hence
\[
\int_\Theta U(q^\times_i(\theta,s),s)\big(\G_i (\theta)-\overline{\G}_i (\theta)\big)\,dG_i(\theta)
=
-\int_0^1 \big(\Phi_i(x)-\co\Phi_i(x)\big)\,d\tilde U(x).
\]
Therefore the objective equals
\[
\int_\Theta U(q^\times_i(\theta,s),s)\,\overline{\G}_i (\theta)\,dG_i(\theta)
-\int_0^1 \big(\Phi_i(x)-\co\Phi_i(x)\big)\,d\tilde U(x).
\]

Since $\Phi_i(x)\ge \co \Phi_i(x)$ by construction of the convex envelope and incentive compatibility implies that $\tilde{U}(\cdot)$ is nondecreasing, the second term is weakly nonpositive. Therefore, the designer's payoff is bounded above by the first term. The stationary condition of the associated Lagrangian is thus:

\[
u\!\left(q_i^\times(\theta,s),s\right) \overline{\G}_i (\theta) - \varepsilon_i^\times(s) = 0
\]

Moreover, at an optimum the second term can be driven to zero: on any region where $\Phi_i(\theta) > \co \Phi_i(\theta)$ the convex envelope $\co \Phi_i(\theta)$ is linear, so $\overline{\G}_i (\theta)$ is locally constant; optimality then entails pooling on that region (i.e., $q^\times_i(\theta,s)$ locally constant), which implies $\frac{\partial U(q^\times_i(\theta,s),s)}{\partial \theta}=0$ there and thus the second term is null.

\end{proof}



\subsection{Proof Boundary invariance of the preference ratio}

\begin{lemma}[Boundary invariance of the preference ratio under $\beta_i^\times=0$]\label{lem:boundary}
Suppose $\beta_i^\times=0$, so that $\mathcal{G}_i(\theta)=\Gamma_i(\theta)=\tilde\lambda_i J_i(\theta)+\Lambda_i(\theta)$. For any admissible perturbation $\Delta(\cdot)$ of $\lambda_i(\cdot)$, define $R_\Gamma(\theta):=\Delta\Gamma_i(\theta)/\Gamma_i(\theta)$. Then
\[R_\Gamma(\underline\theta_i)=R_\Gamma(\overline\theta_i)=\frac{\Delta\tilde\lambda_i}{\tilde\lambda_i}.\]
\end{lemma}

\begin{proof}
Since $J_i$ depends only on $G_i$ and $g_i$, which are fixed primitives, any perturbation of $\lambda_i(\cdot)$ yields
\[\Delta\Gamma_i(\theta)=\Delta\tilde\lambda_i\,J_i(\theta)+\Delta\Lambda_i(\theta),\]
where, from the explicit form of $\Lambda_i$,
\[\Delta\Lambda_i(\theta)=\frac{\Delta\tilde\lambda_i-\int_{\underline\theta_i}^{\theta}\Delta\lambda_i(t)\,\mathrm{d}t}{g_i(\theta)}.\]
\textbf{Lower boundary.} Since $\int_{\underline\theta_i}^{\underline\theta_i}\Delta\lambda_i=0$, we have $\Delta\Lambda_i(\underline\theta_i)=\Delta\tilde\lambda_i/g_i(\underline\theta_i)$. Using $J_i(\underline\theta_i)=\underline\theta_i-1/g_i(\underline\theta_i)$ and $\Gamma_i(\underline\theta_i)=\tilde\lambda_i\underline\theta_i$,
\[\Delta\Gamma_i(\underline\theta_i)=\Delta\tilde\lambda_i\!\left(\underline\theta_i-\frac{1}{g_i(\underline\theta_i)}\right)+\frac{\Delta\tilde\lambda_i}{g_i(\underline\theta_i)}=\Delta\tilde\lambda_i\,\underline\theta_i,\]
hence $R_\Gamma(\underline\theta_i)=\Delta\tilde\lambda_i\underline\theta_i\,/\,\tilde\lambda_i\underline\theta_i=\Delta\tilde\lambda_i/\tilde\lambda_i$.
\textbf{Upper boundary.} Since $\int_{\underline\theta_i}^{\overline\theta_i}\Delta\lambda_i=\Delta\tilde\lambda_i$, we have $\Delta\Lambda_i(\overline\theta_i)=0$. Using $J_i(\overline\theta_i)=\overline\theta_i$ and $\Gamma_i(\overline\theta_i)=\tilde\lambda_i\overline\theta_i$,
\[\Delta\Gamma_i(\overline\theta_i)=\Delta\tilde\lambda_i\,\overline\theta_i,\]
hence $R_\Gamma(\overline\theta_i)=\Delta\tilde\lambda_i/\tilde\lambda_i$.\end{proof}



% ============================================================
% Preference perturbation analysis (Delta) -- fixed k_i, general beta_i^times
% ============================================================

Recall the (pointwise) FOC for each $(\theta,s)$:
\begin{equation}\label{KKT-qtimes}\tag{KKT-$q^\times$}
u\!\left(q_i^\times(\theta,s),s\right)\,\mathcal{G}_i(\theta)-\varepsilon_i^\times(s)=0,
\end{equation}
where $\mathcal{G}_i(\theta)=\Gamma_i(\theta)+\beta_i^\times J_i(\theta)=(\tilde\lambda_i+\beta_i^\times)J_i(\theta)+\Lambda_i(\theta)$, with
\[J_i(\theta)=\theta-\frac{1-G_i(\theta)}{g_i(\theta)},\qquad \Lambda_i(\theta)=\frac{\tilde\lambda_i-\int_{\underline\theta_i}^{\theta}\lambda_i(t)\,\mathrm{d}t}{g_i(\theta)},\qquad \tilde\lambda_i=\int_{\underline\theta_i}^{\overline\theta_i}\lambda_i(t)\,\mathrm{d}t.\]
Since $J_i$ is invariant to perturbations of $\lambda_i(\cdot)$, a preference perturbation $\Delta(\cdot)$ acting on $\lambda_i$ produces
\begin{equation}\label{DeltaG}\tag{$\Delta\mathcal{G}$}
\Delta\mathcal{G}_i(\theta)=\Delta\Gamma_i(\theta)+J_i(\theta)\,\Delta\beta_i^\times=(\Delta\tilde\lambda_i+\Delta\beta_i^\times)J_i(\theta)+\Delta\Lambda_i(\theta),
\end{equation}
where $\Delta\Lambda_i(\theta)=\bigl(\Delta\tilde\lambda_i-\int_{\underline\theta_i}^{\theta}\Delta\lambda_i(t)\,\mathrm{d}t\bigr)/g_i(\theta)$ and $\Delta\tilde\lambda_i=\int_{\underline\theta_i}^{\overline\theta_i}\Delta\lambda_i(t)\,\mathrm{d}t$. Note that when $\beta_i^\times=0$ and $\Delta\beta_i^\times=0$, \eqref{DeltaG} reduces to $\Delta\mathcal{G}_i=\Delta\Gamma_i$.

\textbf{Upper boundary.} Since $J_i(\overline\theta_i)=\overline\theta_i$ and $\Lambda_i(\overline\theta_i)=0$, the boundary values are
\[\mathcal{G}_i(\overline\theta_i)=(\tilde\lambda_i+\beta_i^\times)\overline\theta_i,\qquad \Delta\mathcal{G}_i(\overline\theta_i)=(\Delta\tilde\lambda_i+\Delta\beta_i^\times)\overline\theta_i,\]
so $R_{\mathcal{G}}(\overline\theta_i):=\Delta\mathcal{G}_i(\overline\theta_i)/\mathcal{G}_i(\overline\theta_i)=(\Delta\tilde\lambda_i+\Delta\beta_i^\times)/(\tilde\lambda_i+\beta_i^\times)=:\Delta\mu_i/\mu_i$, where $\mu_i:=\tilde\lambda_i+\beta_i^\times$.

\end{proof}

\subsection{Proof of concavity of the outer problem}


\begin{proposition}[Concavity of the inner value in $(k_i,I_i)$ without sign restrictions on $J_i$]
Fix a category $i$. Let $\mathcal M_i(k_i,I_i)$ be the set of (possibly randomized) direct mechanisms
$\left(q_i(\theta,s),t_i(\theta,s)\right)$ such that, for every report $\hat\theta$ and true type $\theta$,
\[
\E_s\!\left[\theta\,U\!\left(q_i(\theta,s),s\right)-t_i(\theta,s)\right]
\;\ge\;
\E_s\!\left[\theta\,U\!\left(q_i(\hat\theta,s),s\right)-t_i(\hat\theta,s)\right]
\qquad\text{(IC)}
\]
and
\[
\E_s\!\left[\theta\,U\!\left(q_i(\theta,s),s\right)-t_i(\theta,s)\right]\ge 0
\qquad\text{(IR)},
\]
together with the resource constraints
\[
\E_{\theta_i}\!\left[q_i(\theta,s)\right]\le k_i \qquad \forall s,
\qquad
\E_{(s,\theta_i)}\!\left[t_i(\theta,s)\right]\ge I_i.
\]
Define the value function
\[
\widetilde V_i(k_i,I_i)
:=
\sup_{(q_i,t_i)\in\mathcal M_i(k_i,I_i)}
\E_{(s,\theta_i)}\!\left[\Gamma_i(\theta)\,U\!\left(q_i(\theta,s),s\right)\right].
\]
Then $\widetilde V_i$ is concave in $(k_i,I_i)$. Consequently,
\[
V_i(k_i,I_i)
=
\widetilde V_i(k_i,I_i)-\tilde\lambda_i I_i
\]
is concave in $(k_i,I_i)$.
\end{proposition}
\begin{proof}
Take two parameter pairs $(k_i^0,I_i^0)$ and $(k_i^1,I_i^1)$, and pick
$\left(q_i^0,t_i^0\right)\in\mathcal M_i(k_i^0,I_i^0)$ and
$\left(q_i^1,t_i^1\right)\in\mathcal M_i(k_i^1,I_i^1)$.
Fix $\alpha\in[0,1]$ and define $(k_i^\alpha,I_i^\alpha):=\left(\alpha k_i^0+(1-\alpha)k_i^1,\ \alpha I_i^0+(1-\alpha)I_i^1\right)$.
Construct the mixed mechanism $M^\alpha$ as follows: with probability $\alpha$ (independent of reports and types) run mechanism $M^0$, and with probability $1-\alpha$ run mechanism $M^1$.
Equivalently, $M^\alpha$ induces lotteries over $\left(q_i^0,t_i^0\right)$ and $\left(q_i^1,t_i^1\right)$.
\textbf{Feasibility.}
Linearity of expectations implies, for every $s$,
\[
\E_{\theta_i}\!\left[q_i^\alpha(\theta,s)\right]
=
\alpha\,\E_{\theta_i}\!\left[q_i^0(\theta,s)\right]+(1-\alpha)\,\E_{\theta_i}\!\left[q_i^1(\theta,s)\right]
\le \alpha k_i^0+(1-\alpha)k_i^1
= k_i^\alpha,
\]
and
\[
\E_{(s,\theta_i)}\!\left[t_i^\alpha(\theta,s)\right]
=
\alpha\,\E_{(s,\theta_i)}\!\left[t_i^0(\theta,s)\right]+(1-\alpha)\,\E_{(s,\theta_i)}\!\left[t_i^1(\theta,s)\right]
\ge \alpha I_i^0+(1-\alpha)I_i^1
= I_i^\alpha.
\]
\textbf{IC and IR.}
For any $\theta,\hat\theta$, the truthful interim utility under $M^\alpha$ is
\[
\alpha\,\E_s\!\left[\theta U\!\left(q_i^0(\theta,s),s\right)-t_i^0(\theta,s)\right]
+(1-\alpha)\,\E_s\!\left[\theta U\!\left(q_i^1(\theta,s),s\right)-t_i^1(\theta,s)\right],
\]
and the deviation utility is the same convex combination with $\theta$ replaced by $\hat\theta$ inside $(q_i^\ell,t_i^\ell)$.
Since $M^0$ and $M^1$ are IC (and IR), the convex combination preserves the inequalities, hence $M^\alpha$ is IC and IR. Therefore $M^\alpha\in\mathcal M_i(k_i^\alpha,I_i^\alpha)$.
\textbf{Objective.}
The designer’s objective under $M^\alpha$ is linear in the mixing probability:
\[
\E_{(s,\theta_i)}\!\left[\Gamma_i(\theta)\,U\!\left(q_i^\alpha(\theta,s),s\right)\right]
=
\alpha\,\E_{(s,\theta_i)}\!\left[\Gamma_i(\theta)\,U\!\left(q_i^0(\theta,s),s\right)\right]
+(1-\alpha)\,\E_{(s,\theta_i)}\!\left[\Gamma_i(\theta)\,U\!\left(q_i^1(\theta,s),s\right)\right].
\]
Taking suprema over feasible mechanisms yields
\[
\widetilde V_i(k_i^\alpha,I_i^\alpha)
\ge
\alpha\,\widetilde V_i(k_i^0,I_i^0)+(1-\alpha)\,\widetilde V_i(k_i^1,I_i^1),
\]
so $\widetilde V_i$ is concave. Finally, $V_i=\widetilde V_i-\tilde\lambda_i I_i$ is concave because subtracting a linear function preserves concavity.
\end{proof}


\subsection{Pertubartion IR constrained}

% ============================================================
% Preference perturbation analysis (Delta) -- fixed k_i
% Key difference: \Delta \mathcal G_i(\theta)=\Delta \Gamma_i(\theta)+J_i(\theta)\Delta\beta_i^\times
% ============================================================

Recall the (pointwise) FOC for each $(\theta,s)$:
\begin{equation}
\label{KKT-qtimes}\tag{KKT-$q^\times$}
u\!\left(q_i^\times(\theta,s),s\right)\,\mathcal G_i(\theta)-\varepsilon_i^\times(s)=0,
\end{equation}
where $\mathcal G_i(\theta)=\Gamma_i(\theta)+\beta_i^\times J_i(\theta)$.

Consider a preference perturbation $\Delta(\cdot)$ acting on $\Gamma_i$, so that
\begin{equation}
\label{DeltaG}\tag{$\Delta \mathcal G$}
\Delta \mathcal G_i(\theta)=\Delta\Gamma_i(\theta)+J_i(\theta)\,\Delta\beta_i^\times.
\end{equation}
Linearizing \eqref{KKT-qtimes} yields, for each $(\theta,s)$,
\begin{equation}
\label{DeltaKKT}\tag{$\Delta$KKT}
u_q\!\left(q_i^\times(\theta,s),s\right)\,\Delta q_i^\times(\theta,s)\,\mathcal G_i(\theta)
+u\!\left(q_i^\times(\theta,s),s\right)\,\Delta \mathcal G_i(\theta)
-\Delta\varepsilon_i^\times(s)=0,
\end{equation}
hence
\begin{equation}
\label{Deltaq-general}\tag{$\Delta q^\times$}
\Delta q_i^\times(\theta,s)=
\frac{\Delta\varepsilon_i^\times(s)-u\!\left(q_i^\times(\theta,s),s\right)\,\Delta \mathcal G_i(\theta)}
{\mathcal G_i(\theta)\,u_q\!\left(q_i^\times(\theta,s),s\right)}.
\end{equation}

Fix a state $s\in S^\times$ such that the capacity constraint is slack. Then $\varepsilon_i^\times(s)=0$, and \eqref{KKT-qtimes} implies
\begin{equation}
\label{offpeak-u0}\tag{$u=0$ off-peak}
u\!\left(q_i^\times(\theta,s),s\right)\,\mathcal G_i(\theta)=0.
\end{equation}
Under the maintained condition $\mathcal G_i(\theta)>0$, \eqref{offpeak-u0} gives $u\!\left(q_i^\times(\theta,s),s\right)=0$ for all $\theta$, so that
\begin{equation}
\label{Deltaq-offpeak}\tag{$\Delta q=0$ off-peak}
\Delta q_i^\times(\theta,s)=0
\qquad \forall \theta,\ \forall s\in S^\times,
\end{equation}
locally (i.e.\ as long as $s$ remains slack under the perturbation).

Fix $s\in T^\times$. Since $k_i$ is held fixed, the (within-$i$) capacity constraint implies
\begin{equation}
\label{Cap-Delta}\tag{$\Delta$Cap}
0=\E_{\theta_i}\!\left[\Delta q_i^\times(\theta,s)\right].
\end{equation}
Define, for each $(\theta,s)$,
\begin{equation}
\label{w-def}\tag{$w^\times$}
w_i^\times(\theta,s):=-\frac{1}{\mathcal G_i(\theta)\,u_q\!\left(q_i^\times(\theta,s),s\right)} \;>\;0,
\qquad
\overline w_i^\times(s):=\E_{\theta_i}\!\left[w_i^\times(\theta,s)\right],
\end{equation}
and for any integrable $X(\theta,s)$ define
\begin{equation}
\label{Ew-def}\tag{$\E_w$}
\E_w\!\left[X\mid s\right]:=\frac{1}{\overline w_i^\times(s)}\E_{\theta_i}\!\left[w_i^\times(\theta,s)\,X(\theta,s)\right].
\end{equation}
Using \eqref{Deltaq-general} and $w_i^\times=-1/(\mathcal G_i u_q)$, for $s\in T^\times$ we can write
\begin{equation}
\label{Deltaq-peak}\tag{$\Delta q^\times$ peak}
\Delta q_i^\times(\theta,s)
=-w_i^\times(\theta,s)\,\Delta\varepsilon_i^\times(s)
+w_i^\times(\theta,s)\,u\!\left(q_i^\times(\theta,s),s\right)\,\Delta \mathcal G_i(\theta).
\end{equation}
Taking $\E_{\theta_i}[\cdot]$ in \eqref{Deltaq-peak} and imposing \eqref{Cap-Delta} yields, for each $s\in T^\times$,
\begin{equation}
\label{Deltaeps}\tag{$\Delta\varepsilon^\times$}
\Delta\varepsilon_i^\times(s)
=\E_w\!\left[u\,\Delta \mathcal G\mid s\right]
=\E_w\!\left[u\,\Delta\Gamma\mid s\right]
+\Delta\beta_i^\times\,\E_w\!\left[uJ\mid s\right],
\end{equation}
where $uJ$ is shorthand for $u\!\left(q_i^\times(\theta,s),s\right)J_i(\theta)$.

Assume the within-category budget/IR constraint binds:
\begin{equation}
\label{budget-bind}\tag{B}
\E_{s,\theta_i}\!\left[U\!\left(q_i^\times(\theta,s),s\right)J_i(\theta)\right]=I_i.
\end{equation}
Taking $\Delta$ in \eqref{budget-bind} gives
\begin{equation}
\label{B-Delta}\tag{$\Delta$B}
0=\E_{s,\theta_i}\!\left[u\!\left(q_i^\times(\theta,s),s\right)J_i(\theta)\,\Delta q_i^\times(\theta,s)\right].
\end{equation}
Since $\Delta q_i^\times(\theta,s)=0$ for $s\in S^\times$, \eqref{B-Delta} reduces to
\begin{equation}
\label{B-Delta-peak}\tag{$\Delta$B, peak}
0=\E_s\!\left[\E_{\theta_i}\!\left[uJ\,\Delta q_i^\times(\theta,s)\right]\mathbf 1_{\{s\in T^\times\}}\right].
\end{equation}
Fix $s\in T^\times$. Substitute \eqref{Deltaq-peak} into $\E_{\theta_i}[uJ\,\Delta q]$:
\begin{align}
\label{ujDq1}\tag{$uJ\cdot \Delta q$}
\E_{\theta_i}\!\left[uJ\,\Delta q_i^\times(\theta,s)\right]
&=
-\Delta\varepsilon_i^\times(s)\,\E_{\theta_i}\!\left[w_i^\times(\theta,s)\,uJ\right]
+\E_{\theta_i}\!\left[w_i^\times(\theta,s)\,(uJ)\,u\,\Delta \mathcal G_i(\theta)\right].
\end{align}
Using \eqref{Ew-def}, $\E_{\theta_i}[w\,uJ]=\overline w\,\E_w[uJ\mid s]$ and
$\E_{\theta_i}[w\,(uJ)\,u\Delta\mathcal G]=\overline w\,\E_w[(uJ)\,u\Delta\mathcal G\mid s]$, so
\begin{equation}
\label{ujDq2}\tag{$uJ\cdot \Delta q$ (weighted)}
\E_{\theta_i}\!\left[uJ\,\Delta q_i^\times(\theta,s)\right]
=
\overline w_i^\times(s)\left(
\E_w\!\left[(uJ)\,u\,\Delta \mathcal G\mid s\right]
-\E_w\!\left[uJ\mid s\right]\E_w\!\left[u\,\Delta \mathcal G\mid s\right]
\right).
\end{equation}
That is,
\begin{equation}
\label{ujDq-cov}\tag{$uJ\cdot \Delta q$ (covariance)}
\E_{\theta_i}\!\left[uJ\,\Delta q_i^\times(\theta,s)\right]
=
\overline w_i^\times(s)\,\Cov_w\!\left(uJ,\;u\,\Delta \mathcal G\mid s\right),
\end{equation}
where $\Cov_w(X,Y\mid s):=\E_w[XY\mid s]-\E_w[X\mid s]\E_w[Y\mid s]$.

Since $\Delta \mathcal G=\Delta\Gamma+J\,\Delta\beta^\times$, we have
\begin{equation}
\label{cov-split}\tag{cov split}
\Cov_w\!\left(uJ,\;u\,\Delta \mathcal G\mid s\right)
=
\Cov_w\!\left(uJ,\;u\,\Delta\Gamma\mid s\right)
+\Delta\beta_i^\times\,Var_w\!\left(uJ\mid s\right),
\end{equation}
with $Var_w(uJ\mid s):=\Cov_w(uJ,uJ\mid s)$. Plugging \eqref{ujDq-cov}--\eqref{cov-split} into \eqref{B-Delta-peak} yields
\begin{equation}
\label{DeltaBeta-eq}\tag{$\Delta\beta$ equation}
0=
\E_s\!\left[\overline w_i^\times(s)\left(
\Cov_w\!\left(uJ,\;u\,\Delta\Gamma\mid s\right)
+\Delta\beta_i^\times\,Var_w\!\left(uJ\mid s\right)
\right)\mathbf 1_{\{s\in T^\times\}}\right].
\end{equation}
Solving gives
\begin{equation}
\label{DeltaBeta-sol}\tag{$\Delta\beta^\times$}
\Delta\beta_i^\times
=
-\frac{\E_s\!\left[\overline w_i^\times(s)\,\Cov_w\!\left(uJ,\;u\,\Delta\Gamma\mid s\right)\mathbf 1_{\{s\in T^\times\}}\right]}
{\E_s\!\left[\overline w_i^\times(s)\,Var_w\!\left(uJ\mid s\right)\mathbf 1_{\{s\in T^\times\}}\right]}.
\end{equation}

Finally, combining \eqref{Deltaq-peak} and \eqref{Deltaeps}, for each $s\in T^\times$,
\begin{equation}
\label{Deltaq-final}\tag{$\Delta q^\times$ closed form}
\Delta q_i^\times(\theta,s)
=
w_i^\times(\theta,s)\left(
u\!\left(q_i^\times(\theta,s),s\right)\left(\Delta\Gamma_i(\theta)+J_i(\theta)\Delta\beta_i^\times\right)
-\E_w\!\left[u\left(\Delta\Gamma+J\Delta\beta^\times\right)\mid s\right]
\right),
\end{equation}
and $\Delta q_i^\times(\theta,s)=0$ for all $s\in S^\times$.


% ============================================================
% Simplified preference-perturbation analysis WITH beta:
% \mathcal G(\theta)=\Gamma(\theta)+\beta^\times J(\theta),
% \Delta \mathcal G(\theta)=\Delta\Gamma(\theta)+J(\theta)\Delta\beta^\times

Fix a peak state $s\in T^\times$. The pointwise FOC is
\begin{equation}
\label{peakFOC-beta}\tag{FOC}
u\!\left(q_i^\times(\theta,s),s\right)\,\mathcal G_i(\theta)=\varepsilon_i^\times(s),
\qquad 
\mathcal G_i(\theta)=\Gamma_i(\theta)+\beta_i^\times J_i(\theta).
\end{equation}

\paragraph{Monotonicity.}
Differentiating \eqref{peakFOC-beta} w.r.t.\ $\theta$ gives
\[
u_q\!\left(q_i^\times(\theta,s),s\right)\,q_{\theta,i}^\times(\theta,s)\,\mathcal G_i(\theta)
+u\!\left(q_i^\times(\theta,s),s\right)\,\mathcal G_i'(\theta)=0,
\]
hence
\[
q_{\theta,i}^\times(\theta,s)
=
-\frac{u(q_i^\times,s)}{u_q(q_i^\times,s)}\,\frac{\mathcal G_i'(\theta)}{\mathcal G_i(\theta)}
=
R_q(\theta,s)\,\frac{\mathcal G_i'(\theta)}{\mathcal G_i(\theta)},
\qquad
R_q(\theta,s):=-\frac{u(q_i^\times,s)}{u_q(q_i^\times,s)}>0.
\]
Thus $q_{\theta,i}^\times(\theta,s)\ge 0$ if and only if $\mathcal G_i'(\theta)\ge 0$.

\paragraph{First-order perturbation and ratio form.}
Consider a preference perturbation $\Delta(\cdot)$ acting on $\Gamma_i$, with induced $\Delta\beta_i^\times$.
Then
\begin{equation}
\label{DeltaG-beta}\tag{$\Delta\mathcal G$}
\Delta\mathcal G_i(\theta)=\Delta\Gamma_i(\theta)+J_i(\theta)\,\Delta\beta_i^\times,
\qquad
R_{\mathcal G}(\theta):=\frac{\Delta\mathcal G_i(\theta)}{\mathcal G_i(\theta)}.
\end{equation}
Linearizing \eqref{peakFOC-beta} yields
\[
u_q(q_i^\times,s)\,\mathcal G_i(\theta)\,\Delta q_i^\times(\theta,s)
+u(q_i^\times,s)\,\Delta\mathcal G_i(\theta)
=\Delta\varepsilon_i^\times(s).
\]
Using \eqref{peakFOC-beta} to substitute $\varepsilon_i^\times(s)=u(q_i^\times,s)\mathcal G_i(\theta)$ gives
\begin{equation}
\label{Deltaq-ratio-beta}\tag{$\Delta q^\times$ ratio}
\Delta q_i^\times(\theta,s)
=
R_q(\theta,s)\left(
R_{\mathcal G}(\theta)-\frac{\Delta\varepsilon_i^\times(s)}{\varepsilon_i^\times(s)}
\right).
\end{equation}

\paragraph{Feasibility pins down $\Delta\varepsilon/\varepsilon$.}
Since $k_i$ is fixed, feasibility in state $s$ implies $\E_{\theta_i}[\Delta q_i^\times(\theta,s)]=0$.
Define the positive weight
\[
w_i^\times(\theta,s):=-\frac{1}{u_q(q_i^\times,s)\,\mathcal G_i(\theta)}\;>\;0,
\qquad
\overline w_i^\times(s):=\E_{\theta_i}[w_i^\times(\theta,s)],
\qquad
w_i^\times(\theta,s):=\frac{w_i^\times(\theta,s)}{\overline w_i^\times(s)},
\]
and the associated weighted expectation $\E_w[X\mid s]:=\E_{\theta_i}[w_i^\times(\theta,s)\,X(\theta,s)]$.
Then feasibility yields
\begin{equation}
\label{Deltaeps-ratio-beta}\tag{$\Delta\varepsilon/\varepsilon$}
\frac{\Delta\varepsilon_i^\times(s)}{\varepsilon_i^\times(s)}
=
\E_w\!\left[R_{\mathcal G}\mid s\right].
\end{equation}
Substituting \eqref{Deltaeps-ratio-beta} into \eqref{Deltaq-ratio-beta} gives the centered reallocation:
\begin{equation}
\label{Deltaq-centered-beta}\tag{$\Delta q^\times$ centered}
\Delta q_i^\times(\theta,s)
=
R_q(\theta,s)\left(
R_{\mathcal G}(\theta)-\E_w\!\left[R_{\mathcal G}\mid s\right]
\right),
\qquad s\in T^\times.
\end{equation}


\[\partial_\theta \Delta q_i^\times(\theta,s) = \partial_\theta R_q(\theta,s)\cdot\left(R_{\mathcal{G}}(\theta)-\mathbb{E}^w[R_{\mathcal{G}}|s]\right) + R_q(\theta,s)\cdot R_{\mathcal{G}}'(\theta).\]

\[\partial_\theta \Delta q_i^\times(\overline\theta_i,s) = \partial_\theta R_q(\overline\theta_i,s)\cdot\left(\frac{\Delta\mu_i}{\mu_i}-\mathbb{E}^w[R_{\mathcal{G}}|s]\right)+R_q(\overline\theta_i,s)\cdot R_{\mathcal{G}}'(\overline\theta_i).\]


:
\[\partial_\theta \Delta q_i^\times(\underline\theta_i,s) = \partial_\theta R_q(\underline\theta_i,s)\cdot\left(R_{\mathcal{G}}(\underline\theta_i)-\mathbb{E}^w[R_{\mathcal{G}}|s]\right)+R_q(\underline\theta_i,s)\cdot R_{\mathcal{G}}'(\underline\theta_i).\]


\textbf{Differentiating with respect to $\theta$.}

Since $\mathbb{E}^w[R_{\mathcal{G}}|s]$ is constant in $\theta$, the product rule gives:
\[\partial_\theta \Delta q_i^\times(\theta,s) = \partial_\theta R_q(\theta,s)\cdot\left(R_{\mathcal{G}}(\theta)-\mathbb{E}^w[R_{\mathcal{G}}|s]\right) + R_q(\theta,s)\cdot R_{\mathcal{G}}'(\theta).\]

\textbf{At the upper boundary.} Since $R_{\mathcal{G}}(\overline\theta_i)=\Delta\mu_i/\mu_i$:
\[\partial_\theta \Delta q_i^\times(\overline\theta_i,s) = \partial_\theta R_q(\overline\theta_i,s)\cdot\left(\frac{\Delta\mu_i}{\mu_i}-\mathbb{E}^w[R_{\mathcal{G}}|s]\right)+R_q(\overline\theta_i,s)\cdot R_{\mathcal{G}}'(\overline\theta_i).\]

\textbf{At the lower boundary.} Since $R_{\mathcal{G}}(\underline\theta_i)=(\Delta\tilde\lambda_i\cdot\underline\theta_i+\Delta\beta_i^\times\cdot J_i(\underline\theta_i))/(\tilde\lambda_i\cdot\underline\theta_i+\beta_i^\times\cdot J_i(\underline\theta_i))$:
\[\partial_\theta \Delta q_i^\times(\underline\theta_i,s) = \partial_\theta R_q(\underline\theta_i,s)\cdot\left(R_{\mathcal{G}}(\underline\theta_i)-\mathbb{E}^w[R_{\mathcal{G}}|s]\right)+R_q(\underline\theta_i,s)\cdot R_{\mathcal{G}}'(\underline\theta_i).\]

\textbf{When $\beta_i^\times=0$.} Lemma~\ref{lem:boundary} gives $R_{\mathcal{G}}(\underline\theta_i)=R_{\mathcal{G}}(\overline\theta_i)=\Delta\tilde\lambda_i/\tilde\lambda_i$, so the two endpoint derivatives become:
\[\partial_\theta \Delta q_i^\times(\overline\theta_i,s) = \partial_\theta R_q(\overline\theta_i,s)\cdot\left(\frac{\Delta\tilde\lambda_i}{\tilde\lambda_i}-\mathbb{E}^w[R_{\mathcal{G}}|s]\right)+R_q(\overline\theta_i,s)\cdot R_{\mathcal{G}}'(\overline\theta_i),\]
\[\partial_\theta \Delta q_i^\times(\underline\theta_i,s) = \partial_\theta R_q(\underline\theta_i,s)\cdot\left(\frac{\Delta\tilde\lambda_i}{\tilde\lambda_i}-\mathbb{E}^w[R_{\mathcal{G}}|s]\right)+R_q(\underline\theta_i,s)\cdot R_{\mathcal{G}}'(\underline\theta_i).\]

Since $\partial_\theta R_q(\theta,s)>0$ and $R_q(\theta,s)>0$ for all $\theta$, the sign of each endpoint derivative is determined entirely by the signs of $\Delta\tilde\lambda_i/\tilde\lambda_i - \mathbb{E}^w[R_{\mathcal{G}}|s]$ and $R_{\mathcal{G}}'$ respectively. The first terms are strictly positive multiples of the same scalar $\Delta\tilde\lambda_i/\tilde\lambda_i - \mathbb{E}^w[R_{\mathcal{G}}|s]$, hence share the same sign at both endpoints. Any difference in sign between the two endpoint derivatives can therefore only arise from the second terms $R_q\cdot R_{\mathcal{G}}'$, whose signs are determined by $R_{\mathcal{G}}'(\overline\theta_i)$ and $R_{\mathcal{G}}'(\underline\theta_i)$ respectively. Since these remain unrestricted, the derivative approach alone is insufficient to establish non-monotonicity, and the argument is better made directly on $\Delta q_i^\times$.

\textbf{Deriving $R_{\mathcal{G}}'(\theta)$.}

By the quotient rule applied to $R_{\mathcal{G}}(\theta)=\Delta\mathcal{G}_i(\theta)/\mathcal{G}_i(\theta)$:
\[R_{\mathcal{G}}'(\theta) = \frac{\Delta\mathcal{G}_i'(\theta)}{\mathcal{G}_i(\theta)} - R_{\mathcal{G}}(\theta)\cdot\frac{\mathcal{G}_i'(\theta)}{\mathcal{G}_i(\theta)}.\]

Since $\mathcal{G}_i(\theta)=\mu_i J_i(\theta)+\Lambda_i(\theta)$ and $\Delta\mathcal{G}_i(\theta)=\Delta\mu_i J_i(\theta)+\Delta\Lambda_i(\theta)$, the derivatives are:
\[\mathcal{G}_i'(\theta)=\mu_i J_i'(\theta)+\Lambda_i'(\theta), \qquad \Delta\mathcal{G}_i'(\theta)=\Delta\mu_i J_i'(\theta)+\Delta\Lambda_i'(\theta),\]
where $\Lambda_i'(\theta)=-\lambda_i(\theta)/g_i(\theta)-\Lambda_i(\theta)\cdot g_i'(\theta)/g_i(\theta)$ and $\Delta\Lambda_i'(\theta)=-\Delta\lambda_i(\theta)/g_i(\theta)-\Delta\Lambda_i(\theta)\cdot g_i'(\theta)/g_i(\theta)$.

\textbf{At the upper boundary.} Since $G_i(\overline\theta_i)=1$, we have $J_i'(\overline\theta_i)=2$ and $\Lambda_i(\overline\theta_i)=0$, so $\Lambda_i'(\overline\theta_i)=-\lambda_i(\overline\theta_i)/g_i(\overline\theta_i)$ and $\Delta\Lambda_i'(\overline\theta_i)=-\Delta\lambda_i(\overline\theta_i)/g_i(\overline\theta_i)$. Using $\mathcal{G}_i(\overline\theta_i)=\mu_i\overline\theta_i$ and $R_{\mathcal{G}}(\overline\theta_i)=\Delta\mu_i/\mu_i$:
\[R_{\mathcal{G}}'(\overline\theta_i) = \frac{2\Delta\mu_i - \Delta\lambda_i(\overline\theta_i)/g_i(\overline\theta_i)}{\mu_i\overline\theta_i} - \frac{\Delta\mu_i}{\mu_i}\cdot\frac{2\mu_i - \lambda_i(\overline\theta_i)/g_i(\overline\theta_i)}{\mu_i\overline\theta_i}.\]
Expanding and cancelling the $2\Delta\mu_i$ terms:
\[R_{\mathcal{G}}'(\overline\theta_i) = \frac{\lambda_i(\overline\theta_i)}{\mu_i g_i(\overline\theta_i)\overline\theta_i}\left(\frac{\Delta\mu_i}{\mu_i} - \frac{\Delta\lambda_i(\overline\theta_i)}{\lambda_i(\overline\theta_i)}\right).\]

\textbf{At the lower boundary ($\beta_i^\times=0$).} Since $G_i(\underline\theta_i)=0$, we have $\Lambda_i(\underline\theta_i)=\tilde\lambda_i/g_i(\underline\theta_i)$ and $\Delta\Lambda_i(\underline\theta_i)=\Delta\tilde\lambda_i/g_i(\underline\theta_i)$, and $\mathcal{G}_i(\underline\theta_i)=\tilde\lambda_i\underline\theta_i$. An analogous computation gives:
\[R_{\mathcal{G}}'(\underline\theta_i) = \frac{\lambda_i(\underline\theta_i)}{\tilde\lambda_i g_i(\underline\theta_i)\underline\theta_i}\left(\frac{\Delta\tilde\lambda_i}{\tilde\lambda_i} - \frac{\Delta\lambda_i(\underline\theta_i)}{\lambda_i(\underline\theta_i)}\right).\]

\textbf{Interpretation.} Both endpoint expressions for $R_{\mathcal{G}}'$ share the same structure: a strictly positive prefactor times the gap between the aggregate perturbation ratio $\Delta\mu_i/\mu_i$ (or $\Delta\tilde\lambda_i/\tilde\lambda_i$ when $\beta_i^\times=0$) and the local perturbation ratio $\Delta\lambda_i/\lambda_i$ evaluated at the boundary. The sign therefore depends on whether the perturbation is relatively stronger at the boundary point than on average. Since $\partial_\theta R_q>0$ and $R_q>0$, plugging back into the endpoint derivatives gives:
\[\partial_\theta \Delta q_i^\times(\overline\theta_i,s) = \partial_\theta R_q(\overline\theta_i,s)\cdot\left(\frac{\Delta\tilde\lambda_i}{\tilde\lambda_i}-\mathbb{E}^w[R_{\mathcal{G}}|s]\right)+R_q(\overline\theta_i,s)\cdot\frac{\lambda_i(\overline\theta_i)}{\mu_i g_i(\overline\theta_i)\overline\theta_i}\left(\frac{\Delta\mu_i}{\mu_i} - \frac{\Delta\lambda_i(\overline\theta_i)}{\lambda_i(\overline\theta_i)}\right),\]
\[\partial_\theta \Delta q_i^\times(\underline\theta_i,s) = \partial_\theta R_q(\underline\theta_i,s)\cdot\left(\frac{\Delta\tilde\lambda_i}{\tilde\lambda_i}-\mathbb{E}^w[R_{\mathcal{G}}|s]\right)+R_q(\underline\theta_i,s)\cdot\frac{\lambda_i(\underline\theta_i)}{\tilde\lambda_i g_i(\underline\theta_i)\underline\theta_i}\left(\frac{\Delta\tilde\lambda_i}{\tilde\lambda_i} - \frac{\Delta\lambda_i(\underline\theta_i)}{\lambda_i(\underline\theta_i)}\right).\]
The second terms are both unrestricted in sign, confirming that the derivative approach cannot deliver non-monotonicity without additional restrictions on the perturbation class.

\textbf{Upper boundary.} Since $J_i(\overline\theta_i)=\overline\theta_i$ and $\Lambda_i(\overline\theta_i)=0$, the boundary values are
\[\mathcal{G}_i(\overline\theta_i)=(\tilde\lambda_i+\beta_i^\times)\overline\theta_i,\qquad \Delta\mathcal{G}_i(\overline\theta_i)=(\Delta\tilde\lambda_i+\Delta\beta_i^\times)\overline\theta_i,\]
so $R_{\mathcal{G}}(\overline\theta_i):=\Delta\mathcal{G}_i(\overline\theta_i)/\mathcal{G}_i(\overline\theta_i)=(\Delta\tilde\lambda_i+\Delta\beta_i^\times)/(\tilde\lambda_i+\beta_i^\times)=:\Delta\mu_i/\mu_i$, where $\mu_i:=\tilde\lambda_i+\beta_i^\times$.

\textbf{Lower boundary.} Since $J_i(\underline\theta_i)=\underline\theta_i-1/g_i(\underline\theta_i)$, $\Lambda_i(\underline\theta_i)=\tilde\lambda_i/g_i(\underline\theta_i)$, and $\Delta\Lambda_i(\underline\theta_i)=\Delta\tilde\lambda_i/g_i(\underline\theta_i)$,
\[\mathcal{G}_i(\underline\theta_i)=\mu_i\underline\theta_i-\frac{\beta_i^\times}{g_i(\underline\theta_i)},\qquad \Delta\mathcal{G}_i(\underline\theta_i)=\Delta\mu_i\,\underline\theta_i-\frac{\Delta\beta_i^\times}{g_i(\underline\theta_i)}.\]
Boundary invariance $R_{\mathcal{G}}(\underline\theta_i)=\Delta\mu_i/\mu_i$ then requires $\Delta\beta_i^\times/\beta_i^\times=\Delta\tilde\lambda_i/\tilde\lambda_i$, which holds if $\beta_i^\times=0$ (so that the lower boundary reduces to $\mathcal{G}_i(\underline\theta_i)=\tilde\lambda_i\underline\theta_i$ as in Lemma~\ref{lem:boundary}) but is not guaranteed in general.


\[R_{\mathcal{G}}(\underline\theta_i) = \frac{\Delta\tilde\lambda_i\cdot\underline\theta_i+\Delta\beta_i^\times\cdot J_i(\underline\theta_i)}{\tilde\lambda_i\cdot\underline\theta_i+\beta_i^\times\cdot J_i(\underline\theta_i)},\qquad R_{\mathcal{G}}(\overline\theta_i) = \frac{\Delta\tilde\lambda_i\cdot\overline\theta_i+\Delta\beta_i^\times\cdot\overline\theta_i}{\tilde\lambda_i\cdot\overline\theta_i+\beta_i^\times\cdot\overline\theta_i} = \frac{\Delta\mu_i}{\mu_i}.\]

\section{Proof of Corollary Linear}

\begin{proof}
    
We start from the identity
\begin{equation}
\label{F-def}
F(s):=\E_{\theta_i}\!\left[w_i^\times(\theta,s)\,uJ\right]
=\E_{\theta_i}\!\left[-\frac{u\!\left(q_i^\times(\theta,s),s\right)}{u_q\!\left(q_i^\times(\theta,s),s\right)}\frac{J_i(\theta)}{\mathcal G_i(\theta)}\right]
=\E_{\theta_i}\!\left[R_q^\times(\theta,s)\,R_\theta(\theta)\right],
\end{equation}
where $R_q^\times(\theta,s):=-u(q_i^\times(\theta,s),s)/u_q(q_i^\times(\theta,s),s)$ and $R_\theta(\theta):=J_i(\theta)/\mathcal G_i(\theta)$.

Differentiating \eqref{F-def} with respect to $s$ gives
\begin{equation}
\label{F-deriv}
F'(s)=\E_{\theta_i}\!\left[\left(\partial_q R_q\!\left(q_i^\times(\theta,s),s\right)\,\partial_s q_i^\times(\theta,s)+\partial_s R_q\!\left(q_i^\times(\theta,s),s\right)\right)R_\theta(\theta)\right],
\end{equation}
since $R_\theta(\theta)$ does not depend on $s$.

Next, differentiate the FOC $u(q_i^\times(\theta,s),s)\mathcal G_i(\theta)-\varepsilon_i^\times(s)=0$ with respect to $s$:
\begin{equation}
\label{dqds}
\partial_s q_i^\times(\theta,s)=\frac{\partial_s\varepsilon_i^\times(s)/\mathcal G_i(\theta)-u_s\!\left(q_i^\times(\theta,s),s\right)}{u_q\!\left(q_i^\times(\theta,s),s\right)}.
\end{equation}
On-peak feasibility implies $\E_{\theta_i}[\partial_s q_i^\times(\theta,s)]=0$, hence
\begin{equation}
\label{epss}
\partial_s\varepsilon_i^\times(s)=
\frac{\E_{\theta_i}\!\left[u_s\!\left(q_i^\times(\theta,s),s\right)/u_q\!\left(q_i^\times(\theta,s),s\right)\right]}
{\E_{\theta_i}\!\left[1/\left(u_q\!\left(q_i^\times(\theta,s),s\right)\mathcal G_i(\theta)\right)\right]}
=-\frac{1}{\overline w_i^\times(s)}\E_{\theta_i}\!\left[\frac{u_s\!\left(q_i^\times(\theta,s),s\right)}{u_q\!\left(q_i^\times(\theta,s),s\right)}\right],
\end{equation}
where $\overline w_i^\times(s):=\E_{\theta_i}[w_i^\times(\theta,s)]$ and $w_i^\times(\theta,s):=-1/(\mathcal G_i(\theta)u_q(q_i^\times(\theta,s),s))$.

Now impose the parametric assumptions $u(q,s)=v(q)+\zeta(s)$ with $\zeta_{ss}=0$ and $v''(q)=0$ (so $u_{qs}=0$ and $u_q$ is constant). Then
\begin{equation}
\label{Rq-partials}
\partial_q R_q(q,s)=-1
\qquad\text{and}\qquad
\partial_s R_q(q,s)=-\frac{u_s(q,s)}{u_q(q,s)}.
\end{equation}
Moreover, from \eqref{dqds} we have $\partial_s q_i^\times(\theta,s)=\partial_s\varepsilon_i^\times(s)/(u_q\mathcal G_i(\theta))-\;u_s/u_q$, so the bracket in \eqref{F-deriv} simplifies to
\[
\partial_q R_q\,\partial_s q_i^\times+\partial_s R_q
=-\partial_s q_i^\times-\frac{u_s}{u_q}
=-\frac{\partial_s\varepsilon_i^\times(s)}{u_q\,\mathcal G_i(\theta)}.
\]
Substituting into \eqref{F-deriv} yields
\begin{equation}
\label{Fprime-final}
F'(s)
=
-\frac{\partial_s\varepsilon_i^\times(s)}{u_q}\,
\E_{\theta_i}\!\left[\frac{R_\theta(\theta)}{\mathcal G_i(\theta)}\right]
=
-\frac{\partial_s\varepsilon_i^\times(s)}{u_q}\,
\E_{\theta_i}\!\left[\frac{J_i(\theta)}{\mathcal G_i(\theta)^2}\right].
\end{equation}
Under $u(q,s)=v(q)+\zeta(s)$ with $v''=0$ and $\zeta_{ss}=0$, both $u_q$ and $u_s$ are constants, hence \eqref{epss} implies $\partial_s\varepsilon_i^\times(s)$ is constant in $s$, so \eqref{Fprime-final} shows that $F'(s)$ is constant in $s$. Therefore $F(s)$ is monotone in $s$ (linear), with sign determined by $\E_{\theta_i}[J_i(\theta)/\mathcal G_i(\theta)^2]$ (and the sign of $u_s$ through $\partial_s\varepsilon_i^\times$).

\end{proof}
